92/// Interns the Store- and Load field-strings
93class FieldStringManager {
94public:
95 FieldStringManager();
96
97 [[nodiscard]] FieldStringNodeId intern(FieldStringNode Nod) {
98 auto [Id, Inserted] = NodeCompressor.insert(Nod);
99
100 if (Inserted) {
101 Depth.push_back(Depth[Nod.Next] + 1);
102 }
103
104 return Id;
105 }
106
107 [[nodiscard]] FieldStringNodeId prepend(int32_t Head,
108 FieldStringNodeId Tail) {
109 auto Ret = intern(FieldStringNode{.Next = Tail, .Offset = Head});
110 PHASAR_LOG_LEVEL(DEBUG, "[prepend]: " << Head << " :: #" << uint32_t(Tail)
111 << " = #" << uint32_t(Ret));
112 return Ret;
113 }
114
115 [[nodiscard]] FieldStringNode operator[](FieldStringNodeId NId) const {
116 return NodeCompressor[NId];
117 }
118
119 [[nodiscard]] llvm::SmallVector<int32_t>
120 getFullFieldString(FieldStringNodeId NId) const;
121
122 [[nodiscard]] FieldStringNodeId
123 fromFullFieldString(llvm::ArrayRef<int32_t> FieldString);
124
125 [[nodiscard]] uint32_t depth(FieldStringNodeId NId) const {
126 return Depth[NId];
127 }
128
129private:
130 Compressor<FieldStringNode, FieldStringNodeId> NodeCompressor{};
131 TypedVector<FieldStringNodeId, uint32_t> Depth{};

134/// A single CFL Field-Access String consisting of: gep, loads, kills, and
135/// stores
136struct AccessPath {
137 static constexpr int32_t TopOffset = INT32_MIN;
138
139 FieldStringNodeId Loads{};
140 FieldStringNodeId Stores{};
141 llvm::SmallDenseSet<int32_t, 2> Kills{};
142 // Add an offset for pending GEPs; INT32_MIN is Top
143 int32_t Offset = {0};
144 int32_t EmptyTombstone = 0;
145
146 [[nodiscard]] bool empty() const noexcept {
147 return Loads == FieldStringNodeId::None &&
148 Stores == FieldStringNodeId::None && Kills.empty() && Offset == 0;
149 }
150
151 [[nodiscard]] bool kills(int32_t Off) const {
152 return Off != TopOffset && Kills.contains(Off);
153 }
154
155 [[nodiscard]] constexpr bool
156 operator==(const AccessPath &Other) const noexcept {
157 return EmptyTombstone == Other.EmptyTombstone && Loads == Other.Loads &&
158 Stores == Other.Stores && Kills == Other.Kills;
159 }
160
161 bool operator!=(const AccessPath &Other) const noexcept {
162 return !(*this == Other);
163 }
164
165 friend size_t hash_value(const AccessPath &FieldString) noexcept;
166
167 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
168 const AccessPath &FieldString);
169
170 void print(llvm::raw_ostream &OS, const FieldStringManager &Mgr) const;
171};

173struct AccessPathDMI {
174 static AccessPath getEmptyKey() {
175 AccessPath Ret{};
176 Ret.EmptyTombstone = 1;
177 return Ret;
178 }
179 static AccessPath getTombstoneKey() {
180 AccessPath Ret{};
181 Ret.EmptyTombstone = 2;
182 return Ret;
183 }
184 static auto getHashValue(const AccessPath &FieldString) noexcept {
185 return hash_value(FieldString);
186 }
187 static bool isEqual(const AccessPath &L, const AccessPath &R) noexcept {
188 if (L.EmptyTombstone != R.EmptyTombstone) {
189 return false;
190 }
191 if (L.EmptyTombstone) {
192 return true;
193 }
194 return L == R;
195 }

198/// An edge-value consisting of a set if CFL field access strings.
199struct IFDSEdgeValue {
200 [[clang::require_explicit_initialization]] FieldStringManager *Mgr{};
201 llvm::SmallDenseSet<AccessPath, 2, AccessPathDMI> Paths;
202
203 static constexpr llvm::StringLiteral LogCategory = "IFDSEdgeValue";
204
205 void applyTransforms(const IFDSEdgeValue &Txns, uint8_t DepthKLimit);
206
207 bool operator==(const IFDSEdgeValue &Other) const noexcept {
208 assert(Mgr == Other.Mgr);
209 assert(Mgr != nullptr);
210 return Paths == Other.Paths;
211 }
212 bool operator!=(const IFDSEdgeValue &Other) const noexcept {
213 return !(*this == Other);
214 }
215
216 [[nodiscard]] friend auto hash_value(const IFDSEdgeValue EV) {
217 return llvm::hash_combine_range(EV.Paths.begin(), EV.Paths.end());
218 }
219
220 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
221 const IFDSEdgeValue &EV);
222
223 [[nodiscard]] bool isEpsilon() const {
224 return Paths.size() == 1 && Paths.begin()->empty();
225 }
226
227 [[nodiscard]] static IFDSEdgeValue epsilon(FieldStringManager *Mgr) {
228 IFDSEdgeValue Ret{.Mgr = &assertNotNull(Mgr), .Paths = {}};
229 Ret.Paths.insert({}); // Not using initializer_list to prevent copying
230 return Ret;
231 }
232
233 // To be picked up via ADL by psr::join(LatticeDomain, LatticeDomain)
234 [[nodiscard]] friend auto join(const IFDSEdgeValue &L,
235 const IFDSEdgeValue &R) {
236 assert(L.Mgr == R.Mgr);
237 assert(L.Mgr != nullptr);
238 const bool LeftSmaller = L.Paths.size() < R.Paths.size();
239 auto Ret = LeftSmaller ? R : L;
240 const auto &Smaller = LeftSmaller ? L : R;
241 Ret.Paths.insert(Smaller.Paths.begin(), Smaller.Paths.end());
242 // XXX: k-limit num-paths: This may not be necessary, as join() is only
243 // called from IDE-Phase-II
244 return Ret;
245 }
246};

270/// Utility to strip off potential pointer-arithmetic from V and accumulating
271/// the byte-offset.
272[[nodiscard]] inline std::pair<const llvm::Value *, int32_t>
273getBaseAndOffset(const llvm::Value *V, const llvm::DataLayout &DL) {
274 llvm::APInt Offset(64, 0);
275 int32_t OffsVal = AccessPath::TopOffset;
276 const auto *Base = V->stripAndAccumulateConstantOffsets(DL, Offset, true);
277
278 if (llvm::isa<llvm::GEPOperator>(Base)) {
279 return {Base->stripPointerCastsAndAliases(), AccessPath::TopOffset};
280 }
281
282 auto RawOffsVal = Offset.getSExtValue();
283 if (RawOffsVal <= INT32_MAX && RawOffsVal >= INT32_MIN) {
284 OffsVal = int32_t(RawOffsVal);
285 }
286
287 return {Base->stripPointerCastsAndAliases(), OffsVal};
288}

290/// Checks whether Fact holds at Inst in a field-sensitive way
291template <bool AllowDeepTaints = true, typename ResultsT>
292[[nodiscard]] inline bool holdsFactAt(const ResultsT &Results,
293 IFDSDomain::n_t Inst,
294 IFDSDomain::d_t Fact) {
295 const auto &Fields = Results.resultAt(Inst, Fact);
296
297 if (Fields.isTop()) {
298 // Was not computed by the IDE Solver
299 return false;
300 }
301
302 if (const IFDSEdgeValue *FieldStrings = Fields.getValueOrNull()) {
303 if constexpr (!AllowDeepTaints) {
304 // whether Facts itself holds, not whether any fields of it may hold
305 return FieldStrings->isEpsilon();
306 }
307 if (FieldStrings->Paths.empty()) {
308 // has been killed entirely
309 return false;
310 }
311 }
312
313 return true;
314}

316/// Given a QueryMap of the form map<n_t, set<d_t>>, calls the Handler for all
317/// inst-fact pairs that hold in a field-sensitive way and filters out all
318/// others.
319///
320/// The Handler may opt into early exit by returning false. Returning void is
321/// permitted.
322template <bool AllowDeepTaints = true, typename ResultsT>
323bool filterFieldSensFacts(
324 const ResultsT &Results, const auto &QueryMap,
325 std::invocable<IFDSDomain::n_t, IFDSDomain::d_t> auto Handler) {
326 const IFDSDomain::l_t Top = psr::Top{};
327
328 for (const auto &[Inst, FactsAtInst] : QueryMap) {
329 const auto &Row = Results.row(Inst);
330 for (const auto &Fact : FactsAtInst) {
331 const auto &Fields = getOr(Row, Fact, Top);
332
333 if (Fields.isTop()) {
334 // Was not computed by the IDE Solver
335 continue;
336 }
337
338 if (const auto *FieldStrings = Fields.getValueOrNull()) {
339 if (!AllowDeepTaints && !FieldStrings->isEpsilon()) {
340 // Fact does not hold itself, but fields of Fact may hold. In
341 // aggressive mode, we ignore them
342 continue;
343 }
344 if (FieldStrings->Paths.empty()) {
345 // has been killed entirely
346 continue;
347 }
348 }
349
350 if constexpr (std::convertible_to<
351 std::invoke_result_t<decltype(Handler) &,
352 IFDSDomain::n_t, IFDSDomain::d_t>,
353 bool>) {
354 if (!std::invoke(Handler, Inst, Fact)) {
355 return false;
356 }
357 } else {
358 std::invoke(Handler, Inst, Fact);
359 }
360 }
361 }
362 return true;
363}

367/// An IFDS-Problem adaptor that makes any field-insensitive IFDS analysis
368/// field-sensitive. Just wrap your IFDS problem with
369/// FieldSensAllocSitesAwareIFDSProblem and use the IterativeIDESolver instead
370/// of the IFDSSolver.
371///
372/// The only thing to change in your usual IFDS problem is not to kill data-flow
373/// facts when only parts of the fields should be killed. This is now handled by
374/// the FieldSensAllocSitesAwareIFDSProblem. For that, provide a
375/// FieldSensAllocSitesAwareIFDSProblemConfig with a proper KillsAt
376/// implementation.
377class CFLFieldSensIFDSProblem

380
381 static decltype(cfl_fieldsens::IFDSProblemConfig::KillsAt)
382 deriveKillsAt(auto *UserProblem) {
383 assert(UserProblem != nullptr);
384 if constexpr (requires() {
385 {
386 UserProblem->killsAt()
387 } -> psr::invocable_r<std::optional<int32_t>, n_t, d_t>;
388 }) {
389 return UserProblem->killsAt();
390 } else if constexpr (requires() {
391 {
392 UserProblem->killsAt()
393 } -> std::invocable<n_t, d_t>;
394 }) {
395 // Intentionally leaving an unused variable, so that the compiler emits a
396 // warning here
397 auto KillsAtHasWrongReturnType = UserProblem->killsAt();
398 return nullptr;
399 } else {
400 return nullptr;
401 }
402 }

404public:
