25/// Wrapper over an edge function and a pointer to a compatible
26/// EdgeFunctionSingletonCache.
27///
28/// When converting CachedEdgeFunction to EdgeFunction, signals to the
29/// EdgeFunction-allocator that, if not small-object-optimized, this edge
30/// function EF wishes to be allocated within the provided cache.
31///
32/// Cache may not be nullptr.
33template <typename EdgeFunctionTy> struct CachedEdgeFunction {
34 using EdgeFunctionType = EdgeFunctionTy;
35
36 EdgeFunctionTy EF{};
37 EdgeFunctionSingletonCache<EdgeFunctionTy> *Cache{};
38};
39
40/// Base-class for edge function caches. Compatible with the caching mechanism
41/// built in EdgeFunction.
42template <typename EdgeFunctionTy> class EdgeFunctionSingletonCache {
43public:
44 virtual ~EdgeFunctionSingletonCache() = default;
45
46 /// Checks whether the edge function EF is cached in this cache. Returns the
47 /// cached entry if found, else nullptr.
48 [[nodiscard]] virtual const void *
49 lookup(const EdgeFunctionTy &EF) const noexcept = 0;
50
51 /// Inserts the cache-entry Mem for the edge function *EF into the cache.
52 /// Typically, EF points into the buffer pointed to by Mem. Both pointers are
53 /// being stored and should not be null.
54 ///
55 /// Precondition: *EF Must not be in the cache.
56 virtual void insert(const EdgeFunctionTy *EF, const void *Mem) = 0;
57
58 /// Erases the cache-entry associated with the edge function EF from the
59 /// cache.
60 virtual void erase(const EdgeFunctionTy &EF) noexcept = 0;
61
62 template <typename... ArgTys>
63 [[nodiscard]] auto createEdgeFunction(ArgTys &&...Args) {
64 return CachedEdgeFunction<EdgeFunctionTy>{
65 EdgeFunctionTy{std::forward<ArgTys>(Args)...}, this};
66 }
67};
68
69template <typename EdgeFunctionTy>
70CachedEdgeFunction(EdgeFunctionTy, EdgeFunctionSingletonCache<EdgeFunctionTy> *)
71 -> CachedEdgeFunction<EdgeFunctionTy>;
