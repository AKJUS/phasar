20enum class SecureHeapFact { ZERO, INITIALIZED };
21enum class SecureHeapValue { TOP, INITIALIZED, BOT };
22
23template <> struct JoinLatticeTraits<SecureHeapValue> {
24 static constexpr auto bottom() noexcept { return SecureHeapValue::BOT; }
25 static constexpr auto top() noexcept { return SecureHeapValue::TOP; }
26 static constexpr auto join(SecureHeapValue LHS,
27 SecureHeapValue RHS) noexcept {
28 if (LHS == RHS || RHS == top()) {
29 return LHS;
30 }
31 if (LHS == top()) {
32 return RHS;
33 }
34 return bottom();
35 }
36};
