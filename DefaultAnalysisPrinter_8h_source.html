34/// \brief A default implementation of the AnalysisPrinterBase. Aggregates all
35/// analysis results in a vector and prints them when the analysis is finished.
36template <typename AnalysisDomainTy>
37class DefaultAnalysisPrinter : public AnalysisPrinterBase<AnalysisDomainTy> {
38 using n_t = typename AnalysisDomainTy::n_t;
39 using d_t = typename AnalysisDomainTy::d_t;
40 using l_t = typename AnalysisDomainTy::l_t;
41
42public:
43 DefaultAnalysisPrinter() noexcept = default;
44
45private:
46 void doOnResult(n_t Instr, d_t DfFact, l_t Lattice,
47 DataFlowAnalysisType AnalysisType) override {
48 AnalysisResults.emplace_back(Instr, std::move(DfFact), std::move(Lattice),
49 AnalysisType);
50 }
51
52 void doOnFinalize(llvm::raw_ostream &OS) override {
53 for (const auto &Iter : AnalysisResults) {
54 OS << "At IR statement: " << NToString(Iter.Instr) << '\n';
55 OS << " Fact: " << DToString(Iter.Fact) << '\n';
56
57 if constexpr (!std::is_same_v<l_t, BinaryDomain>) {
58 OS << " Value: " << LToString(Iter.LatticeElement) << '\n';
59 }
60 OS << '\n';
61 }
62 AnalysisResults.clear();
63 }
64
65 std::vector<Warning<AnalysisDomainTy>> AnalysisResults{};
66};
