60template <typename T>
61concept IsPointsToIterator =
62 requires(const T &PI, typename T::o_t Obj, typename T::n_t Inst,
63 llvm::function_ref<void(typename T::o_t)> Callback) {

67/// A type-erased reference to any object implementing the IsPointsToIterator
68/// interface. Use this, if your alias-aware analysis just needs
69/// a minimal interface to work with points-to relations and does not require
70/// the versatility of PointsToInfoRef.
71///
72/// This is a *non-owning* reference similar to std::string_view and
73/// llvm::ArrayRef. Pass values of this type by value.
74///
75/// \note You can also convert an AliasInfoRef to PointsToIteratorRef. In this
76/// case, it will use the getReachableAllocationSites() API.
77template <typename V, typename O, typename N>
78class [[gsl::Pointer]] PointsToIteratorRef : protected TypeErasureUtils {
79public:
80 using v_t = V;
81 using o_t = O;
82 using n_t = N;

89 ByConstRef<n_t>);

91 };

160 template <typename ConcretePTA>
161 static o_t asAbstractObjectThunk(const void *PT,
162 ByConstRef<v_t> Pointer) noexcept {
163 if constexpr (detail::HasAsAbstractObject<ConcretePTA>) {
164 return fromOpaquePtr<ConcretePTA>(PT)->asAbstractObject(Pointer);
165 } else if constexpr (std::is_convertible_v<v_t, o_t>) {
166 return Pointer;
167 } else {
168 static_assert(detail::HasAsAbstractObject<ConcretePTA>);
169 }

172 template <typename ConcretePTA>
173 static void forallPointeesOfThunk(const void *PT, ByConstRef<o_t> Pointer,
174 ByConstRef<n_t> At,
175 llvm::function_ref<void(O)> WithPointee) {
176 const auto *CPT = fromOpaquePtr<ConcretePTA>(PT);
177 if constexpr (IsPointsToIterator<ConcretePTA>) {
178 return (void)CPT->forallPointeesOf(Pointer, At, WithPointee);
179 } else {
180 auto &&PointsToSet = CPT->getPointsToSet(Pointer, At);
181 // The PointsToSet can be a set or a pointer to a set
182 auto *PointsToSetPtr = getPointerFrom(PointsToSet);
183 for (auto &&Pointee : *PointsToSetPtr) {
184 WithPointee(PSR_FWD(Pointee));
185 }
186 }
187 }

189 template <typename ConcretePTA>
190 static void
191 forallReachableAllocationSitesThunk(const void *AS, ByConstRef<o_t> Pointer,
192 ByConstRef<n_t> At,
193 llvm::function_ref<void(O)> WithPointee) {
194 auto *CAS = fromOpaquePtr<ConcretePTA>(AS);
195 // Note: The getReachableAllocationSites() API requires non-const access to
196 // support lazy computation (if requested in the LLVMAliasSet ctor).
197 // The below should still be safe, as we restrict our PointsToInfoRef ctor
198 // to take a non-const pointer to an aliasinfo.
199 auto AliasSetPtr =
200 const_cast<ConcretePTA *>(CAS)->getReachableAllocationSites(Pointer,
201 true, At);
202
203 for (auto &&Alias : *AliasSetPtr) {
204 WithPointee(PSR_FWD(Alias));
205 }
206 }
207
208 template <typename ConcretePTA>
209 static bool mayPointsToThunk(const void *PT, ByConstRef<o_t> Pointer,
210 ByConstRef<o_t> Obj, ByConstRef<n_t> At) {
211 const auto *CPT = fromOpaquePtr<ConcretePTA>(PT);
212 if constexpr (detail::HasMayPointsTo<ConcretePTA>) {
213 return CPT->mayPointsTo(Pointer, Obj, At);
214 } else if constexpr (detail::HasGetPointsToSet<ConcretePTA>) {
215 auto &&PointsToSet = CPT->getPointsToSet(Pointer, At);
216 // The PointsToSet can be a set or a pointer to a set
217 auto *PointsToSetPtr = getPointerFrom(PointsToSet);
218 return PointsToSetPtr->count(Obj);
219 } else {
220 bool Ret = false;
221 CPT->forallPointeesOf([&Ret, Obj](o_t Pointee) {
222 if (Pointee == Obj) {
223 Ret = true;
224 }
225 });
226 return Ret;
227 }
228 }
229
230 template <typename ConcretePTA>
231 static bool maybeInReachableAllocationSitesThunk(const void *AS,
232 ByConstRef<o_t> Pointer,
233 ByConstRef<o_t> Obj,
234 ByConstRef<n_t> At) {
235 if (Pointer == Obj) {
236 return true;
237 }
238
239 auto *CAS = fromOpaquePtr<ConcretePTA>(AS);
240 // Note: Same as for getReachableAllocationSites() above.
241 return const_cast<ConcretePTA *>(CAS)->isInReachableAllocationSites(
242 Pointer, Obj, true, At);
243 }

245 template <typename ConcretePTA>
246 static void destroyThunk(const void *PT) noexcept {
247 if constexpr (!CanSSO<ConcretePTA>) {
248 delete fromOpaquePtr<ConcretePTA>(PT);
249 }
250 }

252 template <typename ConcretePTA>
253 static constexpr VTable VTableFor = {
254 &asAbstractObjectThunk<ConcretePTA>,
255 &forallPointeesOfThunk<ConcretePTA>,
256 &mayPointsToThunk<ConcretePTA>,
257 &destroyThunk<ConcretePTA>,
258 };
259
260 template <typename ConcreteAA>
261 static constexpr VTable ReachableAllocSitesVTFor = {
262 &asAbstractObjectThunk<ConcreteAA>,
263 &forallReachableAllocationSitesThunk<ConcreteAA>,
264 &maybeInReachableAllocationSitesThunk<ConcreteAA>,
265 &destroyThunk<ConcreteAA>,
266 };

269 const void *PT{};
270 const VTable *VT{};
