77/// A type-erased reference to any object implementing the IsPointsToIterator
78/// interface. Use this, if your alias-aware analysis just needs
79/// a minimal interface to work with points-to relations and does not require
80/// the versatility of PointsToInfoRef.
81///
82/// This is a *non-owning* reference similar to std::string_view and
83/// llvm::ArrayRef. Pass values of this type by value.
84///
85/// \note You can also convert an AliasInfoRef to PointsToIteratorRef. In this
86/// case, it will use the getReachableAllocationSites() API.
87template <typename V, typename O, typename N>
88class [[gsl::Pointer]] PointsToIteratorRef : protected TypeErasureUtils {
89public:
90 using v_t = V;
91 using o_t = O;
92 using n_t = N;
93
94 struct VTable {
95 o_t (*AsAbstractObject)(const void *, ByConstRef<v_t>) noexcept;
96 void (*ForallPointeesOf)(const void *, ByConstRef<o_t>, ByConstRef<n_t>,
97 llvm::function_ref<void(O)>);
98 bool (*MayPointsTo)(const void *, ByConstRef<o_t>, ByConstRef<o_t>,
99 ByConstRef<n_t>);
100 void (*Destroy)(const void *) noexcept; // Useful for the owning variant
101 };

181 template <typename ConcretePTA>
182 static o_t asAbstractObjectThunk(const void *PT,
183 ByConstRef<v_t> Pointer) noexcept {
184 if constexpr (detail::HasAsAbstractObject<ConcretePTA>::value) {
185 return fromOpaquePtr<ConcretePTA>(PT)->asAbstractObject(Pointer);
186 } else if constexpr (std::is_convertible_v<v_t, o_t>) {
187 return Pointer;
188 } else {
189 static_assert(detail::HasAsAbstractObject<ConcretePTA>::value);
190 }

193 template <typename ConcretePTA>
194 static void forallPointeesOfThunk(const void *PT, ByConstRef<o_t> Pointer,
195 ByConstRef<n_t> At,
196 llvm::function_ref<void(O)> WithPointee) {
197 const auto *CPT = fromOpaquePtr<ConcretePTA>(PT);
198 if constexpr (IsPointsToIterator<ConcretePTA>) {
199 return (void)CPT->forallPointeesOf(Pointer, At, WithPointee);
200 } else {
201 auto &&PointsToSet = CPT->getPointsToSet(Pointer, At);
202 // The PointsToSet can be a set or a pointer to a set
203 auto *PointsToSetPtr = getPointerFrom(PointsToSet);
204 for (auto &&Pointee : *PointsToSetPtr) {
205 WithPointee(PSR_FWD(Pointee));
206 }
207 }
208 }

210 template <typename ConcretePTA>
211 static void
212 forallReachableAllocationSitesThunk(const void *AS, ByConstRef<o_t> Pointer,
213 ByConstRef<n_t> At,
214 llvm::function_ref<void(O)> WithPointee) {
215 auto *CAS = fromOpaquePtr<ConcretePTA>(AS);
216 // Note: The getReachableAllocationSites() API requires non-const access to
217 // support lazy computation (if requested in the LLVMAliasSet ctor).
218 // The below should still be safe, as we restrict our PointsToInfoRef ctor
219 // to take a non-const pointer to an aliasinfo.
220 auto AliasSetPtr =
221 const_cast<ConcretePTA *>(CAS)->getReachableAllocationSites(Pointer,
222 true, At);
223
224 for (auto &&Alias : *AliasSetPtr) {
225 WithPointee(PSR_FWD(Alias));
226 }
227 }
228
229 template <typename ConcretePTA>
230 static bool mayPointsToThunk(const void *PT, ByConstRef<o_t> Pointer,
231 ByConstRef<o_t> Obj, ByConstRef<n_t> At) {
232 const auto *CPT = fromOpaquePtr<ConcretePTA>(PT);
233 if constexpr (detail::HasMayPointsTo<ConcretePTA>::value) {
234 return CPT->mayPointsTo(Pointer, Obj, At);
235 } else if constexpr (detail::HasGetPointsToSet<ConcretePTA>::value) {
236 auto &&PointsToSet = CPT->getPointsToSet(Pointer, At);
237 // The PointsToSet can be a set or a pointer to a set
238 auto *PointsToSetPtr = getPointerFrom(PointsToSet);
239 return PointsToSetPtr->count(Obj);
240 } else {
241 bool Ret = false;
242 CPT->forallPointeesOf([&Ret, Obj](o_t Pointee) {
243 if (Pointee == Obj) {
244 Ret = true;
245 }
246 });
247 return Ret;
248 }
249 }
250
251 template <typename ConcretePTA>
252 static bool maybeInReachableAllocationSitesThunk(const void *AS,
253 ByConstRef<o_t> Pointer,
254 ByConstRef<o_t> Obj,
255 ByConstRef<n_t> At) {
256 if (Pointer == Obj) {
257 return true;
258 }
259
260 auto *CAS = fromOpaquePtr<ConcretePTA>(AS);
261 // Note: Same as for getReachableAllocationSites() above.
262 return const_cast<ConcretePTA *>(CAS)->isInReachableAllocationSites(
263 Pointer, Obj, true, At);
264 }
265
266 template <typename ConcretePTA>
267 static void destroyThunk(const void *PT) noexcept {
268 if constexpr (!CanSSO<ConcretePTA>) {
269 delete fromOpaquePtr<ConcretePTA>(PT);
270 }
271 }
272
273 template <typename ConcretePTA>
274 static constexpr VTable VTableFor = {
275 &asAbstractObjectThunk<ConcretePTA>,
276 &forallPointeesOfThunk<ConcretePTA>,
277 &mayPointsToThunk<ConcretePTA>,
278 &destroyThunk<ConcretePTA>,
279 };
280
281 template <typename ConcreteAA>
282 static constexpr VTable ReachableAllocSitesVTFor = {
283 &asAbstractObjectThunk<ConcreteAA>,
284 &forallReachableAllocationSitesThunk<ConcreteAA>,
285 &maybeInReachableAllocationSitesThunk<ConcreteAA>,
286 &destroyThunk<ConcreteAA>,
287 };

289protected:
290 const void *PT{};
291 const VTable *VT{};
292};
