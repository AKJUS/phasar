121 [[nodiscard]] EdgeFunctionCacheStats getStats() const noexcept {
122 size_t Sz = 0;
123 size_t MaxInnerSz = 0;
124 llvm::SmallVector<unsigned> Sizes;
125 for (const auto &[Edge, NormalAndCtrEF] : NormalAndCtrEFCache) {
126 Sz += NormalAndCtrEF->size();
127 MaxInnerSz = std::max(MaxInnerSz, NormalAndCtrEF->size());
128 Sizes.push_back(NormalAndCtrEF->size());
129 }
130
131 std::sort(Sizes.begin(), Sizes.end());
132
133 EdgeFunctionCacheStats Ret = *this;
134 Ret.NormalAndCtrEFCacheCumulSizeReduced = Sz;
135 Ret.AvgEFPerEdge = double(Sz) / NormalAndCtrEFCache.size();
136 Ret.MaxEFPerEdge = MaxInnerSz;
137 Ret.MedianEFPerEdge = Sizes.empty() ? 0 : Sizes[Sizes.size() / 2];
138 return Ret;
139 }
