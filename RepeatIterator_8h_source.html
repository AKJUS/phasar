23/// An iterator that iterates over the same value a specified number of times
24template <typename T> class RepeatIterator {
25public:
26 using value_type = T;
27 using reference = const T &;
28 using pointer = const T *;
29 using difference_type = ptrdiff_t;
30 using iterator_category = std::forward_iterator_tag;
31
32 reference operator*() const noexcept {
33 assert(Elem.has_value() && "Dereferencing end()-iterator");
34 return *Elem;
35 }
36 pointer operator->() const noexcept {
37 assert(Elem.has_value() && "Dereferencing end()-iterator");
38 return &*Elem;
39 }
40
41 RepeatIterator &operator++() noexcept {
42 ++Index;
43 return *this;
44 }
45 RepeatIterator operator++(int) noexcept {
46 auto Ret = *this;
47 ++*this;
48 return Ret;
49 }
50
51 bool operator==(const RepeatIterator &Other) const noexcept {
52 return Other.Index == Index;
53 }
54 bool operator!=(const RepeatIterator &Other) const noexcept {
55 return !(*this == Other);
56 }
57
58 template <typename TT,
59 typename = std::enable_if_t<std::is_same_v<T, std::decay_t<TT>>>>
60 explicit RepeatIterator(TT &&Elem) : Elem(std::forward<TT>(Elem)) {}
61 explicit RepeatIterator(size_t Index, std::true_type /*AsEndIterator*/)
62 : Index(Index), Elem(std::nullopt) {}
63
64 RepeatIterator() noexcept = default;
65
66private:
67 size_t Index{};
68 std::optional<T> Elem{};
69};

71template <typename T>
72using RepeatRangeType = llvm::iterator_range<RepeatIterator<T>>;
73template <typename T> auto repeat(T &&Elem, size_t Num) {
74 using iterator_type = RepeatIterator<std::decay_t<T>>;
75 auto Ret = llvm::make_range(iterator_type(std::forward<T>(Elem)),
76 iterator_type(Num, std::true_type{}));
77 return Ret;
78}

80static_assert(is_iterable_over_v<RepeatRangeType<int>, int>);
