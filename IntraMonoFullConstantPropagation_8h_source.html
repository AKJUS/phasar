108 size_t operator()(const psr::IntraMonoFCAFact &P) const {
109 std::hash<const llvm::Value *> HashPtr;
110 size_t HP = HashPtr(P.Fact);
111 size_t HU = 0;
112 // returns nullptr if P.second is Top or Bottom, a valid pointer otherwise
113 if (const auto *Ptr =
114 std::get_if<psr::IntraMonoFullConstantPropagation::plain_d_t>(
115 &P.Value)) {
116 HU = *Ptr;
117 }
118 return HP ^ (HU << 1);
119 }
