23namespace psr {
24template <typename Derived> class ICFGBase : public CRTPBase<Derived> {
25 friend Derived;
26 using CRTPBase<Derived>::self;
27
28public:
29 using n_t = typename CFGTraits<Derived>::n_t;
30 using f_t = typename CFGTraits<Derived>::f_t;
31
32 /// Returns an iterable range of all function definitions or declarations in
33 /// the ICFG
34 [[nodiscard]] decltype(auto) getAllFunctions() const {
35 return self().getAllFunctionsImpl();
36 }
37
38 /// returns the function definition or declaration with the given name. If
39 /// there is no such function, returns a default constructed f_t (nullptr for
40 /// pointers).
41 [[nodiscard]] f_t getFunction(llvm::StringRef Fun) const {
42 return self().getFunctionImpl(Fun);
43 }
44
45 /// Returns true, iff the given instruction is a call-site where the callee is
46 /// called via a function-pointer or another kind of virtual call.
47 /// NOTE: Trivial cases where a bitcast of a function is called may still
48 /// count as indirect call.
49 [[nodiscard]] bool isIndirectFunctionCall(ByConstRef<n_t> Inst) const {
50 return self().isIndirectFunctionCallImpl(Inst);
51 }
52 /// Returns true, iff the given instruction is a call-site where the callee is
53 /// called via virtual dispatch. NOTE: Here, a class-hierarchy is required and
54 /// a simple function-pointer is not sufficient.
55 [[nodiscard]] bool isVirtualFunctionCall(ByConstRef<n_t> Inst) const {
56 return self().isVirtualFunctionCallImpl(Inst);
57 }
58 /// Returns an iterable range of all instructions in all functions of the ICFG
59 /// that are neither call-sites nor start-points of a function
60 [[nodiscard]] decltype(auto) allNonCallStartNodes() const {
61 static_assert(
62 is_iterable_over_v<decltype(self().allNonCallStartNodesImpl()), n_t>);
63 return self().allNonCallStartNodesImpl();
64 }
65
66 /// Returns a view to the underlying call-graph
67 [[nodiscard]] decltype(auto) getCallGraph() const noexcept {
68 static_assert(
69 is_crtp_base_of_v<CallGraphBase,
70 std::decay_t<decltype(self().getCallGraphImpl())>>);
71 return self().getCallGraphImpl();
72 }
73
74 /// Returns an iterable range of all possible callee candidates at the given
75 /// call-site induced by the used call-graph.
76 [[nodiscard]] decltype(auto) getCalleesOfCallAt(ByConstRef<n_t> Inst) const {
77 return getCallGraph().getCalleesOfCallAt(Inst);
78 }
79 /// Returns an iterable range of all possible call-site candidates that may
80 /// call the given function induced by the used call-graph.
81 [[nodiscard]] decltype(auto) getCallersOf(ByConstRef<f_t> Fun) const {
82 return getCallGraph().getCallersOf(Fun);
83 }
84 /// Returns an iterable range of all call-instruction in the given function
85 [[nodiscard]] decltype(auto) getCallsFromWithin(ByConstRef<f_t> Fun) const {
86 static_assert(
87 is_iterable_over_v<decltype(self().getCallsFromWithinImpl(Fun)), n_t>);
88 return self().getCallsFromWithinImpl(Fun);
89 }
90 /// Returns an iterable range of all return-sites of the given
91 /// call-instruction. Often the same as getSuccsOf. NOTE: This function is
92 /// typically called in a hot part of the analysis and should therefore be
93 /// very fast
94 [[nodiscard]] decltype(auto)
95 getReturnSitesOfCallAt(ByConstRef<n_t> Inst) const {
96 static_assert(
97 is_iterable_over_v<decltype(self().getReturnSitesOfCallAtImpl(Inst)),
98 n_t>);
99 return self().getReturnSitesOfCallAtImpl(Inst);
100 }
101
102 /// Prints the underlying call-graph as DOT to the given output-stream
103 void print(llvm::raw_ostream &OS = llvm::outs()) const {
104 self().printImpl(OS);
105 }
106
107 /// Prints the underlying call-graph as Json to the given output-stream
108 void printAsJson(llvm::raw_ostream &OS = llvm::outs()) const {
109 self().printAsJsonImpl(OS);
110 }
111
112 [[nodiscard]] size_t getNumCallSites() const noexcept {
113 return self().getNumCallSitesImpl();
114 }
115};
116
117/// True, iff ICF is a proper instantiation of ICFGBase with n_t and f_t taken
118/// from the given analysis-Domain
119template <typename ICF, typename Domain>
120// NOLINTNEXTLINE(readability-identifier-naming)
121concept is_icfg_v = is_crtp_base_of_v<ICFGBase, ICF> &&
122 std::is_same_v<typename ICF::n_t, typename Domain::n_t> &&
123 std::is_same_v<typename ICF::f_t, typename Domain::f_t>;
124
125} // namespace psr
