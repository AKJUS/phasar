64/// \brief An extension of the path-reconstruction mechanism of the
65/// PathSensitivityManager that provides means to extract concrete combined
66/// control- and data-flow paths.
67///
68/// Filters out paths that are considered infeasible by the Z3
69/// constraint solver.
70template <typename AnalysisDomainTy,
71 typename = std::enable_if_t<std::is_same_v<
72 typename AnalysisDomainTy::n_t, const llvm::Instruction *>>>
73class Z3BasedPathSensitivityManager
74 : public Z3BasedPathSensitivityManagerBase,
75 public PathSensitivityManagerMixin<
76 Z3BasedPathSensitivityManager<AnalysisDomainTy>, AnalysisDomainTy,
77 typename Z3BasedPathSensitivityManagerBase::graph_type> {
78 using base_t = PathSensitivityManagerBase<typename AnalysisDomainTy::n_t>;
79 using mixin_t = PathSensitivityManagerMixin<Z3BasedPathSensitivityManager,
80 AnalysisDomainTy,
81 typename base_t::graph_type>;
82
83public:
84 using n_t = typename AnalysisDomainTy::n_t;
85 using d_t = typename AnalysisDomainTy::d_t;
86 using typename PathSensitivityManagerBase<n_t>::graph_type;
87 using MaybeFlowPathSeq = std::variant<FlowPathSequence<n_t>, z3::expr>;
88
89 explicit Z3BasedPathSensitivityManager(
90 const ExplodedSuperGraph<AnalysisDomainTy> *ESG,
91 Z3BasedPathSensitivityConfig Config, LLVMPathConstraints *LPC = nullptr)
92 : mixin_t(ESG), Config(std::move(Config)), LPC(LPC) {
93 if (!LPC) {
94 this->LPC = std::make_unique<LLVMPathConstraints>();
95 }
96 }
97
98 /// Reconstruct the feasible control- and data-flow paths the lead to
99 /// the given data-flow fact Fact holding right after Inst.
100 ///
101 /// The result is given as list of paths, where cycles are unrolled once in an
102 /// implementation-defined way.
103 /// It is strongly recommended to Use the Z3BasedPathSensitivityConfig in the
104 /// Z3BasedPathSensitivityManager's ctor to limit the returned paths;
105 /// otherwise this function quickly becomes a performance bottleneck.
106 FlowPathSequence<n_t> pathsTo(n_t Inst, d_t Fact) const {
107 if (Config.DAGSizeThreshold != SIZE_MAX) {
108 PHASAR_LOG_LEVEL(
109 WARNING,
110 "Attempting to compute FlowPaths without conditionally "
111 "falling back to constraint collection, but a DAGSizeThreshold "
112 "is specified. It will be ignored here. To make use of it, "
113 "please call the pathsOrConstraintTo function instead!");
114 }
115
116 graph_type Dag = this->pathsDagTo(Inst, std::move(Fact), Config);
117
118 PHASAR_LOG_LEVEL_CAT(
119 DEBUG, "PathSensitivityManager",
120 "PathsTo with MaxDAGDepth: " << Config.DAGDepthThreshold);
121
122#ifndef NDEBUG
123 {
124 std::error_code EC;
125 llvm::raw_fd_stream ROS(
126 "dag-" +
127 std::filesystem::path(psr::getFilePathFromIR(Inst))
128 .filename()
129 .string() +
130 "-" + psr::getMetaDataID(Inst) + ".dot",
131 EC);
132 assert(!EC);
133 printGraph(Dag, ROS, "DAG", [](llvm::ArrayRef<n_t> PartialPath) {
134 std::string Buf;
135 llvm::raw_string_ostream ROS(Buf);
136 ROS << "[ ";
137 llvm::interleaveComma(PartialPath, ROS, [&ROS](const auto *Inst) {
138 ROS << psr::getMetaDataID(Inst);
139 });
140 ROS << " ]";
141 ROS.flush();
142 return Buf;
143 });
144
145 llvm::errs() << "Paths DAG has " << Dag.Roots.size() << " roots\n";
146 }
147#endif
148
149 vertex_t Leaf = [&Dag] {
150 for (auto Vtx : graph_traits_t::vertices(Dag)) {
151 if (graph_traits_t::outDegree(Dag, Vtx) == 0) {
152 return Vtx;
153 }
154 }
155 llvm_unreachable("Expect the DAG to have a leaf node!");
156 }();
157
158 z3::expr Constraint = filterOutUnreachableNodes(Dag, Leaf, Config, *LPC);
159
160 if (Constraint.is_false()) {
161 PHASAR_LOG_LEVEL_CAT(INFO, "PathSensitivityManager",
162 "The query position is unreachable");
163 return FlowPathSequence<n_t>();
164 }
165
166 auto Ret = filterAndFlattenRevDag(Dag, Leaf, Inst, Config, *LPC);
167
168 deduplicatePaths(Ret);
169
170#ifndef NDEBUG
171#ifdef DYNAMIC_LOG
172 PHASAR_LOG_LEVEL_CAT(DEBUG, "PathSensitivityManager",
173 "Recorded " << Ret.size() << " valid paths:");
174
175 std::string Str;
176 for (const FlowPath<n_t> &Path : Ret) {
177 Str.clear();
178 llvm::raw_string_ostream ROS(Str);
179 ROS << "> ";
180 llvm::interleaveComma(Path.Path, ROS,
181 [&ROS](auto *Inst) { ROS << getMetaDataID(Inst); });
182 ROS << ": " << Path.Constraint.to_string();
183 ROS.flush();
184 PHASAR_LOG_LEVEL_CAT(DEBUG, "PathSensitivityManager", Str);
185 }
186
187#endif // DYNAMIC_LOG
188#endif // NDEBUG
189
190 return Ret;
191 }
192
193private:
194 Z3BasedPathSensitivityConfig Config{};
195 /// FIXME: Not using 'mutable' here
196 mutable MaybeUniquePtr<LLVMPathConstraints, true> LPC{};
197};
