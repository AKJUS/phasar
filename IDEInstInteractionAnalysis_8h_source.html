263
264 FlowFunctionPtrType getNormalFlowFunction(n_t Curr, n_t /* Succ */) override {
265 // Generate all local variables
266 //
267 // Flow function:
268 //
269 // 0
270 // |\
271 // x = alloca y | \
272 // v v
273 // 0 x
274 //
275 if (const auto *Alloca = llvm::dyn_cast<llvm::AllocaInst>(Curr)) {
276 PHASAR_LOG_LEVEL(DFADEBUG, "AllocaInst");
277 return this->generateFromZero(Alloca);
278 }
279
280 // Handle indirect taints, i. e., propagate values that depend on branch
281 // conditions whose operands are tainted.
282 if constexpr (EnableIndirectTaints) {
283 if (const auto *Br = llvm::dyn_cast<llvm::BranchInst>(Curr);
284 Br && Br->isConditional()) {
285 // If the branch is conditional and its condition is tainted, then we
286 // need to propagates the instructions that are depending on this
287 // branch, too.
288 //
289 // Flow function:
290 //
291 // Let I be the set of instructions of the branch instruction's
292 // successors.
293 //
294 // 0 c x
295 // | |\
296 // x = br C, label if.then, label if.else | | \--\
297 // v v v v
298 // 0 c x I
299 //
300 return this->lambdaFlow([Br](d_t Src) {
301 container_type Facts;
302 Facts.insert(Src);
303 if (Src == Br->getCondition()) {
304 Facts.insert(Br);
305 for (const auto *Succs : Br->successors()) {
306 for (const auto &Inst : Succs->instructionsWithoutDebug()) {
307 Facts.insert(&Inst);
308 }
309 }
310 }
311 return Facts;
312 });
313 }
314 }
315
316 // Handle points-to information if the user wishes to conduct a
317 // non-syntax-only inst-interaction analysis.
318 if constexpr (!SyntacticAnalysisOnly) {
319
320 // (ii) Handle semantic propagation (pointers) for load instructions.
321
322 if (const auto *Load = llvm::dyn_cast<llvm::LoadInst>(Curr)) {
323 // If one of the potentially many loaded values holds, the load itself
324 // (dereferenced value) must also be generated and populated.
325 //
326 // Flow function:
327 //
328 // Let Y = pts(y), be the points-to set of y.
329 //
330 // 0 Y x
331 // | |\
332 // x = load y | | \
333 // v v v
334 // 0 Y x
335 //
336 return this->generateFlowIf(
337 Load, [PointerOp = Load->getPointerOperand(),
338 PTS = PT.getReachableAllocationSites(
339 Load->getPointerOperand(), OnlyConsiderLocalAliases)](
340 d_t Src) { return Src == PointerOp || PTS->count(Src); });
341 }
342
343 // (ii) Handle semantic propagation (pointers) for store instructions.
344 if (const auto *Store = llvm::dyn_cast<llvm::StoreInst>(Curr)) {
345 // If the value to be stored holds, the potential memory location(s)
346 // that it is stored to must be generated and populated, too.
347 //
348 // Flow function:
349 //
350 // Let X be
351 // - pts(x), the points-to set of x, if x is an intersting pointer.
352 // - a singleton set containing x, otherwise.
353 //
354 // Let Y be pts(y), the points-to set of y.
355 //
356 // 0 X y
357 // | |\ |
358 // store x y | | \|
359 // v v v
360 // 0 x Y
361 //
362 return this->lambdaFlow(
363 [Store, PointerPTS = PT.getReachableAllocationSites(
364 Store->getPointerOperand(), OnlyConsiderLocalAliases,
365 Store)](d_t Src) -> container_type {
366 if (Store->getPointerOperand() == Src || PointerPTS->count(Src)) {
367 // Here, we are unsound!
368 return {};
369 }
370 container_type Facts;
371 Facts.insert(Src);
372 // y/Y now obtains its new value(s) from x/X
373 // If a value is stored that holds we must generate all potential
374 // memory locations the store might write to.
375 if (Store->getValueOperand() == Src) {
376 Facts.insert(Store->getPointerOperand());
377 Facts.insert(PointerPTS->begin(), PointerPTS->end());
378 }
379 // ... or from zero, if a constant literal is stored to y
380 if (llvm::isa<llvm::ConstantData>(Store->getValueOperand()) &&
381 IDEInstInteractionAnalysisT::isZeroValueImpl(Src)) {
382 Facts.insert(Store->getPointerOperand());
383 Facts.insert(PointerPTS->begin(), PointerPTS->end());
384 }
385 return Facts;
386 });
387 }
388 }
389
390 // (i) Handle syntactic propagation
391
392 // Handle load instruction
393 //
394 // Flow function:
395 //
396 // 0 y x
397 // | |\
398 // x = load y | | \
399 // v v v
400 // 0 y x
401 //
402 if (const auto *Load = llvm::dyn_cast<llvm::LoadInst>(Curr)) {
403 return this->generateFlow(Load, Load->getPointerOperand());
404 }
405 // Handle store instructions
406 //
407 // Flow function:
408 //
409 // 0 x y
410 // | |\ |
411 // store x y | | \|
412 // v v v
413 // 0 x y
414 //
415 if (const auto *Store = llvm::dyn_cast<llvm::StoreInst>(Curr)) {
416 return this->lambdaFlow([Store](d_t Src) -> container_type {
417 // Override old value, i.e., kill value that is written to and
418 // generate from value that is stored.
419 if (Store->getPointerOperand() == Src) {
420 return {};
421 }
422 container_type Facts = {Src};
423 // y now obtains its new value from x
424 if (Store->getValueOperand() == Src) {
425 Facts.insert(Store->getPointerOperand());
426 }
427 // ... or from zero, if a constant literal is stored to y
428 if (llvm::isa<llvm::ConstantData>(Store->getValueOperand()) &&
429 IDEInstInteractionAnalysisT::isZeroValueImpl(Src)) {
430 Facts.insert(Store->getPointerOperand());
431 }
432 IF_LOG_ENABLED({
433 for (const auto Fact : Facts) {
434 PHASAR_LOG_LEVEL(DFADEBUG, "Create edge: "
435 << llvmIRToShortString(Src) << " --"
436 << llvmIRToShortString(Store)
437 << "--> " << Fact);
438 }
439 });
440 return Facts;
441 });
442 }
443 // At last, we can handle all other (unary/binary) instructions.
444 //
445 // Flow function:
446 //
447 // 0 x o p
448 // | | /| /|
449 // x = instruction o p | |/ |/ |
450 // | | /| |
451 // | |/ | |
452 // v v v v
453 // 0 x o p
454 //
455 return this->lambdaFlow([Inst = Curr](d_t Src) {
456 container_type Facts;
457 if (IDEInstInteractionAnalysisT::isZeroValueImpl(Src)) {
458 // keep the zero flow fact
459 Facts.insert(Src);
460 return Facts;
461 }
462
463 // continue syntactic propagation: populate and propagate other existing
464 // facts
465 for (auto &Op : Inst->operands()) {
466 // if one of the operands holds, also generate the instruction using
467 // it
468 if (Op == Src) {
469 Facts.insert(Inst);
470 }
471 }
472 // pass everything that already holds as identity
473 Facts.insert(Src);
474 IF_LOG_ENABLED({
475 for (const auto Fact : Facts) {
476 PHASAR_LOG_LEVEL(DFADEBUG,
477 "Create edge: " << llvmIRToShortString(Src) << " --"
478 << llvmIRToShortString(Inst)
479 << "--> " << Fact);
480 }
481 });
482 return Facts;
483 });
484 }

486 inline FlowFunctionPtrType getCallFlowFunction(n_t CallSite,
487 f_t DestFun) override {
488 if (this->ICF->isHeapAllocatingFunction(DestFun)) {
489 // Kill add facts and model the effects in getCallToRetFlowFunction().
490 return this->killAllFlows();
491 }
492 if (DestFun->isDeclaration()) {
493 // We don't have anything that we could analyze, kill all facts.
494 return this->killAllFlows();
495 }
496 const auto *CS = llvm::cast<llvm::CallBase>(CallSite);
497
498 // Map actual to formal parameters.
499 auto MapFactsToCalleeFF = mapFactsToCallee<d_t>(
500 CS, DestFun, [CS](const llvm::Value *ActualArg, ByConstRef<d_t> Src) {
501 if (d_t(ActualArg) != Src) {
502 return false;
503 }
504
505 if (CS->hasStructRetAttr() && ActualArg == CS->getArgOperand(0)) {
506 return false;
507 }
508
509 return true;
510 });
511
512 // Generate the artificially introduced RVO parameters from zero value.
513 auto SRetFormal = CS->hasStructRetAttr() ? DestFun->getArg(0) : nullptr;
514
515 if (SRetFormal) {
516 return this->unionFlows(
517 std::move(MapFactsToCalleeFF),
518 this->generateFlowAndKillAllOthers(SRetFormal, this->getZeroValue()));
519 }
520
521 return MapFactsToCalleeFF;
522 }

524 inline FlowFunctionPtrType getRetFlowFunction(n_t CallSite, f_t /*CalleeFun*/,
525 n_t ExitInst,
526 n_t /* RetSite */) override {
527 // Map return value back to the caller. If pointer parameters hold at the
528 // end of a callee function generate all of those in the caller context.
529 if (CallSite == nullptr) {
530 return this->killAllFlows();
531 }
532 auto MapFactsToCallerFF =
533 mapFactsToCaller<d_t>(llvm::cast<llvm::CallBase>(CallSite), ExitInst,
534 {}, [](const llvm::Value *RetVal, d_t Src) {
535 if (Src == RetVal) {
536 return true;
537 }
538 if (isZeroValueImpl(Src)) {
539 if (llvm::isa<llvm::ConstantData>(RetVal)) {
540 return true;
541 }
542 }
543 return false;
544 });
545
546 return MapFactsToCallerFF;
547 }

549 inline FlowFunctionPtrType
550 getCallToRetFlowFunction(n_t CallSite, n_t /* RetSite */,
551 llvm::ArrayRef<f_t> Callees) override {
552 // Model call to heap allocating functions (new, new[], malloc, etc.) --
553 // only model direct calls, though.
554 if (Callees.size() == 1) {
555 const auto *Callee = Callees.front();
556 if (this->ICF->isHeapAllocatingFunction(Callee)) {
557 // In case a heap allocating function is called, generate the pointer
558 // that is returned.
559 //
560 // Flow function:
561 //
562 // Let H be a heap allocating function.
563 //
564 // 0
565 // |\
566 // x = call H | \
567 // v v
568 // 0 x
569 //
570 return generateFromZero(CallSite);
571 }
572 }
573 // Just use the auto mapping for values; pointer parameters and global
574 // variables are killed and handled by getCallFlowfunction() and
575 // getRetFlowFunction().
576 // However, if only declarations are available as callee targets we would
577 // lose the data-flow facts involved in the call which is usually not the
578 // behavior that is intended. In that case, we must propagate all data-flow
579 // facts alongside the call site.
580 bool OnlyDecls = true;
581 bool AllVoidRetTys = true;
582 for (auto Callee : Callees) {
583 if (!Callee->isDeclaration()) {
584 OnlyDecls = false;
585 }
586 if (!Callee->getReturnType()->isVoidTy()) {
587 AllVoidRetTys = false;
588 }
589 }
590
591 return this->lambdaFlow([CallSite = llvm::cast<llvm::CallBase>(CallSite),
592 OnlyDecls,
593 AllVoidRetTys](d_t Source) -> container_type {
594 // There are a few things to consider, in case only declarations of
595 // callee targets are available.
596 if (OnlyDecls) {
597 if (!AllVoidRetTys) {
598 // If one or more of the declaration-only targets return a value, it
599 // must be generated from zero!
600 if (isZeroValueImpl(Source)) {
601 return {Source, CallSite};
602 }
603 }
604 // If all declaration-only callee targets return void, just pass
605 // everything as identity.
606 return {Source};
607 }
608 // Do not pass global variables if definitions of the callee
609 // function(s) are available, since the effect of the callee on these
610 // values will be modelled using combined getCallFlowFunction and
611 // getReturnFlowFunction.
612 if (llvm::isa<llvm::Constant>(Source.getBase())) {
613 return {};
614 }
615 // Pass everything else as identity. In particular, also do not kill
616 // pointer or reference parameters since this then also captures usages
617 // oft the parameters, which we wish to compute using this analysis.
618 return {Source};
619 });
620 }

628 inline InitialSeeds<n_t, d_t, l_t> initialSeeds() override {
629 InitialSeeds<n_t, d_t, l_t> Seeds;
630
631 forallStartingPoints(this->EntryPoints, ICF, [this, &Seeds](n_t SP) {
632 // Set initial seeds at the required entry points and generate the global
633 // variables using generalized initial seeds
634
635 // Generate zero value at the entry points
636 Seeds.addSeed(SP, this->getZeroValue(), Bottom{});
637 // Generate formal parameters of entry points, e.g. main(). Formal
638 // parameters will otherwise cause trouble by overriding alloca
639 // instructions without being valid data-flow facts themselves.
640 for (const auto &Arg : SP->getFunction()->args()) {
641 Seeds.addSeed(SP, &Arg, BitVectorSet<e_t>());
642 }
643 // Generate all global variables using generalized initial seeds
644
645 for (const auto &G : this->IRDB->getModule()->globals()) {
646 if (const auto *GV = llvm::dyn_cast<llvm::GlobalVariable>(&G)) {
647 l_t InitialValues = bvSetFrom(invoke_or_default(EdgeFactGen, GV));
648 Seeds.addSeed(SP, GV, std::move(InitialValues));
649 }
650 }
651 });
652
653 return Seeds;
654 }

666
667 inline EdgeFunctionType getStrongUpdateStoreEF(const llvm::StoreInst *Store,
668 d_t CurrNode, d_t SuccNode,
669 l_t UserEdgeFacts) {
670
671 // Overriding edge: obtain labels from value to be stored (and may add
672 // UserEdgeFacts, if any).
673 //
674 // x --> y
675 //
676 // Edge function:
677 //
678 // x
679 // \
680 // store x y \ \x.x \cup { commit of('store x y') }
681 // v
682 // y
683 //
684
685 if (CurrNode == Store->getValueOperand() ||
686 (isZeroValue(CurrNode) &&
687 llvm::isa<llvm::Constant>(Store->getValueOperand()))) {
688 if (SuccNode == Store->getPointerOperand() ||
689 PT.isInReachableAllocationSites(Store->getPointerOperand(), SuccNode,
690 true, Store)) {
691 if (isZeroValue(CurrNode)) {
692 return IIAAKillOrReplaceEFCache.createEdgeFunction(
693 std::move(UserEdgeFacts));
694 }
695 return IIAAAddLabelsEFCache.createEdgeFunction(
696 std::move(UserEdgeFacts));
697 }
698 }
699
700 if (SyntacticAnalysisOnly) {
701 // Kill all labels that are propagated along the edge of the value that
702 // is overridden.
703 //
704 // y --> y
705 //
706 // Edge function:
707 //
708 // y
709 // |
710 // store x y | \x.{ commit of('store x y') }
711 // v
712 // y
713 //
714 if (CurrNode == SuccNode && CurrNode == Store->getPointerOperand()) {
715 // y obtains its value(s) from its original allocation and the store
716 // instruction under analysis.
717 IF_LOG_ENABLED({
718 PHASAR_LOG_LEVEL(DFADEBUG,
719 "Const-Replace at '" << llvmIRToString(Store));
720 PHASAR_LOG_LEVEL(DFADEBUG, "Replacement label(s): ");
721 for (const auto &Item : UserEdgeFacts.assertGetValue()) {
722 PHASAR_LOG_LEVEL(DFADEBUG, Item << ", ");
723 }
724 PHASAR_LOG_LEVEL(DFADEBUG, '\n');
725 });
726 // obtain label from the original allocation
727 return IIAAKillOrReplaceEFCache.createEdgeFunction(
728 std::move(UserEdgeFacts));
729 }
730
731 } else {
732 // Use points-to information to find all possible overriding edges.
733
734 // Kill all labels that are propagated along the edge of the
735 // value/values that is/are overridden.
736 //
737 // y --> y
738 //
739 // Edge function:
740 //
741 // y
742 // |
743 // store x y | \x.{}
744 // v
745 // y
746 //
747 if (CurrNode == SuccNode && (Store->getPointerOperand() == CurrNode ||
748 this->PT.isInReachableAllocationSites(
749 Store->getPointerOperand(), CurrNode,
750 OnlyConsiderLocalAliases))) {
751 return IIAAKillOrReplaceEFCache.createEdgeFunction(BitVectorSet<e_t>());
752 }
753 }
754
755 if (CurrNode == SuccNode) {
756 // Everything unrelated to the store
757 return EdgeIdentity<l_t>{};
758 }
759
760 llvm::report_fatal_error(
761 llvm::Twine("Unhandled edge: \n> Store: ") + llvmIRToString(Store) +
762 "\n> CurrNode: " + CurrNode.str() + "\n> SuccNode: " + SuccNode.str());

766 n_t /* Succ */,

770 << llvmIRToString(Curr) << "--> "
771 << llvmIRToShortString(SuccNode));

775 return EdgeIdentity<l_t>{};

817 inline EdgeFunctionType getCallEdgeFunction(n_t CallSite, d_t SrcNode,
818 f_t /* DestinationMethod */,
819 d_t DestNode) override {
820 // Handle the case in which a parameter that has been artificially
821 // introduced by the compiler is passed. Such a value must be generated from
822 // the zero value, to reflact the fact that the data flows from the callee
823 // to the caller (at least) according to the source code.
824 //
825 // Let a_i be an argument that is annotated by the sret attribute.
826 //
827 // 0 --> a_i
828 //
829 // Edge function:
830 //
831 // 0
832 // \
833 // call/invoke f(a_i) \ \x.{}
834 // v
835 // a_i
836 //
837 std::set<d_t> SRetParams;
838 const auto *CS = llvm::cast<llvm::CallBase>(CallSite);
839 for (unsigned Idx = 0; Idx < CS->arg_size(); ++Idx) {
840 if (CS->paramHasAttr(Idx, llvm::Attribute::StructRet)) {
841 SRetParams.insert(CS->getArgOperand(Idx));
842 }
843 }
844 if (isZeroValue(SrcNode) && SRetParams.count(DestNode)) {
845 return IIAAKillOrReplaceEFCache.createEdgeFunction();
846 }
847 // Everything else can be passed as identity.
848 return EdgeIdentity<l_t>{};

882 inline EdgeFunctionType
883 getCallToRetEdgeFunction(n_t CallSite, d_t CallNode, n_t /* RetSite */,
884 d_t RetSiteNode,
885 llvm::ArrayRef<f_t> Callees) override {
886 // Check if the user has registered a fact generator function
887 l_t UserEdgeFacts = bvSetFrom(invoke_or_default(EdgeFactGen, CallSite));
888
889 // Model call to heap allocating functions (new, new[], malloc, etc.) --
890 // only model direct calls, though.
891 if (Callees.size() == 1) {
892 const auto *Callee = Callees.front();
893
894 if (this->ICF->isHeapAllocatingFunction(Callee)) {
895 // Let H be a heap allocating function.
896 //
897 // 0 --> x
898 //
899 // Edge function:
900 //
901 // 0
902 // \
903 // %i = call H \ \x.x \cup { commit of('%i = call H') }
904 // v
905 // i
906 //
907 if (isZeroValue(CallNode) && RetSiteNode == CallSite) {
908 return IIAAKillOrReplaceEFCache.createEdgeFunction(
909 std::move(UserEdgeFacts));
910 }
911 }
912 }
913 // Capture interactions of the call instruction and its arguments.
914 const auto *CS = llvm::dyn_cast<llvm::CallBase>(CallSite);
915 for (const auto &Arg : CS->args()) {
916 //
917 // o_i --> o_i
918 //
919 // Edge function:
920 //
921 // o_i
922 // |
923 // %i = call o_i | \ \x.x \cup { commit of('%i = call H') }
924 // v
925 // o_i
926 //
927 if (CallNode == Arg && CallNode == RetSiteNode) {
928 return IIAAAddLabelsEFCache.createEdgeFunction(
929 std::move(UserEdgeFacts));
930 }
931 }
932 // Otherwise stick to identity
933 return EdgeIdentity<l_t>{};
