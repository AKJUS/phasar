24template <typename ContainerTy,
25 typename = std::enable_if_t<std::is_same_v<
26 typename ContainerTy::value_type, const llvm::Value *>>>
27void collectGeneratedFacts(ContainerTy &Dest, const LLVMTaintConfig &Config,
28 const llvm::CallBase *CB,
29 const llvm::Function *Callee) {
30 const auto &Callback = Config.getRegisteredSourceCallBack();
31 if (Callback) {
32 auto CBFacts = Callback(CB);
33 Dest.insert(CBFacts.begin(), CBFacts.end());
34 }
35
36 if (Config.isSource(CB)) {
37 Dest.insert(CB);
38 }
39
40 for (unsigned I = 0, End = Callee->arg_size(); I < End; ++I) {
41 if (Config.isSource(Callee->getArg(I))) {
42 Dest.insert(CB->getArgOperand(I));
43 }
44 }
45}

47template <typename ContainerTy, typename Pred,
48 typename = std::enable_if_t<std::is_same_v<
49 typename ContainerTy::value_type, const llvm::Value *>>>
50void collectLeakedFacts(ContainerTy &Dest, const LLVMTaintConfig &Config,
51 const llvm::CallBase *CB, const llvm::Function *Callee,
52 Pred &&LeakIf) {
53
54 const auto &Callback = Config.getRegisteredSinkCallBack();
55 if (Callback) {
56 auto CBLeaks = Callback(CB);
57 std::copy_if(CBLeaks.begin(), CBLeaks.end(),
58 std::inserter(Dest, Dest.end()), LeakIf);
59 }
60
61 for (unsigned I = 0, End = Callee->arg_size(); I < End; ++I) {
62 if (Config.isSink(Callee->getArg(I)) && LeakIf(CB->getArgOperand(I))) {
63 Dest.insert(CB->getArgOperand(I));
64 }
65 }
66}

68template <typename ContainerTy>

70 const llvm::CallBase *CB,
71 const llvm::Function *Callee) {

73 [](const llvm::Value * /*V*/) { return true; });

77 typename = std::enable_if_t<std::is_same_v<
78 typename ContainerTy::value_type, const llvm::Value *>>>
79void collectSanitizedFacts(ContainerTy &Dest, const LLVMTaintConfig &Config,
80 const llvm::CallBase *CB,
81 const llvm::Function *Callee) {
82 for (unsigned I = 0, End = Callee->arg_size(); I < End; ++I) {
83 if (Config.isSanitizer(Callee->getArg(I))) {
84 Dest.insert(CB->getArgOperand(I));
85 }
