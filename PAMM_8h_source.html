37/// This class offers functionality to measure different performance metrics.
38/// All relevant functions are wrapped into preprocessor macros and should only
39/// be used through those macros. Using these macros allows us to disable PAMM
40/// completely when no performance evaluation is needed. Macros are defined
41/// in @see PAMMMacros.h
42///
43/// Currently, PAMM can be run with all metrics (severity level 2 = Full) or
44/// only core metrics (severity level 1 = Core) enabled. The severity level can
45/// be changed when building PhASAR. The CMake option is
46/// -DPHASAR_ENABLE_PAMM=[Off/Core/Full]. Note that PAMM will be disabled
47/// (severity level 0 = Off) when building and running unittests.
48///
49/// For better compile times it is advised to include @see PAMMMacros.h instead
50/// of PAMM.h.
51///
52/// @brief This class offers functionality to assist a performance analysis of
53/// the PhASAR framework.
54/// @note This class implements the Singleton Pattern - use the
55/// PAMM_GET_INSTANCE macro to retrieve an instance of PAMM before you use any
56/// other macro from this class.
57class PAMM final {
58public:
59 using TimePoint_t = std::chrono::high_resolution_clock::time_point;
60 using Duration_t = std::chrono::milliseconds;
61
62 PAMM() noexcept = default;
63 ~PAMM() = default;
64 // PAMM is used as singleton.
65 PAMM(const PAMM &PM) = delete;
66 PAMM(PAMM &&PM) = delete;
67 PAMM &operator=(const PAMM &PM) = delete;
68 PAMM &operator=(PAMM &&PM) = delete;

72 [[nodiscard]] static PAMM &getInstance();

74 /// \brief Resets PAMM, i.e. discards all gathered information (timer, counter
75 /// etc.) - associated macro: RESET_PAMM.
76 /// \note Only used for unit testing to reset PAMM in between test runs.
77 void reset();

79 /// \brief Starts a timer under the given timer id - associated macro:
80 /// START_TIMER(TIMER_ID, SEV_LVL).
81 /// \param TimerId Unique timer id.
82 void startTimer(llvm::StringRef TimerId);
83
84 /// \brief Resets timer under the given timer id - associated macro:
85 /// RESET_TIMER(TIMER_ID, SEV_LVL).
86 /// \param TimerId Unique timer id.
87 void resetTimer(llvm::StringRef TimerId);
88
89 /// If pauseTimer is true, a running timer gets paused, its start time point
90 /// will paired with a current time point, and stored as an accumulated timer.
91 /// This enables us to repeatedly compute execution time for a certain portion
92 /// of code which is executed multiple times, e.g. a loop or a function
93 /// call, without using a different timer id for every time computation.
94 /// Times of all executions of one timer are saved as distinct time point
95 /// pairs. Associated macro:
96 /// PAUSE_TIMER(TIMER_ID, SEV_LVL)
97 ///
98 /// Otherwise, the timer will be simply stopped. Associated macro:
99 /// STOP_TIMER(TIMER_ID, SEV_LVL)
100 /// \brief Stops or pauses a timer under the given timer id.
101 /// \param TimerId Unique timer id.
102 /// \param PauseTimer If true, timer will be paused instead of stopped.
103 void stopTimer(llvm::StringRef TimerId, bool PauseTimer = false);
104
105 /// \brief Computes the elapsed time of the given timer up until now or up to
106 /// the moment the timer was stopped - associated macro: GET_TIMER(TIMERID)
107 /// \param TimerId Unique timer id.
108 /// \return Timer duration.
109 uint64_t elapsedTime(llvm::StringRef TimerId);
110
111 /// For each accumulated timer a vector holds all recorded durations.
112 /// \brief Computes the elapsed time for all accumulated timer being used.
113 /// \return Map containing measured durations of all accumulated timer.
114 [[nodiscard]] llvm::StringMap<std::vector<uint64_t>>
115 elapsedTimeOfRepeatingTimer();
116
117 /// A running timer will not be stopped. The precision for time computation
118 /// is set to milliseconds and the output is similar to a timestamp, e.g.
119 /// '4h 8m 15sec 16ms'.
120 ///
121 /// Associated macro PRINT_TIMER(TIMERID) does not check PAMM's severity level
122 /// explicitly.
123 /// \brief Returns the elapsed time for a given timer id.
124 /// \param timerId Unique timer id.
125 [[nodiscard]] static std::string getPrintableDuration(uint64_t Duration);
126
127 /// \brief Registers a new counter under the given counter id - associated
128 /// macro: REG_COUNTER(COUNTER_ID, INIT_VALUE, SEV_LVL).
129 /// \param CounterId Unique counter id.
130 void regCounter(llvm::StringRef CounterId, unsigned IntialValue = 0);
131
132 /// \brief Increases the count for the given counter - associated macro:
133 /// INC_COUNTER(COUNTER_ID, VALUE, SEV_LVL).
134 /// \param CounterId Unique counter id.
135 /// \param CValue to be added to the current counter.
136 void incCounter(llvm::StringRef CounterId, unsigned CValue = 1);
137
138 /// \brief Decreases the count for the given counter - associated macro:
139 /// DEC_COUNTER(COUNTER_ID, VALUE, SEV_LVL).
140 /// \param CounterId Unique counter id.
141 /// \param CValue to be subtracted from the current counter.
142 void decCounter(llvm::StringRef CounterId, unsigned CValue = 1);
143
144 /// The associated macro does not check PAMM's severity level explicitly.
145 /// \brief Returns the current count for the given counter - associated macro:
146 /// GET_COUNTER(COUNTER_ID).
147 /// \param CounterId Unique counter id.
148 std::optional<unsigned> getCounter(llvm::StringRef CounterId);
149
150 /// The associated macro does not check PAMM's severity level explicitly.
151 /// \brief Sums the counts for the given counter ids - associated macro:
152 /// GET_SUM_COUNT(...).
153 /// \param CounterIds Unique counter ids.
154 /// \note Macro uses variadic parameters, e.g. GET_SUM_COUNT({"foo", "bar"}).
155 std::optional<uint64_t> getSumCount(const std::set<std::string> &CounterIds);
156 std::optional<uint64_t>
157 getSumCount(llvm::ArrayRef<llvm::StringRef> CounterIds);
158 std::optional<uint64_t>
159 getSumCount(std::initializer_list<llvm::StringRef> CounterIds);
160
161 /// \brief Registers a new histogram - associated macro:
162 /// REG_HISTOGRAM(HISTOGRAM_ID, SEV_LVL).
163 /// \param HistogramId Unique hitogram id.
164 void regHistogram(llvm::StringRef HistogramId);
165
166 /// \brief Adds a new observed data point to the corresponding histogram -
167 /// associated macro: ADD_TO_HISTOGRAM(HISTOGRAM_ID, DATAPOINT_ID,
168 /// DATAPOINT_VALUE, SEV_LVL).
169 /// \param HistogramId ID of the histogram that tracks given data points.
170 /// \param DataPointId ID of the given data point.
171 /// \param DataPointValue Value of the given data point.
172 void addToHistogram(llvm::StringRef HistogramId, llvm::StringRef DataPointId,
173 uint64_t DataPointValue = 1);
174
175 void stopAllTimers();
176
177 void printTimers(llvm::raw_ostream &OS);
178
179 void printCounters(llvm::raw_ostream &OS);
180
181 void printHistograms(llvm::raw_ostream &OS);
182
183 /// \brief Prints the measured data to the commandline - associated macro:
184 /// PRINT_MEASURED_DATA
185 void printMeasuredData(llvm::raw_ostream &OS);
186
187 /// \brief Exports the measured data to JSON - associated macro:
188 /// EXPORT_MEASURED_DATA(PATH).
189 /// \param OutputPath to exported JSON file.
190 void exportMeasuredData(
191 const llvm::Twine &OutputPath,
192 llvm::StringRef ProjectId = "default-phasar-project",
193 const std::vector<std::string> *Modules = nullptr,
194 const std::vector<std::string> *DataFlowAnalyses = nullptr);
195
196 [[nodiscard]] const auto &getHistogram() const noexcept { return Histogram; }
197
198private:
199 llvm::StringMap<TimePoint_t> RunningTimer;
200 llvm::StringMap<std::pair<TimePoint_t, TimePoint_t>> StoppedTimer;
201 llvm::StringMap<std::vector<std::pair<TimePoint_t, TimePoint_t>>>
202 RepeatingTimer;
203 llvm::StringMap<unsigned> Counter;
204 llvm::StringMap<llvm::StringMap<uint64_t>> Histogram;
205};
