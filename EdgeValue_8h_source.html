118 size_t operator()(const psr::glca::EdgeValue &Val) const {
119 auto Hash = hash<int>()(Val.getKind());
120 int64_t AsInt;
121 double AsFloat;
122 string AsString;
123 if (Val.tryGetInt(AsInt)) {
124 return hash<int64_t>()(AsInt) * 31 + Hash;
125 }
126 if (Val.tryGetFP(AsFloat)) {
127 return hash<double>()(round(AsFloat)) * 31 + Hash;
128 }
129 if (Val.tryGetString(AsString)) {
130 return hash<string>()(AsString) * 31 + Hash;
131 }
132 return Hash;
133 }
