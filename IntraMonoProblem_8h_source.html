38/// \brief The analysis problem interface for intraprocedural monotone problems
39/// (solvable by the IntraMonoSolver). Create a subclass from this and override
40/// all pure-virtual functions to create your own mono analysis.
41template <typename AnalysisDomainTy> class IntraMonoProblem {
42public:
43 using n_t = typename AnalysisDomainTy::n_t;
44 using d_t = typename AnalysisDomainTy::d_t;
45 using f_t = typename AnalysisDomainTy::f_t;
46 using t_t = typename AnalysisDomainTy::t_t;
47 using v_t = typename AnalysisDomainTy::v_t;
48 using i_t = typename AnalysisDomainTy::i_t;
49 using c_t = typename AnalysisDomainTy::c_t;
50 using db_t = typename AnalysisDomainTy::db_t;
51 using mono_container_t = typename AnalysisDomainTy::mono_container_t;
52
53 using ProblemAnalysisDomain = AnalysisDomainTy;
54
55protected:
56 const ProjectIRDBBase<db_t> *IRDB;
57 const TypeHierarchy<t_t, f_t> *TH;
58 const CFGBase<c_t> *CF;
59 AliasInfoRef<v_t, n_t> PT;
60 std::vector<std::string> EntryPoints;
61 [[maybe_unused]] Soundness S = Soundness::Soundy;
62
63public:
64 using ConfigurationTy = HasNoConfigurationType;
65
66 /// An intraprocedural monotone problem generated from an intermediate
67 /// representation, a type hierarchy of said representation, a control flow
68 /// graph, points-to information and optionally a vector of entry points.
69 /// @param[in] IRDB A project IR database.
70 /// @param[in] TH A type hierarchy based on the given IRDB.
71 /// @param[in] CF A control flow graph based on the given IRDB.
72 /// @param[in] PT Points-to information based on the given IRDB.
73 /// @param[in] EntryPoints A vector of entry points. Provide at least one.
74 IntraMonoProblem(const ProjectIRDBBase<db_t> *IRDB,
75 const TypeHierarchy<t_t, f_t> *TH, const CFGBase<c_t> *CF,
76 AliasInfoRef<v_t, n_t> PT,
77 std::vector<std::string> EntryPoints = {})
78 : IRDB(IRDB), TH(TH), CF(CF), PT(PT),
79 EntryPoints(std::move(EntryPoints)) {}
80
81 virtual ~IntraMonoProblem() = default;
82
83 virtual mono_container_t normalFlow(n_t Inst, const mono_container_t &In) = 0;
84
85 virtual mono_container_t merge(const mono_container_t &Lhs,
86 const mono_container_t &Rhs) = 0;
87
88 virtual bool equal_to( // NOLINT - this would break client analyses

90
91 virtual mono_container_t allTop() { return mono_container_t{}; }
92
93 virtual std::unordered_map<n_t, mono_container_t> initialSeeds() = 0;
94
95 [[nodiscard]] const std::vector<std::string> &getEntryPoints() const {
96 return EntryPoints;
97 }
98
99 [[nodiscard]] const ProjectIRDBBase<db_t> *getProjectIRDB() const {
100 return IRDB;
101 }
102
103 [[nodiscard]] const TypeHierarchy<t_t, f_t> *getTypeHierarchy() const {
104 return TH;
105 }
106
107 [[nodiscard]] const CFGBase<c_t> *getCFG() const { return CF; }
108
109 [[nodiscard]] AliasInfoRef<v_t, n_t> getPointstoInfo() const { return PT; }
110
111 virtual bool setSoundness(Soundness /*S*/) { return false; }
112
113 virtual void printContainer(llvm::raw_ostream &OS, mono_container_t C) const {
114 }
115};
