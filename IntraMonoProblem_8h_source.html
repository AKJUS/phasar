40/// \brief The analysis problem interface for intraprocedural monotone problems
41/// (solvable by the IntraMonoSolver). Create a subclass from this and override
42/// all pure-virtual functions to create your own mono analysis.
43template <MonoAnalysisDomain AnalysisDomainTy> class IntraMonoProblem {
44public:
45 using n_t = typename AnalysisDomainTy::n_t;
46 using d_t = typename AnalysisDomainTy::d_t;
47 using f_t = typename AnalysisDomainTy::f_t;
48 using t_t = typename AnalysisDomainTy::t_t;
49 using v_t = typename AnalysisDomainTy::v_t;
50 using i_t = typename AnalysisDomainTy::i_t;
51 using c_t = typename AnalysisDomainTy::c_t;
52 using db_t = typename AnalysisDomainTy::db_t;
53 using mono_container_t = typename AnalysisDomainTy::mono_container_t;
54
55 using ProblemAnalysisDomain = AnalysisDomainTy;
56
57protected:
58 const db_t *IRDB;
59 const c_t *CF;
60 AliasInfoRef<v_t, n_t> PT;
61 std::vector<std::string> EntryPoints;
62 [[maybe_unused]] Soundness S = Soundness::Soundy;
63
64public:
65 using ConfigurationTy = HasNoConfigurationType;
66
67 /// An intraprocedural monotone problem generated from an intermediate
68 /// representation, a type hierarchy of said representation, a control flow
69 /// graph, points-to information and optionally a vector of entry points.
70 /// @param[in] IRDB A project IR database.
71 /// @param[in] TH A type hierarchy based on the given IRDB.
72 /// @param[in] CF A control flow graph based on the given IRDB.
73 /// @param[in] PT Points-to information based on the given IRDB.
74 /// @param[in] EntryPoints A vector of entry points. Provide at least one.
75 IntraMonoProblem(const db_t *IRDB, const c_t *CF, AliasInfoRef<v_t, n_t> PT,
76 std::vector<std::string> EntryPoints = {})
77 : IRDB(IRDB), CF(CF), PT(PT), EntryPoints(std::move(EntryPoints)) {}
78
79 virtual ~IntraMonoProblem() = default;
80
81 virtual mono_container_t normalFlow(n_t Inst, const mono_container_t &In) = 0;
82
83 virtual mono_container_t merge(const mono_container_t &Lhs,
84 const mono_container_t &Rhs) = 0;
85
86 virtual bool equal_to( // NOLINT - this would break client analyses
87 const mono_container_t &Lhs, const mono_container_t &Rhs) = 0;
88
89 virtual mono_container_t allTop() { return mono_container_t{}; }
90
91 virtual std::unordered_map<n_t, mono_container_t> initialSeeds() = 0;
92
93 [[nodiscard]] const std::vector<std::string> &getEntryPoints() const {
94 return EntryPoints;
95 }
96
97 [[nodiscard]] const db_t *getProjectIRDB() const { return IRDB; }
98
99 [[nodiscard]] const c_t *getCFG() const { return CF; }
100
101 [[nodiscard]] AliasInfoRef<v_t, n_t> getPointstoInfo() const { return PT; }
102
103 virtual bool setSoundness(Soundness /*S*/) { return false; }
104
105 virtual void printContainer(llvm::raw_ostream &OS, mono_container_t C) const {
106 }
107};
