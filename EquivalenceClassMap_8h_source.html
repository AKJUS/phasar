26/// \brief EquivalenceClassMap is a special map type that splits the keys into
27/// equivalence classes regarding their mapped values.
28///
29/// Meaning, that all keys that are equivalent are mapped to the same value. Two
30/// keys are treated as equivalent and merged into an equivalence class when
31/// they refer to Values that are considered equal according to operator==.
32template <typename KeyT, typename ValueT> struct EquivalenceClassMap {
33 template <typename... Ts> using SetType = std::set<Ts...>;
34 using EquivalenceClassBucketT = std::pair<SetType<KeyT>, ValueT>;
35 // Use SmallVector here, since it has a smaller struct-size than std::vector;
36 // we may store a lot of them in the FlowEdgeFunctionCache
37 using StorageT = llvm::SmallVector<EquivalenceClassBucketT, 0>;
38
39public:
40 using size_type = size_t;
41 using key_type = KeyT;
42 using mapped_type = ValueT;
43 using value_type = EquivalenceClassBucketT;
44
45 using const_iterator = typename StorageT::const_iterator;
46
47 using insert_return_type =
48 std::pair<typename SetType<KeyT>::const_iterator, bool>;
49
50 EquivalenceClassMap(unsigned InitialEquivalenceClasses = 0) {
51 StoredData.reserve(InitialEquivalenceClasses);
52 }
53
54 template <typename InputIt>
55 EquivalenceClassMap(const InputIt &I, const InputIt &End) {
56 this->insert(I, End);
57 }
58
59 EquivalenceClassMap(
60 std::initializer_list<std::pair<key_type, mapped_type>> Vals) {
61 this->insert(Vals.begin(), Vals.end());
62 }

64 [[nodiscard]] inline const_iterator begin() const {
65 return StoredData.begin();
66 }
67 [[nodiscard]] inline const_iterator end() const { return StoredData.end(); }
68 [[nodiscard]] inline llvm::iterator_range<const_iterator>
69 equivalenceClasses() const {
70 return llvm::make_range(begin(), end());
71 }

73 /// Inserts Key into the corresponding equivalence class for Value. If Value
74 /// is not already in the map a new equivalence class is created.
75 template <typename ValueType = ValueT>
76 insert_return_type insert(const KeyT &Key, ValueType &&Value) {
77 return try_emplace(Key, std::forward<ValueType>(Value));

99 /// Insert a range of Key Values pairs into the map.
100 template <typename InputIt> void insert(InputIt I, InputIt End) {
101 for (; I != End; ++I) {
102 try_emplace(I->first, I->second);
103 }
104 }
105
106 /// Inserts Key into the corresponding equivalence class for Value. If Value
107 /// is not already in the map a new equivalence class is created.
108 template <typename... Ts>
109 insert_return_type try_emplace(KeyT &&Key, Ts &&...Args) {
110 ValueT Val{std::forward<Ts...>(Args...)};
111
112 for (auto &KVPair : StoredData) {
113 if (KVPair.second == Val) {
114 return KVPair.first.insert(Key);
115 }
116 }
117
118 StoredData.emplace_back(SetType<KeyT>{std::move(Key)}, std::move(Val));
119 return std::make_pair(StoredData.back().first.begin(), true);
120 }
121
122 /// Inserts Key into the corresponding equivalence class for Value. If Value
123 /// is not already in the map a new equivalence class is created.
124 template <typename... Ts>
125 insert_return_type try_emplace(const KeyT &Key, Ts &&...Args) {
126 ValueT Val{std::forward<Ts...>(Args...)};
127
128 for (auto &KVPair : StoredData) {
129 if (KVPair.second == Val) {
130 return KVPair.first.insert(Key);
131 }
132 }
133
134 StoredData.emplace_back(SetType<KeyT>{Key}, std::move(Val));
135 return std::make_pair(StoredData.back().first.begin(), true);
136 }
137
138 /// Return 1 if the specified key is in the map, 0 otherwise.
139 [[nodiscard]] inline size_type count(const KeyT &Key) const {
140 for (auto &KVPair : StoredData) {
141 if (KVPair.first.count(Key) >= 1) {
142 return 1;
143 }
144 }
145 return 0;
146 }
147
148 [[nodiscard]] inline size_type numEquivalenceClasses() const {
149 return StoredData.size();
150 }
151
152 /// Returns the size of the map, i.e., the number of equivalence classes.
153 [[nodiscard]] inline size_type size() const {
154 return numEquivalenceClasses();
155 }
156
157 [[nodiscard]] const_iterator find(key_type Key) const {
158 return llvm::find_if(StoredData,
159 [&Key](const EquivalenceClassBucketT &Val) -> bool {
160 return Val.first.count(Key) >= 1;
161 });
162 }
163
164 [[nodiscard]] std::optional<ValueT> findValue(key_type Key) const {
165 auto Search = find(Key);
166 if (Search != StoredData.end()) {
167 return Search->second;
168 }
169 return std::nullopt;
170 }
171
172 inline void clear() { StoredData.clear(); }
173
174private:
175 StorageT StoredData{};
176};

234 template <typename KK, typename VV>
235 std::pair<const_iterator, bool> insert(KK &&Key, VV &&Value) {
236 ValueComparator VComp;
237 for (size_t I = 0, End = Values.size(); I != End; ++I) {
238 if (VComp(Values[I], Value)) {
239 return {getIterator(I), Keys[I].insert(std::forward<KK>(Key)).second};
240 }
241 }
242
243 Values.emplace_back(std::forward<VV>(Value));
244 Keys.emplace_back().insert(std::forward<KK>(Key));
245 return {getIterator(Values.size() - 1), true};
246 }
247
248 template <typename KK, typename VCtor>
249 const TValue &getOrInsertLazy(KK &&Key, VCtor &&MakeV) {
250 for (size_t I = 0, End = Keys.size(); I != End; ++I) {
251 if (Keys[I].count(Key)) {
252 return Values[I];
253 }
254 }
255 return (*insert(std::forward<KK>(Key), std::invoke(MakeV)).first).second;
256 }
257
258 const_iterator begin() const noexcept {
259 return {Values.begin(), Keys.begin()};
260 }
261 const_iterator end() const noexcept { return {Values.end(), Keys.end()}; }
262
263 const_iterator find(const TKey &Ky) const {
264 for (size_t I = 0, End = Keys.size(); I < End; ++I) {
265 if (Keys[I].count(Ky)) {
266 return getIterator(I);
267 }
268 }
269
270 return end();
271 }

273 [[nodiscard]] inline size_t numEquivalenceClasses() const noexcept {
274 return Values.size();
275 }
276
277 /// Returns the size of the map, i.e., the number of equivalence classes.
278 [[nodiscard]] inline size_t size() const noexcept {
279 return numEquivalenceClasses();
280 }
281
282 [[nodiscard]] bool empty() const noexcept { return Values.empty(); }
283
284 void clear() noexcept {
285 Values.clear();
286 Keys.clear();
287 }
288
289private:
290 const_iterator getIterator(size_t I) const noexcept {
291 return {std::next(Values.begin(), I), std::next(Keys.begin(), I)};
292 }

294 llvm::SmallVector<TValue, 2> Values;
295 llvm::SmallVector<SetTy, 2> Keys;
296};
