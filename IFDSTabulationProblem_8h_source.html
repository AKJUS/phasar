23/// \brief The analysis problem interface for IFDS problems (solvable by the
24/// IFDSSolver). Create a subclass from this and override all pure-virtual
25/// functions to create your own IFDS analysis.
26///
27/// For more information on how to write an IFDS analysis, see [Writing an IFDS
28/// Analysis](https://github.com/secure-software-engineering/phasar/wiki/Writing-an-IFDS-analysis)

30 typename Container = std::set<typename AnalysisDomainTy::d_t>>

48 /// Takes an IR database (IRDB) and collects information from it to create a
49 /// tabulation problem.
50 /// @param[in] IRDB The project IR database, that holds the code under
51 /// analysis
52 /// @param[in] EntryPoints The (mangled) names of all entry functions of the
53 /// target being analyzed, given as a vector of strings. An example would
54 /// simply be `{"main"}`. To set every function as entry point, pass
55 /// `"__ALL__"`
56 /// @param[in] ZeroValue Provides the special tautological zero value (aka.
57 /// Λ).
58 /// \endlink.
59 explicit IFDSTabulationProblem(const db_t *IRDB,
60 std::vector<std::string> EntryPoints,
61 d_t ZeroValue)
62 : Base(IRDB, std::move(EntryPoints), std::move(ZeroValue)) {}
