19template <typename T, unsigned NumOffs = 32> struct AlignNum {
20 llvm::StringRef Name;
21 T Num;
22
23 constexpr AlignNum(llvm::StringRef Name, T Num) noexcept
24 : Name(Name), Num(Num) {}
25 constexpr AlignNum(llvm::StringRef Name, size_t Numerator,
26 size_t Denominator) noexcept
27 : Name(Name), Num(double(Numerator) / double(Denominator)) {}
28
29 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
30 const AlignNum &AN) {
31 auto Len = AN.Name.size() + 1;
32 auto Diff = -(Len < NumOffs) & (NumOffs - Len);
33
34 OS << AN.Name << ':';
35 // Default is two fixed-point decimal places, so shift the output by three
36 // spaces
37 OS.indent(Diff + std::is_floating_point_v<T> * 3);
38 OS << llvm::formatv("{0,+7}\n", AN.Num);
39
40 return OS;
41 }
42};
43template <typename T> AlignNum(llvm::StringRef, T) -> AlignNum<T>;
44AlignNum(llvm::StringRef, size_t, size_t) -> AlignNum<double>;
