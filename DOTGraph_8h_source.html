86struct DOTNode {
87 /* stmt id = <func-name>_<stmt-id>
88 * e.g. foo_42 for the statement w/ id 42 in function foo
89 *
90 * fact id = <func-name>_<fact-id>_<stmt-id>
91 * e.g. fact i in main valid at statement 3 : main_1_3

165template <typename D> struct DOTGraph {
166 std::string Label;
167 std::map<std::string, DOTFunctionSubGraph> Functions;
168 std::set<DOTEdge> InterCFEdges;
169 std::set<DOTEdge> InterLambdaEdges;
170 std::set<DOTEdge> InterFactEdges;
171
172 DOTGraph() = default;
173
174 unsigned getFactID(D Fact) {
175 unsigned Id = 0;
176 if (DtoFactId.count(Fact)) {
177 Id = DtoFactId[Fact];
178 } else {
179 Id = FactIDCount;
180 DtoFactId[Fact] = FactIDCount++;
181 }
182 return Id;
183 }
184
185 bool containsFactSG(std::string &FName, unsigned FactId) {
186 if (Functions.count(FName)) {
187 if (Functions[FName].Facts.count(FactId)) {
188 return true;
189 }
190 }
191 return false;
192 }
193
194 [[nodiscard]] std::string str() const {
195 std::string Indent = " ";
196 std::string Str = "digraph {\n" + Indent + "label=\"" + Label + "\"\n";
197 // Print function subgraphs
198 Str += '\n' + Indent + "// Function sub graphs\n";
199 for (auto Fsg : Functions) {
200 Fsg.second.createLayoutCFNodes();
201 Fsg.second.createLayoutFactNodes();
202 Fsg.second.createLayoutFactEdges();
203 Str += Fsg.second.str(Indent) + "\n\n";
204 }
205
206 // Print inter control flow edges
207 Str += Indent + "// Inter-procedural control flow edges\n" + Indent +
208 DOTConfig::CFInterEdgeAttr() + '\n';
209 for (const DOTEdge &Edge : InterCFEdges) {
210 Str += Edge.str(Indent) + '\n';
211 }
212
213 // Print inter lambda edges
214 Str += '\n' + Indent + "// Inter-procedural lambda edges\n" + Indent +
215 DOTConfig::LambdaInterEdgeAttr() + '\n';
216 for (const DOTEdge &Edge : InterLambdaEdges) {
217 Str += Edge.str(Indent) + '\n';
218 }
219
220 // Print inter fact edges
221 Str += '\n' + Indent + "// Inter-procedural fact edges\n" + Indent +
222 DOTConfig::FactInterEdgeAttr() + '\n';
223 for (const DOTEdge &Edge : InterFactEdges) {
224 Str += Edge.str(Indent) + '\n';
225 }
226 return Str + '}';
227 }
228
229 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
230 const DOTGraph<D> &Graph) {
231 return OS << Graph.str();
232 }
233
234private:
235 // We introduce a fact-ID for data-flow facts D since only statements N have
236 // an ID
237 unsigned FactIDCount = 1;
238 std::map<D, unsigned> DtoFactId;
239};
