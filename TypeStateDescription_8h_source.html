38/**
39 * Interface for a type state problem to be used with the IDETypeStateAnalysis.
40 * It needs to provide a finite state machine to handle state changes and a list
41 * of functions that are relevant to the analyzed api.
42 * Factory functions are those functions which generate objects/variables of
43 * target type, e.g. file handles generated by fopen().
44 * Consuming functions operate on objects/variables of target type, e.g.
45 * fwrite(). Functions that are both factory and consumer, e.g. freopen(),
46 * cannot be modeled by this interface, since analyzing such functions correctly
47 * introduces dependencies that cannot be solved by a distributive framework
48 * such as IDE.
49 *
50 * @see CSTDFILEIOTypeStateDescription as an example of type state description.
51 */
52template <typename StateTy>
53struct TypeStateDescription : public TypeStateDescriptionBase {
54 /// Type for states of the finite state machine
55 using State = StateTy;
56 ~TypeStateDescription() override = default;
57
58 /**
59 * @brief For a given function name (as a string token) and a state, this
60 * function returns the next state.
61 */
62 [[nodiscard]] virtual State getNextState(llvm::StringRef Tok,
63 State S) const = 0;
64 [[nodiscard]] virtual State
65 getNextState(llvm::StringRef Tok, State S,
66 const llvm::CallBase * /*CallSite*/) const {
67 return getNextState(Tok, S);
68 }
69
70 [[nodiscard]] virtual State bottom() const = 0;
71 [[nodiscard]] virtual State top() const = 0;
72
73 /**
74 * Represents the uninitialized state of an object, e.g. allocation of a file
75 * handle
76 */
77 [[nodiscard]] virtual State uninit() const = 0;
78
79 /**
80 * Represents the error state of an object
81 */
82 [[nodiscard]] virtual State error() const = 0;
83};
