27/// Wraps a llvm::SmallVector, allowing index-based access by IdT, instead of
28/// size_t.
29///
30/// \tparam IdT The index-type that should be used for operator[]. Must be
31/// losslessly convertible from and to size_t.
32/// \tparam ValueT The usual value_type of SmallVector.
33/// \tparam SmallSize The size of the inline-storage of SmallVector (default:
34/// 0)
35template <typename IdT, typename ValueT, unsigned SmallSize = 0>
36class TypedVector {
37public:
38 TypedVector() noexcept = default;
39 TypedVector(std::initializer_list<ValueT> IList) : Vec(IList) {}
40 TypedVector(size_t Size) : Vec(Size) {}
41 TypedVector(size_t Size, ByConstRef<ValueT> Default) : Vec(Size, Default) {};
42
43 template <typename Iter>
44 explicit TypedVector(Iter From, Iter To)
45 : Vec(std::move(From), std::move(To)) {}
46
47 explicit TypedVector(llvm::ArrayRef<ValueT> Arr)
48 : Vec(Arr.begin(), Arr.end()) {}
49
50 void reserve(size_t Capa) { Vec.reserve(Capa); }
51
52 void resize(size_t Sz) { Vec.resize(Sz); }
53
54 void resize(size_t Sz, ByConstRef<ValueT> Val) { Vec.resize(Sz, Val); }
55
56 [[nodiscard]] bool empty() const noexcept { return Vec.empty(); }
57 [[nodiscard]] bool any() const noexcept { return !Vec.empty(); }
58 [[nodiscard]] size_t size() const noexcept { return Vec.size(); }
59 [[nodiscard]] size_t capacity() const noexcept { return Vec.capacity(); }
60
61 [[nodiscard]] bool inbounds(IdT Id) const noexcept {
62 return size_t(Id) < size();
63 }
64
65 [[nodiscard]] const ValueT &operator[](IdT Id) const & {
66 assert(inbounds(Id));

68 }

70 [[nodiscard]] ValueT &operator[](IdT Id) & {

73 }

118 [[nodiscard]] auto enumerate() const noexcept {
119 return llvm::map_range(llvm::enumerate(Vec), [](const auto &IndexAndVal) {
120 return std::pair<IdT, ByConstRef<ValueT>>{IdT(IndexAndVal.index()),
121 IndexAndVal.value()};
122 });
123 }
124 [[nodiscard]] auto enumerate() noexcept {

126 return std::pair<IdT, ValueT &>{IdT(IndexAndVal.index()),
127 IndexAndVal.value()};

129 }

131private:
