17/// AnalysisDomain - This class should be specialized by different static
18/// analyses types... which is why the default version declares all analysis
19/// domains as aliases of void.
20///
21/// Virtually all of PhASAR's internal analyses are implemented in a generic way
22/// using interfaces and template parameters. In order to specify concrete types
23/// for the template parameters such that an analysis can compute some useful
24/// information on some concrete target code, a configuration template parameter
25/// of type AnalysisDomain is passed around to make the necessary information
26/// available to the required analyses.
27///
28/// If a type is not meant to be used by an analysis it should be left as an
29/// alias to void. If any analysis detects that a parameter is required to
30/// conduct an analysis but not correctly set, it will statically report an
31/// error and ask for the missing piece of information.
32struct AnalysisDomain {
33 /// Data-flow fact --- Specifies the type of an individual data-flow fact that
34 /// is propagated through the program under analysis.
35 using d_t = void;
36 // (Control-flow) Node --- Specifies the type of a node in the
37 // (inter-procedural) control-flow graph and can be though of as an individual
38 // statement or instruction of the target program.
39 using n_t = void;
40 // Function --- Specifies the type of functions/procedures in the target
41 // program.
42 using f_t = void;
43 // (User-defined) type --- Specifies the type of a user-defined (i.e. struct
44 // or class) data type.
45 using t_t = void;
46 // (Pointer) value --- Specifies the type of pointers.
47 using v_t = void;
48 // Intra-procedural control flow --- Specifies the type of the
49 // control-flow graph to be used.
50 using c_t = void;
51 // Inter-procedural control flow --- Specifies the type of the
52 // inter-procedural control-flow graph to be used.
53 using i_t = void;
54 // The ProjectIRDB type to use. Must inherit from the ProjectIRDBBase CRTP
55 // template
56 using db_t = void;
57 // Lattice element --- Specifies the type of the underlying lattice; the value
58 // computation domain IDE's edge functions or WPDS's weights operate on.
59 using l_t = void;
60 // Container type to be used for analyses run in the monotone framework.
61 using mono_container_t = void;
62};
