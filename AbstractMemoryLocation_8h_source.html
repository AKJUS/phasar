47/// \brief A Memorylocation abstraction represented by a base-pointer and an

146/// A Wrapper over a pointer to an detail::AbstractMemoryLocationImpl. The impl
147/// can be accessed using the -> operator. In contrast to the impl, the equality
148/// operators only compare the Impl-references.
149///
150/// This type can be hashed.
151class AbstractMemoryLocation {
152public:
153 explicit AbstractMemoryLocation() noexcept = default;
154
155 AbstractMemoryLocation(const AbstractMemoryLocation &) noexcept = default;
156 AbstractMemoryLocation(AbstractMemoryLocation &&) noexcept = default;
157 ~AbstractMemoryLocation() = default;
158
159 AbstractMemoryLocation &
160 operator=(const AbstractMemoryLocation &) noexcept = default;
161 AbstractMemoryLocation &
162 operator=(AbstractMemoryLocation &&) noexcept = default;
163
164 AbstractMemoryLocation(
165 const detail::AbstractMemoryLocationImpl *Impl) noexcept;
166
167 inline const detail::AbstractMemoryLocationImpl *operator->() const {
168 return PImpl;
169 }
170
171 /// Provide an arbitrary partial order for being able to store TaintedValues
172 /// in std::set or as key in std::map
173 inline bool operator<(AbstractMemoryLocation TV) const {
174 return PImpl < TV.PImpl;
175 }
176
177 inline bool operator==(AbstractMemoryLocation AML) const {
178 return PImpl == AML.PImpl;
179 }
180
181 inline bool operator!=(AbstractMemoryLocation AML) const {
182 return !(*this == AML);
183 }
184
185 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
186 AbstractMemoryLocation TV);
187
188 /// Computes the absolute offset-difference between this and TV assuming,
189 /// either this->isProperPrefixOf(TV) or vice versa.
190 [[nodiscard]] inline llvm::ArrayRef<ptrdiff_t>
191 operator-(AbstractMemoryLocation TV) const {
192 return *PImpl - *TV.PImpl;
193 }
194
195 operator const detail::AbstractMemoryLocationImpl &() const { return *PImpl; }
196
197private:
198 const detail::AbstractMemoryLocationImpl *PImpl = nullptr;
199};
