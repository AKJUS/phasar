25/// FIXME: This is not entirely correct: Does not skip ignored statements and

27/// The right way would be to ask the ICFG, but we don't have a reference to it

29template <typename Derived, typename N, typename D, typename L>
30template <typename NTy>
31auto SolverResultsBase<Derived, N, D, L>::resultsAtInLLVMSSA(
32 ByConstRef<n_t> Stmt, bool AllowOverapproximation, bool StripZero) const ->
33 typename std::enable_if_t<
34 std::is_same_v<std::decay_t<std::remove_pointer_t<NTy>>,
35 llvm::Instruction>,
36 std::unordered_map<d_t, l_t>> {
37 std::unordered_map<d_t, l_t> Result = [this, Stmt, AllowOverapproximation]() {
38 if (Stmt->getType()->isVoidTy()) {
39 return self().Results.row(Stmt);
40 }
41 if (!Stmt->getNextNode()) {
42 auto GetStartRow = [this](const llvm::BasicBlock *BB) -> decltype(auto) {
43 const auto *First = &BB->front();
44 if (llvm::isa<llvm::DbgInfoIntrinsic>(First)) {
45 First = First->getNextNonDebugInstruction();
46 }
47 return self().Results.row(First);
48 };
49
50 // We have reached the end of a BasicBlock. If there is a successor BB
51 // that only has one predecessor, we are lucky and can just take results
52 // from there
53 for (const llvm::BasicBlock *Succ : llvm::successors(Stmt)) {
54 if (Succ->hasNPredecessors(1)) {
55 return GetStartRow(Succ);
56 }
57 }
58
59 if (!AllowOverapproximation) {
60 llvm::report_fatal_error("[resultsAtInLLVMSSA]: Cannot precisely "
61 "collect the results at instruction " +
62 llvm::Twine(llvmIRToString(Stmt)));
63 }
64
65 // There is no successor with only one predecessor.
66 // All we can do is merge the results from all successors to get a sound
67 // overapproximation. This is not optimal and may be replaced in the
68 // future.
69 PHASAR_LOG_LEVEL(WARNING, "[resultsAtInLLVMSSA]: Cannot precisely "
70 "collect the results at instruction "
71 << llvmIRToString(Stmt)
72 << ". Use a sound, but potentially "
73 "imprecise overapproximation");
74 std::unordered_map<d_t, l_t> Ret;
75 for (const llvm::BasicBlock *Succ : llvm::successors(Stmt)) {
76 const auto &Row = GetStartRow(Succ);
77 for (const auto &[Fact, Value] : Row) {
78 auto [It, Inserted] = Ret.try_emplace(Fact, Value);
79 if (!Inserted && Value != It->second) {
80 if constexpr (HasJoinLatticeTraits<l_t>) {
81 It->second = JoinLatticeTraits<l_t>::join(It->second, Value);
82 } else {
83 // We have no way of correctly merging, so set the value to the
84 // default constructed l_t hoping it marks BOTTOM.
85 It->second = l_t();
86 }
87 }
88 }
89 }
90 return Ret;
91 }
92 assert(Stmt->getNextNode() && "Expected to find a valid successor node!");
93 return self().Results.row(Stmt->getNextNode());
94 }();
95 if (StripZero) {
96 Result.erase(self().ZV);
97 }
98 return Result;
99}

101template <typename Derived, typename N, typename D, typename L>
102template <typename NTy>
103auto SolverResultsBase<Derived, N, D, L>::resultAtInLLVMSSA(
104 ByConstRef<n_t> Stmt, d_t Value, bool AllowOverapproximation) const ->
105 typename std::enable_if_t<
106 std::is_same_v<std::decay_t<std::remove_pointer_t<NTy>>,
107 llvm::Instruction>,
108 l_t> {
109 if (Stmt->getType()->isVoidTy()) {
110 return self().Results.get(Stmt, Value);
111 }
112 if (!Stmt->getNextNode()) {
113 auto GetStartVal = [this,
114 &Value](const llvm::BasicBlock *BB) -> decltype(auto) {
115 const auto *First = &BB->front();
116 if (llvm::isa<llvm::DbgInfoIntrinsic>(First)) {
117 First = First->getNextNonDebugInstruction();
118 }
119 return self().Results.get(First, Value);
120 };
121
122 // We have reached the end of a BasicBlock. If there is a successor BB
123 // that only has one predecessor, we are lucky and can just take results
124 // from there
125 for (const llvm::BasicBlock *Succ : llvm::successors(Stmt)) {
126 if (Succ->hasNPredecessors(1)) {
127 return GetStartVal(Succ);
128 }
129 }
130
131 if (!AllowOverapproximation) {
132 llvm::report_fatal_error("[resultsAtInLLVMSSA]: Cannot precisely "
133 "collect the results at instruction " +
134 llvm::Twine(llvmIRToString(Stmt)));
135 }
136
137 // There is no successor with only one predecessor.
138 // All we can do is merge the results from all successors to get a sound
139 // overapproximation. This is not optimal and may be replaced in the
140 // future.
141 PHASAR_LOG_LEVEL(WARNING, "[resultAtInLLVMSSA]: Cannot precisely "
142 "collect the results at instruction "
143 << *Stmt
144 << ". Use a sound, but potentially "
145 "imprecise overapproximation");
146 auto It = llvm::succ_begin(Stmt);
147 auto End = llvm::succ_end(Stmt);
148 l_t Ret{};
149 if (It != End) {
150 Ret = GetStartVal(*It);
151 for (++It; It != End; ++It) {
152 const auto &Val = GetStartVal(*It);
153 if constexpr (HasJoinLatticeTraits<l_t>) {
154 Ret = JoinLatticeTraits<l_t>::join(Ret, Val);
155 if (Ret == JoinLatticeTraits<l_t>::bottom()) {
156 break;
157 }
158 } else {
159 if (Ret != Val) {
160 // We have no way of correctly merging, so set the value to the
161 // default constructed l_t hoping it marks BOTTOM.
162 return {};
163 }
164 }
165 }
166 }
167 return Ret;
168 }
169 assert(Stmt->getNextNode() && "Expected to find a valid successor node!");
170 return self().Results.get(Stmt->getNextNode(), Value);
171}
