19/// \brief A simple custom allocator that is used to allocate, e.g.,
20/// AbstractMemoryLocation for the IDEExtendedTaintAnalysis.
21///
22/// It can be used to allocate trivially destructible objects of variable size,
23/// e.g., using llvm::TrailingObjects, that should live as long as the
24/// allocator.
25class MemoryLocationAllocator {
26public:
27 constexpr MemoryLocationAllocator() noexcept = default;
28
29 explicit MemoryLocationAllocator(size_t InitialCapacity,
30 size_t DynamicBlockSize = size_t(1024) * 6 -
31 8);
32
33 constexpr MemoryLocationAllocator(MemoryLocationAllocator &&Other) noexcept
34 : Root(Other.Root), Pos(Other.Pos), End(Other.End) {
35 Other.Root = nullptr;
36 Other.Pos = nullptr;
37 Other.End = nullptr;
38 }
39 MemoryLocationAllocator &operator=(MemoryLocationAllocator &&Other) noexcept {
40 MemoryLocationAllocator(std::move(Other)).swap(*this);
41 return *this;
42 }
43
44 ~MemoryLocationAllocator();
45
46 MemoryLocationAllocator(const MemoryLocationAllocator &) = delete;
47 MemoryLocationAllocator &operator=(const MemoryLocationAllocator &) = delete;
48
49 void swap(MemoryLocationAllocator &Other) noexcept {
50 std::swap(Root, Other.Root);
51 std::swap(Pos, Other.Pos);
52 std::swap(End, Other.End);
53 }
54
55 /// \brief Allocates a chunk of memory of at least NumBytes bytes, aligned to
56 /// alignof(void *).
57 ///
58 /// Fails with llvm::report_bad_alloc_error(), if memory could not be
59 /// allocated.
60 [[nodiscard]] LLVM_ATTRIBUTE_RETURNS_NONNULL void *allocate(size_t NumBytes);
61
62private:
63 struct Block final : public llvm::TrailingObjects<Block, void *> {
64
65 Block *Next = nullptr;
66
67 static Block *create(Block *Next, size_t NumPointerEntries);
68 static void destroy(Block *Blck, size_t NumPointerEntries);
69
70 private:
71 constexpr Block(Block *Next) noexcept : Next(Next) {}
72 };
73
74 Block *Root = nullptr;
75 void **Pos = nullptr;
76 void **End = nullptr;
77 size_t InitialCapacity{};
78 size_t DynamicBlockSize{};
79};
80} // namespace psr
