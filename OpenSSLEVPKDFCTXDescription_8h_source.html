50/**
51 * A type state description for OpenSSL's EVP Key Derivation functions. The
52 * finite state machine is encoded by a two-dimensional array with rows as
53 * function tokens and columns as states.
54 */

58 /**
59 * The STAR token represents all functions besides EVP_KDF_fetch(),

62 */
63 enum class OpenSSLEVTKDFToken {
64 EVP_KDF_CTX_NEW = 0,
65 EVP_KDF_CTX_SET_PARAMS = 1,
66 DERIVE = 2,
