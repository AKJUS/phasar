39template <typename OS_t, typename T> void printHelper(OS_t &OS, const T &Data) {
40 using ElemTy = std::decay_t<T>;
41 using BaseElemTy = std::decay_t<std::remove_pointer_t<T>>;
42
43 using OSTy = std::decay_t<OS_t>;
44
45 if constexpr (std::is_base_of_v<llvm::Value, BaseElemTy>) {
46 OS << llvmIRToString(Data);
47 } else if constexpr (std::is_base_of_v<llvm::Type, BaseElemTy>) {
48 if constexpr (std::is_base_of_v<llvm::raw_ostream, std::decay_t<OSTy>>) {
49 Data->print(OS);
50 } else {
51 std::string Str;
52 llvm::raw_string_ostream SOS(Str);
53 Data->print(SOS);
54 OS << SOS.str();
55 }
56 } else if constexpr (is_printable_v<ElemTy, OS_t>) {
57 OS << Data;
58 } else if constexpr (is_pair_v<ElemTy>) {
59 OS << "(";
60 printHelper(OS, Data.first);
61 OS << ", ";
62 printHelper(OS, Data.second);
63 OS << ")";
64 } else if constexpr (is_tuple_v<ElemTy>) {
65 printTuple(OS, Data, std::make_index_sequence<std::tuple_size_v<ElemTy>>());
66 } else if constexpr (has_str_v<ElemTy>) {
67 OS << Data.str();
68 } else if constexpr (is_iterable_v<ElemTy>) {
69 OS << "{ ";
70 bool Frst = true;
71 for (auto &&Elem : Data) {
72 if (Frst) {
73 Frst = false;
74 } else {
75 OS << ", ";
76 }
77
78 printHelper(OS, Elem);
79 }
80
81 OS << (Frst ? "}" : " }");
82 } else {
83 static_assert(!std::is_same_v<std::decay_t<T>, ElemTy>,
84 "Cannot print elements of the specified type");
85 }
86}
87} // namespace detail
