40template <typename D, typename FFTy> struct AutoAddZeroFF {
41 FFTy FF;
42 D ZeroValue;
43
44 auto computeTargets(ByConstRef<D> Source) {
45 auto Ret = FF.computeTargets(Source);
46 if (Source == ZeroValue) {
47 Ret.insert(ZeroValue);
48 }
49 return Ret;
50 }

96template <typename ProblemTy> struct FlowFunctionCacheBase {
97 using domain_t = typename ProblemTy::ProblemAnalysisDomain;
98 using d_t = typename domain_t::d_t;
99 using n_t = typename domain_t::n_t;
100 using f_t = typename domain_t::f_t;
101 using normal_ff_t =
102 std::decay_t<decltype(std::declval<ProblemTy>().getNormalFlowFunction(
103 std::declval<n_t>(), std::declval<n_t>()))>;
104 using call_ff_t =
105 std::decay_t<decltype(std::declval<ProblemTy>().getCallFlowFunction(
106 std::declval<n_t>(), std::declval<f_t>()))>;
107 using ret_ff_t =
108 std::decay_t<decltype(std::declval<ProblemTy>().getRetFlowFunction(
109 std::declval<n_t>(), std::declval<f_t>(), std::declval<n_t>(),
110 std::declval<n_t>()))>;
111 using ctr_ff_t =
112 std::decay_t<decltype(std::declval<ProblemTy>().getCallToRetFlowFunction(
113 std::declval<n_t>(), std::declval<n_t>(),
114 std::declval<llvm::ArrayRef<f_t>>()))>;
115 using summary_ff_t =
116 std::decay_t<decltype(std::declval<ProblemTy>().getSummaryFlowFunction(
117 std::declval<n_t>(), std::declval<f_t>()))>;
118
119 static_assert(detail::IsFlowFunction<normal_ff_t, d_t> ||
120 detail::IsFlowFunctionPtr<normal_ff_t, d_t>);
121 static_assert(detail::IsFlowFunction<call_ff_t, d_t> ||
122 detail::IsFlowFunctionPtr<call_ff_t, d_t>);
123 static_assert(detail::IsFlowFunction<ret_ff_t, d_t> ||
124 detail::IsFlowFunctionPtr<ret_ff_t, d_t>);
125 static_assert(detail::IsFlowFunction<ctr_ff_t, d_t> ||
126 detail::IsFlowFunctionPtr<ctr_ff_t, d_t>);
127 static_assert(detail::IsFlowFunction<summary_ff_t, d_t> ||
128 detail::IsFlowFunctionPtr<summary_ff_t, d_t> ||
129 std::is_same_v<summary_ff_t, std::nullptr_t>);
130
131 template <typename T>
132 static constexpr bool needs_cache_v =
133 std::is_same_v<std::shared_ptr<FlowFunction<d_t>>, T> ||
134 std::is_same_v<GenericFlowFunction<d_t>, T> ||
135 (detail::IsFlowFunctionPtr<T, d_t> && !std::is_pointer_v<T>);
136};

140/// This class caches flow and edge functions to avoid their reconstruction.
141/// When a flow or edge function must be applied multiple times, a cached
142/// version is used if it exists, otherwise a new one is created and inserted
143/// into the cache. This class is used within both IDE solver implementations.
144template <typename ProblemTy, bool AutoAddZero>
145class FlowFunctionCache
146 : detail::FlowFunctionCacheBase<ProblemTy>,
147 detail::IntraFlowFunctionsMixin<
148 typename detail::FlowFunctionCacheBase<ProblemTy>::normal_ff_t,
149 typename detail::FlowFunctionCacheBase<ProblemTy>::ctr_ff_t> {
150 using base_t = detail::FlowFunctionCacheBase<ProblemTy>;
151
152 using typename base_t::call_ff_t;
153 using typename base_t::ctr_ff_t;
154 using typename base_t::d_t;
155 using typename base_t::domain_t;
156 using typename base_t::f_t;
157 using typename base_t::n_t;
158 using typename base_t::normal_ff_t;
159 using typename base_t::ret_ff_t;
160 using typename base_t::summary_ff_t;
161
162 using detail::IntraFlowFunctionsMixin<normal_ff_t, ctr_ff_t>::NormalFFCache;
163 using detail::IntraFlowFunctionsMixin<normal_ff_t, ctr_ff_t>::CtrFFCache;
164
165 template <typename FFTy>
166 static constexpr bool needs_cache_v = base_t::template needs_cache_v<FFTy>;
167
168 template <typename FFTy>
169 static auto wrapFlowFunction(ProblemTy &Problem, FFTy &&FF) {
170 using ff_t = std::decay_t<FFTy>;
171
172 if constexpr (AutoAddZero) {
173 if constexpr (detail::IsFlowFunctionPtr<ff_t, d_t>) {
174 if constexpr (needs_cache_v<ff_t>) {
175 return detail::AutoAddZeroFF{
176 GenericFlowFunctionView(getPointerFrom(FF)),
177 Problem.getZeroValue()};
178 } else {
179 return detail::AutoAddZeroFF{
180 GenericFlowFunction<d_t>(std::forward<FFTy>(FF)),
181 Problem.getZeroValue()};
182 }
183 } else {
184 return detail::AutoAddZeroFF{std::forward<FFTy>(FF),
185 Problem.getZeroValue()};
186 }
187 } else {
188 if constexpr (detail::IsFlowFunctionPtr<ff_t, d_t>) {
189 if constexpr (needs_cache_v<ff_t>) {
190 return GenericFlowFunctionView(getPointerFrom(FF));
191 } else {
192 return GenericFlowFunction<d_t>(std::forward<FFTy>(FF));
193 }
194 } else {
195 return std::forward<FFTy>(FF);
196 }
197 }
198 }

200 template <typename FFTy, typename CacheTy, typename FFFactory>
201 decltype(auto) getFlowFunction(CacheTy &Cache, uint64_t Id,
202 ProblemTy &Problem, FFFactory Fact) {
203 if constexpr (needs_cache_v<FFTy>) {
204 auto &Ret = Cache.getOrCreate(Id);
205 if (!Ret) {
206 Ret = std::invoke(Fact, Problem);
207 }
208
209 return wrapFlowFunction(Problem, Ret);
210 } else {
211 return wrapFlowFunction(Problem, std::invoke(Fact, Problem));
212 }
213 }

275 return wrapFlowFunction(Problem, Ret);
276 } else {
277 return wrapFlowFunction(
