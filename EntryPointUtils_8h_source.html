25template <typename EntryRange, CFG C, typename HandlerFn>
26void forallStartingPoints(const EntryRange &EntryPoints, const C &CFG,
27 HandlerFn Handler) {
28 for (const auto &EntryPointFn : EntryPoints) {
29 if constexpr (std::is_convertible_v<std::decay_t<decltype(EntryPointFn)>,
30 bool>) {
31 if (!EntryPointFn) {
32 continue;
33 }
34 }
35 for (const auto &SP : CFG.getStartPointsOf(EntryPointFn)) {
36 std::invoke(Handler, SP);
37 }
38 }
39}

41template <typename EntryRange, CFG C, typename ICFGorIRDB, typename HandlerFn>
42void forallStartingPoints(const EntryRange &EntryPoints, const ICFGorIRDB *ICDB,
43 const C &CFG, HandlerFn Handler) {
44
45 if (llvm::hasSingleElement(EntryPoints) &&
46 *llvm::adl_begin(EntryPoints) == "__ALL__") {
47 forallStartingPoints(ICDB->getAllFunctions(), CFG, std::move(Handler));
48 } else {
49 forallStartingPoints(llvm::map_range(EntryPoints,
50 [ICDB](llvm::StringRef Name) {
51 return ICDB->getFunction(Name);
52 }),
53 CFG, std::move(Handler));
54 }
55}
