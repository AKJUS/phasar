42/// \brief Configuration options for the solving process of IFDS/IDE problems

47 /// Returns whether the solver should handle unbalanced returns (default:
48 /// false)
49 [[nodiscard]] bool followReturnsPastSeeds() const;

52 [[nodiscard]] bool autoAddZero() const;

57 /// \note This option may severly hurt the solver's performance

60 /// command-line (default: false)
