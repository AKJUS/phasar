37/// \brief Simple representation of a serializable data-flow summary
38class FunctionDataFlowFacts {
39public:

45 // insert a set of data flow facts
46 void insertSet(llvm::StringRef FuncKey, uint32_t Index,
47 std::vector<DataFlowFact> OutSet) {
48 Fdff[FuncKey].try_emplace(Index, std::move(OutSet));
49 }
50
51 // insert a single data flow fact
52 void addElement(llvm::StringRef FuncKey, uint32_t Index, DataFlowFact Out) {
53 Fdff[FuncKey][Index].emplace_back(Out);
54 }
55
56 // get outset for a function an the parameter index
57 [[nodiscard]] const std::vector<DataFlowFact> &
58 getDataFlowFacts(llvm::StringRef FuncKey, uint32_t Index) const {
59 auto It = Fdff.find(FuncKey);
60 if (It != Fdff.end()) {
61 auto Itt = It->second.find(Index);
62 return Itt->second;
63 }
64
65 return getDefaultValue<std::vector<DataFlowFact>>();
66 }
67
68 [[nodiscard]] auto begin() const noexcept { return Fdff.begin(); }
69 [[nodiscard]] auto end() const noexcept { return Fdff.end(); }
70
71 [[nodiscard]] size_t size() const noexcept { return Fdff.size(); }
72 [[nodiscard]] bool empty() const noexcept { return size() == 0; }
73
74private:
75 [[nodiscard]] const auto &
76 getDataFlowFactsOrEmpty(llvm::StringRef FuncKey) const {
77 auto It = Fdff.find(FuncKey);
78 if (It != Fdff.end()) {
79 return It->second;
80 }
81
82 return getDefaultValue<ParameterMappingTy>();
83 }

85 llvm::StringMap<ParameterMappingTy> Fdff;
