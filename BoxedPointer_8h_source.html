26/// A T** that behaves like a T*. Used as AliasSetPtr in LLVMAliasSet
27template <typename T> class BoxedPtr {
28public:
29 using value_type = T;
30
31 constexpr BoxedPtr() noexcept = default;
32 constexpr BoxedPtr(std::nullptr_t) noexcept {}
33 constexpr BoxedPtr(value_type **Value) noexcept : Value(Value) {
34 assert(Value != nullptr);
35 }
36 constexpr BoxedPtr(const BoxedPtr &) noexcept = default;
37 constexpr BoxedPtr &operator=(const BoxedPtr &) noexcept = default;
38 ~BoxedPtr() noexcept = default;
39
40 [[nodiscard]] constexpr value_type *get() noexcept { return *Value; }
41 [[nodiscard]] constexpr const value_type *get() const noexcept {
42 return *Value;
43 }
44
45 constexpr value_type *operator->() noexcept { return get(); }
46 constexpr const value_type *operator->() const noexcept { return get(); }
47
48 constexpr value_type &operator*() noexcept { return **Value; }
49 constexpr const value_type &operator*() const noexcept { return **Value; }
50
51 constexpr operator bool() const noexcept {
52 return Value && *Value != nullptr;
53 }
54
55 [[nodiscard]] constexpr value_type **value() noexcept { return Value; }
56 [[nodiscard]] constexpr value_type const *const *value() const noexcept {
57 return Value;
58 }
59
60 [[nodiscard]] friend constexpr llvm::hash_code
61 hash_value(BoxedPtr // NOLINT(readability-identifier-naming)
62 Ptr) noexcept {
63 return llvm::hash_value(Ptr.Value);
64 }
65
66 [[nodiscard]] friend constexpr bool operator==(BoxedPtr LHS,
67 BoxedPtr RHS) noexcept {
68 return LHS.Value == RHS.Value;
69 }
70 [[nodiscard]] friend constexpr bool operator!=(BoxedPtr LHS,
71 BoxedPtr RHS) noexcept {
72 return !(LHS == RHS);
73 }
74
75 [[nodiscard]] friend constexpr bool operator<(BoxedPtr LHS,
76 BoxedPtr RHS) noexcept {
77 return LHS.Value < RHS.Value;
78 }
79 [[nodiscard]] friend constexpr bool operator>(BoxedPtr LHS,
80 BoxedPtr RHS) noexcept {
81 return LHS.Value > RHS.Value;
82 }
83
84 [[nodiscard]] friend constexpr bool operator<=(BoxedPtr LHS,
85 BoxedPtr RHS) noexcept {
86 return LHS.Value <= RHS.Value;
87 }
88 [[nodiscard]] friend constexpr bool operator>=(BoxedPtr LHS,
89 BoxedPtr RHS) noexcept {
90 return LHS.Value >= RHS.Value;
91 }
92
93private:
94 friend class BoxedConstPtr<value_type>;

96 value_type **Value = nullptr;
97};

99/// A T const** that behaves like a const T*.

102template <typename T> class BoxedConstPtr : private BoxedPtr<T> {
