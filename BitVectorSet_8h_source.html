33inline bool isLess(const llvm::BitVector &Lhs, const llvm::BitVector &Rhs) {
34 unsigned LhsBits = Lhs.size();
35 unsigned RhsBits = Rhs.size();
36
37 if (LhsBits > RhsBits) {
38 if (Lhs.find_next(RhsBits) != -1) {
39 return false;
40 }
41 } else if (LhsBits < RhsBits) {
42 if (Rhs.find_first_in(LhsBits, RhsBits) != -1) {
43 return true;
44 }
45 }
46
47 // Compare every bit on both sides because Lhs and Rhs either have the same
48 // amount of bits or all other upper bits of the larger one are zero.
49 for (int I = static_cast<int>(std::min(LhsBits, RhsBits)) - 1; I >= 0; --I) {
50 if (LLVM_UNLIKELY(Lhs[I] != Rhs[I])) {
51 return Rhs[I];
52 }
53 }
54 return false;
55}

150 [[nodiscard]] BitVectorSet<T> setUnion(const BitVectorSet<T> &Other) const {
151 const bool ThisSetIsSmaller = Bits.size() < Other.Bits.size();
152 BitVectorSet<T> Res = ThisSetIsSmaller ? Other : *this;

163 Bits.size() > Other.Bits.size() ? *this : Other;

237 friend bool operator==(const BitVectorSet &Lhs, const BitVectorSet &Rhs) {
238 bool LeftEmpty = Lhs.empty();
239 bool RightEmpty = Rhs.empty();
240 if (LeftEmpty || RightEmpty) {
241 return LeftEmpty == RightEmpty;
242 }
243 // Check, whether Lhs and Rhs actually have the same bits set and not
244 // whether their internal representation is exactly identitcal
245
246 uintptr_t LStore{}, RStore{};
247 auto LhsWords = internal::getWords(Lhs.Bits, LStore);
248 auto RhsWords = internal::getWords(Rhs.Bits, RStore);
249 if (LhsWords.size() == RhsWords.size()) {
250 return LhsWords == RhsWords;
251 }
252 auto MinSize = std::min(LhsWords.size(), RhsWords.size());
253 if (LhsWords.slice(0, MinSize) != RhsWords.slice(0, MinSize)) {
254 return false;
255 }
256 auto Rest = (LhsWords.size() > RhsWords.size() ? LhsWords : RhsWords)
257 .slice(MinSize);
258 return std::all_of(Rest.begin(), Rest.end(),
259 [](auto Word) { return Word == 0; });
260 }

270 // NOLINTNEXTLINE(readability-identifier-naming) -- needed for ADL
271 friend llvm::hash_code hash_value(const BitVectorSet &BV) noexcept {
272 if (BV.Bits.empty()) {
273 return {};
274 }
275 uintptr_t Store{};
276 auto Words = internal::getWords(BV.Bits, Store);
277 size_t Idx = Words.size();
278 while (Idx && Words[Idx - 1] == 0) {
279 --Idx;
280 }
281 return llvm::hash_combine_range(Words.begin(),
282 std::next(Words.begin(), Idx));
283 }
