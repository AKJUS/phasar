30inline bool isLess(const llvm::BitVector &Lhs, const llvm::BitVector &Rhs) {
31 unsigned LhsBits = Lhs.size();
32 unsigned RhsBits = Rhs.size();
33
34 if (LhsBits > RhsBits) {
35 if (Lhs.find_next(RhsBits) != -1) {
36 return false;
37 }
38 } else if (LhsBits < RhsBits) {
39 if (Rhs.find_first_in(LhsBits, RhsBits) != -1) {
40 return true;
41 }
42 }
43
44 // Compare every bit on both sides because Lhs and Rhs either have the same
45 // amount of bits or all other upper bits of the larger one are zero.
46 for (int I = static_cast<int>(std::min(LhsBits, RhsBits)) - 1; I >= 0; --I) {
47 if (LLVM_UNLIKELY(Lhs[I] != Rhs[I])) {
48 return Rhs[I];
49 }
50 }
51 return false;
52}

227 void insert(const T &Data) {
228 auto Search = Position.left.find(Data);
229 // Data already known
230 if (Search != Position.left.end()) {
231 if (Bits.size() <= Search->second) {
232 Bits.resize(Search->second + 1);
233 }
234 Bits.set(Search->second);

236 // Data unknown
237 size_t Idx = Position.left.size();
238 Position.left.insert(std::make_pair(Data, Position.left.size()));
239 if (Bits.size() <= Position.left.size()) {
240 Bits.resize(Position.left.size());
241 }
242 Bits.set(Idx);
243 }

299 friend bool operator==(const BitVectorSet &Lhs, const BitVectorSet &Rhs) {
300 bool LeftEmpty = Lhs.empty();
301 bool RightEmpty = Rhs.empty();
302 if (LeftEmpty || RightEmpty) {
303 return LeftEmpty == RightEmpty;
304 }
305 // Check, whether Lhs and Rhs actually have the same bits set and not
306 // whether their internal representation is exactly identitcal
307
308 uintptr_t LStore{}, RStore{};
309 auto LhsWords = internal::getWords(Lhs.Bits, LStore);
310 auto RhsWords = internal::getWords(Rhs.Bits, RStore);
311 if (LhsWords.size() == RhsWords.size()) {
312 return LhsWords == RhsWords;
313 }
314 auto MinSize = std::min(LhsWords.size(), RhsWords.size());
315 if (LhsWords.slice(0, MinSize) != RhsWords.slice(0, MinSize)) {
316 return false;
317 }
318 auto Rest = (LhsWords.size() > RhsWords.size() ? LhsWords : RhsWords)
319 .slice(MinSize);
320 return std::all_of(Rest.begin(), Rest.end(),
321 [](auto Word) { return Word == 0; });
322 }

332 // NOLINTNEXTLINE(readability-identifier-naming) -- needed for ADL
333 friend llvm::hash_code hash_value(const BitVectorSet &BV) noexcept {
334 if (BV.Bits.empty()) {
335 return {};
336 }
337 uintptr_t Store{};
338 auto Words = internal::getWords(BV.Bits, Store);
339 size_t Idx = Words.size();
340 while (Idx && Words[Idx - 1] == 0) {
341 --Idx;
342 }
343 return llvm::hash_combine_range(Words.begin(),
344 std::next(Words.begin(), Idx));
345 }
