35/// An explicit graph-representation of a call-graph. Only represents the data,
36/// not the call-graph analysis that creates it.
37///
38/// This type is immutable. To incrementally build it from your call-graph
39/// analysis, use the CallGraphBuilder
40template <typename N, typename F>
41class CallGraph : public CallGraphBase<CallGraph<N, F>> {
42 using base_t = CallGraphBase<CallGraph<N, F>>;
43 friend base_t;
44 friend class CallGraphBuilder<N, F>;
45
46public:
47 using typename base_t::f_t;
48 using typename base_t::n_t;
49 using FunctionVertexTy = llvm::SmallVector<n_t>;
50 using InstructionVertexTy = llvm::SmallVector<f_t>;

62 /// A range of all functions that are vertices in the call-graph. The number
63 /// of vertex functions can be retrieved by getNumVertexFunctions().
64 [[nodiscard]] auto getAllVertexFunctions() const noexcept {
65 return llvm::make_first_range(CallersOf);
66 }

68 /// A range of all call-sites that are vertices in the call-graph. The number
69 /// of vertex-callsites can be retrived by getNumVertexCallSites().
70 [[nodiscard]] auto getAllVertexCallSites() const noexcept {
71 return llvm::make_first_range(CalleesAt);

74 [[nodiscard]] size_t getNumVertexFunctions() const noexcept {
75 return CallersOf.size();

77 [[nodiscard]] size_t getNumVertexCallSites() const noexcept {
78 return CalleesAt.size();

81 /// The number of functions within this call-graph
82 [[nodiscard]] size_t size() const noexcept { return getNumVertexFunctions(); }
83
84 [[nodiscard]] bool empty() const noexcept { return CallersOf.empty(); }

86 template <typename FunctionIdGetter, typename InstIdGetter>
87 void printAsJson(llvm::raw_ostream &OS, FunctionIdGetter GetFunctionId,
88 InstIdGetter GetInstructionId) const {
89 CallGraphData CGData;
90 CGData.FToFunctionVertexTy.reserve(CallersOf.size());
91
92 for (const auto &[Fun, Callers] : CallersOf) {
93 auto &JCallers =
94 CGData.FToFunctionVertexTy[std::invoke(GetFunctionId, Fun)];
95
96 CGData.FToFunctionVertexTy.reserve(Callers->size());
97 for (const auto &CS : *Callers) {
98 JCallers.push_back(std::invoke(GetInstructionId, CS));
99 }
100 }
101
102 CGData.printAsJson(OS);
103 }

105 template <typename FunctionLabelGetter, typename InstParentGetter,
106 typename InstLabelGetter>
107 void printAsDot(llvm::raw_ostream &OS, FunctionLabelGetter GetFunctionLabel,
108 InstParentGetter GetFunctionFromInst,
109 InstLabelGetter GetInstLabel) const {
110 OS << "digraph CallGraph{\n";
111 scope_exit CloseBrace = [&OS] { OS << "}\n"; };
112
113 llvm::DenseMap<f_t, size_t> Fun2Id;
114 Fun2Id.reserve(CallersOf.size());
115
116 size_t CurrId = 0;
117 for (const auto &Fun : getAllVertexFunctions()) {
118 OS << CurrId << "[label=\"";
119 OS.write_escaped(std::invoke(GetFunctionLabel, Fun)) << "\"];\n";
120 Fun2Id[Fun] = CurrId++;
121 }
122
123 for (const auto &[CS, Callees] : CalleesAt) {
124 const auto &Fun = std::invoke(GetFunctionFromInst, CS);
125
126 for (const auto &Succ : *Callees) {
127 OS << Fun2Id.lookup(Fun) << "->" << Fun2Id.lookup(Succ) << "[label=\"";
128 OS.write_escaped(std::invoke(GetInstLabel, CS)) << "\"];\n";
129 }
130 }
131 }

133private:
134 [[nodiscard]] llvm::ArrayRef<f_t>
135 getCalleesOfCallAtImpl(ByConstRef<n_t> Inst) const noexcept {
136 if (const auto *CalleesPtr = CalleesAt.lookup(Inst)) {
137 return *CalleesPtr;
138 }
139 return {};
140 }
141
142 [[nodiscard]] llvm::ArrayRef<n_t>
143 getCallersOfImpl(ByConstRef<f_t> Fun) const noexcept {
144 if (const auto *CallersPtr = CallersOf.lookup(Fun)) {
145 return *CallersPtr;

147 return {};
148 }

159/// A mutable wrapper over a CallGraph. Use this to build a call-graph from
160/// within your call-graph ananlysis.
161template <typename N, typename F> class CallGraphBuilder {
162public:
163 using n_t = typename CallGraph<N, F>::n_t;
164 using f_t = typename CallGraph<N, F>::f_t;
165 using FunctionVertexTy = typename CallGraph<n_t, f_t>::FunctionVertexTy;
166 using InstructionVertexTy = typename CallGraph<n_t, f_t>::InstructionVertexTy;
167
168 void reserve(size_t MaxNumFunctions) {
169 CG.FunVertexOwner.reserve(MaxNumFunctions);
170 CG.CalleesAt.reserve(MaxNumFunctions);
171 CG.CallersOf.reserve(MaxNumFunctions);
172 }
173
174 /// Registeres a new function in the call-graph. Returns a list of all
175 /// call-sites that are known so far to potentially call this function.
176 /// Do not manually add elements to this vector -- use addCallEdge instead.
177 [[nodiscard]] FunctionVertexTy *addFunctionVertex(f_t Fun) {
178 auto [It, Inserted] = CG.CallersOf.try_emplace(std::move(Fun), nullptr);
179 if (Inserted) {
180 assert(CG.FunVertexOwner.size() < CG.FunVertexOwner.capacity() &&
181 "Trying to add more than MaxNumFunctions Function Vertices");
182 It->second = &CG.FunVertexOwner.emplace_back();
183 }
184 return It->second;
185 }
186
187 /// Registeres a new call-site in the call-graph. Returns a list of all
188 /// callee functions that are known so far to potentially be called by this
189 /// function.
190 /// Do not manually add elements to this vector -- use addCallEdge instead.
191 [[nodiscard]] InstructionVertexTy *addInstructionVertex(n_t Inst) {
192 auto [It, Inserted] = CG.CalleesAt.try_emplace(std::move(Inst), nullptr);
193 if (Inserted) {
194 It->second = &CG.InstVertexOwner.emplace_back();
195 }
196 return It->second;
197 }
198
199 /// Tries to lookup the InstructionVertex for the given call-site. Returns
200 /// nullptr on failure.
201 [[nodiscard]] InstructionVertexTy *
202 getInstVertexOrNull(ByConstRef<n_t> Inst) const noexcept {
203 return CG.CalleesAt.lookup(Inst);
204 }
205
206 /// Adds a new directional edge to the call-graph indicating that CS may call
207 /// Callee
208 void addCallEdge(n_t CS, f_t Callee) {
209 auto IVtx = addInstructionVertex(CS);
210 auto FVtx = addFunctionVertex(Callee);
211 addCallEdge(std::move(CS), IVtx, std::move(Callee), FVtx);
212 }
213
214 /// Same as addCallEdge(n_t, f_t), but uses an already known
215 /// InstructionVertexTy to save a lookup
216 void addCallEdge(n_t CS, InstructionVertexTy *Callees, f_t Callee) {
217 auto *Callers = addFunctionVertex(Callee);
218 addCallEdge(std::move(CS), Callees, std::move(Callee), Callers);
219 }
220
221 /// Same as addCallEdge(n_t, f_t), but uses an already known
222 /// FunctionVertexTy to save a lookup
223 void addCallEdge(n_t CS, f_t Callee, FunctionVertexTy *Callers) {
224 auto *Callees = addInstructionVertex(CS);
225 addCallEdge(std::move(CS), Callees, std::move(Callee), Callers);
226 }
227
228 /// Moves the completely built call-graph out of this builder for further
229 /// use. Do not use the builder after it anymore.
230 [[nodiscard]] CallGraph<n_t, f_t> consumeCallGraph() noexcept {
231 return std::move(CG);
232 }
233
234 /// Returns a view on the current (partial) call-graph that has already been
235 /// constructed
236 [[nodiscard]] const CallGraph<n_t, f_t> &viewCallGraph() const noexcept {
237 return CG;
238 }
239
240private:
241 void addCallEdge(n_t CS, InstructionVertexTy *Callees, f_t Callee,
242 FunctionVertexTy *Callers) {
243 Callees->push_back(std::move(Callee));
244 Callers->push_back(std::move(CS));
245 }
246
247 CallGraph<n_t, f_t> CG{};
248};

250template <typename N, typename F>
251template <typename FunctionGetter, typename InstructionGetter>
252[[nodiscard]] CallGraph<N, F>
253CallGraph<N, F>::deserialize(const CallGraphData &PrecomputedCG,
254 FunctionGetter GetFunctionFromName,
255 InstructionGetter GetInstructionFromId) {
256 CallGraphBuilder<N, F> CGBuilder;
257 CGBuilder.reserve(PrecomputedCG.FToFunctionVertexTy.size());
258
259 for (const auto &[FunName, CallerIDs] : PrecomputedCG.FToFunctionVertexTy) {
260 const auto &Fun = std::invoke(GetFunctionFromName, FunName);
261 if (!Fun) {
262 PHASAR_LOG_LEVEL_CAT(WARNING, "CallGraph",
263 "Invalid function name: " << FunName);
264 continue;
265 }
266
267 auto *CEdges = CGBuilder.addFunctionVertex(Fun);
268 CEdges->reserve(CallerIDs.size());
269
270 for (const auto &JId : CallerIDs) {
271 const auto &CS = std::invoke(GetInstructionFromId, JId);
272 if (!CS) {
273 PHASAR_LOG_LEVEL_CAT(WARNING, "CallGraph",
274 "Invalid Call-Instruction Id: " << JId);
275 }
276
277 CGBuilder.addCallEdge(CS, Fun);
278 }
279 }
280 return CGBuilder.consumeCallGraph();
281}
