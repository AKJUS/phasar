77 /// Parses the given LLVM IR file and owns the resulting IR Module.

175 [[nodiscard]] n_t getInstructionImpl(size_t Id) const noexcept {
176 // Effectively make use of integer overflow here...
177 if (Id - IdOffset < IdToInst.size() - IdOffset) {
178 return llvm::cast<llvm::Instruction>(IdToInst[Id]);
179 }
180 return n_t{};
181 }
182
183 [[nodiscard]] auto getAllInstructionsImpl() const noexcept {
184 return llvm::map_range(
185 llvm::ArrayRef(IdToInst).drop_front(IdOffset),
186 [](const llvm::Value *V) { return llvm::cast<llvm::Instruction>(V); });
187 }
188
189 [[nodiscard]] size_t getInstructionIdImpl(n_t Inst) const {
190 auto It = InstToId.find(Inst);
191 assert(It != InstToId.end());
192 return It->second;
193 }
194 [[nodiscard]] bool isValidImpl() const noexcept;
195
196 void dumpImpl() const;
197
198 void initInstructionIds();
199 /// XXX Later we might get rid of the metadata IDs entirely and therefore of

203 // LLVMContext is not movable, so wrap it into a unique_ptr
204 std::unique_ptr<llvm::LLVMContext> Ctx;
205 MaybeUniquePtr<llvm::Module> Mod = nullptr;
206 size_t IdOffset = 0;
207 llvm::SmallVector<const llvm::Value *, 0> IdToInst;
208 llvm::DenseMap<const llvm::Value *, size_t> InstToId;
