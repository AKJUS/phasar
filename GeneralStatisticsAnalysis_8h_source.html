83/**
84 * This class uses the Module Pass Mechanism of LLVM to compute
85 * some statistics about a Module. This includes the number of
86 * - Function calls
87 * - Global variables
88 * - Basic blocks
89 * - Allocation sites
90 * - Call sites
91 * - Instructions
92 * - Pointers
93 *
94 * and also a set of all allocated Types in that Module.
95 *
96 * This pass does not modify the analyzed Module in any way!
97 *
98 * @brief Computes general statistics for a Module.
99 */
100class GeneralStatisticsAnalysis
101 : public llvm::AnalysisInfoMixin<GeneralStatisticsAnalysis> {
102private:
103 friend llvm::AnalysisInfoMixin<GeneralStatisticsAnalysis>;
104 static llvm::AnalysisKey Key;
105 GeneralStatistics Stats;
106
107public:
108 /// The pass itself stores the results.
109 using Result = GeneralStatistics;
110
111 explicit GeneralStatisticsAnalysis() = default;
112
113 GeneralStatistics runOnModule(llvm::Module &M);
114
115 inline GeneralStatistics run(llvm::Module &M,
116 llvm::ModuleAnalysisManager & /*AM*/) {
117 return runOnModule(M);
118 }
119};
