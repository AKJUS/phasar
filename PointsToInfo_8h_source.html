33/// A type-erased reference to any object implementing the PointsToInfoBase
34/// interface. Use this, if your analysis is not tied to a specific points-to
35/// info implementation.
36///
37/// This is a *non-owning* reference similar to std::string_view and
38/// llvm::ArrayRef. Pass values of this type by value.
39///
40template <is_PointsToTraits_v PTATraits>

103 template <typename ConcretePTA>
104 constexpr static VTable<> makeVTableFor() noexcept {
105 constexpr VTableBase Base = {
106 {
107 [](const void *PT, ByConstRef<v_t> Pointer) noexcept {
108 return static_cast<const ConcretePTA *>(PT)->asAbstractObject(
109 Pointer);
110 },
111 [](const void *PT, ByConstRef<o_t> Pointer, ByConstRef<n_t> At,
112 llvm::function_ref<void(o_t)> WithPointee) {
113 const auto *CPT = static_cast<const ConcretePTA *>(PT);
114 if constexpr (IsPointsToIterator<ConcretePTA>) {
115 return (void)CPT->forallPointeesOf(Pointer, At, WithPointee);
116 } else {
117 auto PointsToSet = CPT->getPointsToSet(Pointer, At);
118 // The PointsToSet can be a set or a pointer to a set
119 auto PointsToSetPtr = getPointerFrom(PointsToSet);
120 for (auto &&Pointee : *PointsToSetPtr) {
121 WithPointee(PSR_FWD(Pointee));
122 }
123 }
124 },
125 [](const void *PT, ByConstRef<o_t> Pointer, ByConstRef<o_t> Obj,
126 ByConstRef<n_t> AtInstruction) {
127 return static_cast<const ConcretePTA *>(PT)->mayPointsTo(
128 Pointer, Obj, AtInstruction);
129 },
130 [](const void *PT) noexcept {
131 delete static_cast<const ConcretePTA *>(PT);
132 },
133 },
134 [](const void *PT, ByConstRef<o_t> Obj) noexcept {
135 return static_cast<const ConcretePTA *>(PT)->asPointerOrNull(Obj);
136 },
137 [](const void *PT, ByConstRef<o_t> Pointer,
138 ByConstRef<n_t> AtInstruction) {
139 return static_cast<const ConcretePTA *>(PT)->getPointsToSet(
140 Pointer, AtInstruction);
141 },
142 };
143 if constexpr (std::is_same_v<o_t, v_t>) {
144 return {Base};
145 } else {
146 return {
147 Base,
148 [](const void *PT, ByConstRef<v_t> Pointer, ByConstRef<o_t> Obj,
149 ByConstRef<n_t> AtInstruction) {
150 return static_cast<const ConcretePTA *>(PT)->mayPointsTo(
151 Pointer, Obj, AtInstruction);
152 },
153 [](const void *PT, ByConstRef<v_t> Pointer,
154 ByConstRef<n_t> AtInstruction) {
155 return static_cast<const ConcretePTA *>(PT)->getPointsToSet(
156 Pointer, AtInstruction);
157 },
158 };
159 }
160 }

216/// Similar to PointsToInfoRef, but owns the held reference. Use this, if you
217/// need to decide dynamically, which points-to info implementation to use.
218///
219/// Implicitly convertible to PointsToInfoRef.
220///
221template <is_PointsToTraits_v PTATraits>
222class [[clang::trivial_abi]] PointsToInfo<PTATraits> final
223 : public PointsToInfoRef<PTATraits> {
224 using base_t = PointsToInfoRef<PTATraits>;
225
226public:
227 using typename base_t::n_t;
228 using typename base_t::o_t;
229 using typename base_t::PointsToSetPtrTy;
230 using typename base_t::PointsToSetTy;
231 using typename base_t::v_t;
232
233 constexpr PointsToInfo() noexcept = default;
234 constexpr PointsToInfo(std::nullptr_t) noexcept {};
235
236 PointsToInfo(const PointsToInfo &) = delete;
237 PointsToInfo &operator=(const PointsToInfo &) = delete;
238
239 constexpr PointsToInfo(PointsToInfo &&Other) noexcept { swap(Other); }
240 constexpr PointsToInfo &operator=(PointsToInfo &&Other) noexcept {
241 PointsToInfo(std::move(Other)).swap(*this);
242 return *this;
243 }
244
245 constexpr void swap(PointsToInfo &Other) noexcept {
246 std::swap(this->PT, Other.PT);
247 std::swap(this->VT, Other.VT);
248 }
249 constexpr friend void swap(PointsToInfo &LHS, PointsToInfo &RHS) noexcept {
250 LHS.swap(RHS);
251 }
252
253 template <typename ConcretePTA, typename... ArgTys>
254 constexpr explicit PointsToInfo(std::in_place_type_t<ConcretePTA> /*unused*/,
255 ArgTys &&...Args)
256 : PointsToInfoRef<PTATraits>(
257 new ConcretePTA(std::forward<ArgTys>(Args)...)) {}
258
259 template <typename ConcretePTA>
260 constexpr PointsToInfo(std::unique_ptr<ConcretePTA> PTA)
261 : PointsToInfoRef<PTATraits>(PTA.release()) {}
262
263 ~PointsToInfo() noexcept {
264 if (*this) {
265 this->VT->Destroy(this->PT);
266 this->VT = nullptr;
267 this->PT = nullptr;
268 }
269 }
