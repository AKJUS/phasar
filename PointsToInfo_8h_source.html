32/// A type-erased reference to any object implementing the PointsToInfoBase
33/// interface. Use this, if your analysis is not tied to a specific points-to
34/// info implementation.
35///
36/// This is a *non-owning* reference similar to std::string_view and
37/// llvm::ArrayRef. Pass values of this type by value.
38///
39template <typename PTATraits>
40class PointsToInfoRef<PTATraits,
41 std::enable_if_t<is_PointsToTraits_v<PTATraits>>>
42 : public PointsToInfoBase<PointsToInfoRef<PTATraits>> {
43 friend class PointsToInfo<PTATraits>;
44 friend PointsToInfoBase<PointsToInfoRef<PTATraits>>;

46 using base_t = PointsToInfoBase<PointsToInfoRef<PTATraits>>;

48public:

50 using typename base_t::o_t;

52 using typename base_t::PointsToSetTy;
53 using typename base_t::v_t;

55 PointsToInfoRef() noexcept = default;
56 PointsToInfoRef(std::nullptr_t) noexcept : PointsToInfoRef() {}

102 template <typename ConcretePTA>
103 constexpr static VTable<> makeVTableFor() noexcept {
104 constexpr VTableBase Base = {
105 [](const void *PT, ByConstRef<v_t> Pointer) noexcept {
106 return static_cast<const ConcretePTA *>(PT)->asAbstractObject(
107 Pointer);
108 },
109 [](const void *PT, ByConstRef<o_t> Obj) noexcept {
110 return static_cast<const ConcretePTA *>(PT)->asPointerOrNull(Obj);
111 },
112 [](const void *PT, ByConstRef<o_t> Pointer, ByConstRef<o_t> Obj,
113 ByConstRef<n_t> AtInstruction) {
114 return static_cast<const ConcretePTA *>(PT)->mayPointsTo(
115 Pointer, Obj, AtInstruction);
116 },
117 [](const void *PT, ByConstRef<o_t> Pointer,
118 ByConstRef<n_t> AtInstruction) {
119 return static_cast<const ConcretePTA *>(PT)->getPointsToSet(
120 Pointer, AtInstruction);
121 },
122 [](const void *PT) noexcept {
123 delete static_cast<const ConcretePTA *>(PT);
124 },
125 };
126 if constexpr (std::is_same_v<o_t, v_t>) {
127 return {Base};
128 } else {
129 return {
130 Base,
131 [](const void *PT, ByConstRef<v_t> Pointer, ByConstRef<o_t> Obj,
132 ByConstRef<n_t> AtInstruction) {
133 return static_cast<const ConcretePTA *>(PT)->mayPointsTo(
134 Pointer, Obj, AtInstruction);
135 },
136 [](const void *PT, ByConstRef<v_t> Pointer,
137 ByConstRef<n_t> AtInstruction) {
138 return static_cast<const ConcretePTA *>(PT)->getPointsToSet(
139 Pointer, AtInstruction);
140 },
141 };
142 }
143 }

199/// Similar to PointsToInfoRef, but owns the held reference. Use this, if you
200/// need to decide dynamically, which points-to info implementation to use.
201///
202/// Implicitly convertible to PointsToInfoRef.
203///
204template <typename PTATraits>
205class [[clang::trivial_abi]] PointsToInfo<
206 PTATraits, std::enable_if_t<is_PointsToTraits_v<PTATraits>>>
207 final : public PointsToInfoRef<PTATraits> {
208 using base_t = PointsToInfoRef<PTATraits>;
209
210public:
211 using typename base_t::n_t;
212 using typename base_t::o_t;
213 using typename base_t::PointsToSetPtrTy;
214 using typename base_t::PointsToSetTy;
215 using typename base_t::v_t;
216
217 PointsToInfo() noexcept = default;
218 PointsToInfo(std::nullptr_t) noexcept {};
219
220 PointsToInfo(const PointsToInfo &) = delete;
221 PointsToInfo &operator=(const PointsToInfo &) = delete;
222
223 PointsToInfo(PointsToInfo &&Other) noexcept { swap(Other); }
224 PointsToInfo &operator=(PointsToInfo &&Other) noexcept {
225 PointsToInfo(std::move(Other)).swap(*this);
226 return *this;
227 }
228
229 void swap(PointsToInfo &Other) noexcept {
230 std::swap(this->PT, Other.PT);
231 std::swap(this->VT, Other.VT);
232 }
233 friend void swap(PointsToInfo &LHS, PointsToInfo &RHS) noexcept {
234 LHS.swap(RHS);
235 }
236
237 template <typename ConcretePTA, typename... ArgTys>
238 explicit PointsToInfo(std::in_place_type_t<ConcretePTA> /*unused*/,
239 ArgTys &&...Args)
240 : PointsToInfoRef<PTATraits>(
241 new ConcretePTA(std::forward<ArgTys>(Args)...)) {}
242
243 template <typename ConcretePTA>
244 PointsToInfo(std::unique_ptr<ConcretePTA> PTA)
245 : PointsToInfoRef<PTATraits>(PTA.release()) {}
246
247 ~PointsToInfo() noexcept {
248 if (*this) {
249 this->VT->Destroy(this->PT);
250 this->VT = nullptr;
251 this->PT = nullptr;
252 }
253 }
254};
