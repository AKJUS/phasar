126 template <typename CallBack, typename = std::enable_if_t<std::is_invocable_v<
127 CallBack, const llvm::Value *>>>
128 void forEachAliasOf(AliasInfoRef<v_t, n_t>::AliasSetPtrTy PTS,
129 const llvm::Value *Of, CallBack &&CB) {
130 if (!HasPreciseAliasInfo) {
131 auto OfFF = makeFlowFact(Of);
132 for (const auto *Alias : *PTS) {
133 if (const auto *AliasGlob = llvm::dyn_cast<llvm::GlobalVariable>(Alias);
134 AliasGlob && AliasGlob->isConstant()) {
135 // Assume, data can never flow into the constant data section
136 // Note: If a global constant is marked as source, it keeps being
137 // propagated. We never assume, that the Of value is part of its
138 // alias-set
139 continue;
140 }
141
142 auto AliasFF = makeFlowFact(Alias);
143
144 if (AliasFF->base() == OfFF->base() && AliasFF != OfFF) {
145 continue;
146 }
147
148 std::invoke(CB, Alias);
149 }
150 } else {
151 for (const auto *Alias : *PTS) {
152 std::invoke(CB, Alias);
153 }
154 }
