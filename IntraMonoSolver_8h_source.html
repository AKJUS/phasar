32/// \brief A solver class for intraprocedual monotone problems. To solve the
33/// problem, call solve().
34///
35/// \tparam AnalysisDomainTy type of the analysis domain.
36template <typename AnalysisDomainTy> class IntraMonoSolver {
37public:
38 using ProblemTy = IntraMonoProblem<AnalysisDomainTy>;
39 using n_t = typename AnalysisDomainTy::n_t;
40 using d_t = typename AnalysisDomainTy::d_t;
41 using f_t = typename AnalysisDomainTy::f_t;
42 using t_t = typename AnalysisDomainTy::t_t;
43 using v_t = typename AnalysisDomainTy::v_t;
44 using i_t = typename AnalysisDomainTy::i_t;
45 using c_t = typename AnalysisDomainTy::c_t;
46 using mono_container_t = typename AnalysisDomainTy::mono_container_t;
47
48protected:
49 ProblemTy &IMProblem;
50 std::deque<std::pair<n_t, n_t>> Worklist;
51 std::unordered_map<n_t, mono_container_t> Analysis;
52 const c_t *CFG;
53
54 void initialize() {
55 const auto &EntryPoints = IMProblem.getEntryPoints();
56 for (const auto &EntryPoint : EntryPoints) {
57 auto Function =
58 IMProblem.getProjectIRDB()->getFunctionDefinition(EntryPoint);
59 auto ControlFlowEdges = CFG->getAllControlFlowEdges(Function);
60 // add all intra-procedural edges to the worklist
61 Worklist.insert(Worklist.begin(), ControlFlowEdges.begin(),
62 ControlFlowEdges.end());
63 // set all analysis information to the empty set
64 for (auto Insts : CFG->getAllInstructionsOf(Function)) {
65 Analysis.insert(std::make_pair(Insts, IMProblem.allTop()));
66 }
67 }
68 // insert initial seeds
69 for (auto &[Node, FlowFacts] : IMProblem.initialSeeds()) {
70 Analysis[Node].insert(FlowFacts.begin(), FlowFacts.end());
71 }
72 }
73
74public:
75 IntraMonoSolver(ProblemTy &IMP) : IMProblem(IMP), CFG(IMP.getCFG()) {}
76
77 virtual ~IntraMonoSolver() = default;
78
79 virtual void solve() {
80 // step 1: Initalization (of Worklist and Analysis)
81 initialize();
82 // step 2: Iteration (updating Worklist and Analysis)
83 while (!Worklist.empty()) {
84 // llvm::outs() << "worklist size: " << Worklist.size() << "\n";
85 std::pair<n_t, n_t> Edge = Worklist.front();
86 Worklist.pop_front();
87 n_t Src = Edge.first;
88 n_t Dst = Edge.second;
89 auto Out = IMProblem.normalFlow(Src, Analysis[Src]);
90 // need to merge if Dst is a branch target
91 if (CFG->isBranchTarget(Src, Dst)) {
92 for (auto Pred : CFG->getPredsOf(Dst)) {
93 if (Pred != Src) {
94 // we need to compute the out set of Pred and merge it with the out
95 // set of Src on-the-fly as we do not have a dedicated storage for
96 // merge points (otherwise we run into trouble with merge operator
97 // such as set union)
98 auto OtherPredOut = IMProblem.normalFlow(Pred, Analysis[Pred]);
99 Out = IMProblem.merge(Out, OtherPredOut);
100 }
101 }
102 }
103 if (!IMProblem.equal_to(Out, Analysis[Dst])) {
104 Analysis[Dst] = Out;
105 for (auto Nprimeprime : CFG->getSuccsOf(Dst)) {
106 Worklist.push_back({Dst, Nprimeprime});
107 }
108 }
109 }
110 // step 3: Presenting the result (MFP_in and MFP_out)
111 // MFP_in[s] = Analysis[s];
112 // MFP out[s] = IMProblem.flow(Analysis[s]);
113 for (auto &[Node, FlowFacts] : Analysis) {
114 FlowFacts = IMProblem.normalFlow(Node, FlowFacts);
115 }
116 }
117
118 [[nodiscard]] const mono_container_t &getResultsAt(n_t Stmt) const {
119 auto It = Analysis.find(Stmt);
120 if (It != Analysis.end()) {
121 return It->second;
122 }
123 return getDefaultValue<mono_container_t>();
124 }
125
126 virtual void dumpResults(llvm::raw_ostream &OS = llvm::outs()) {
127 OS << "\n***************************************************************\n"
128 << "* Raw IntraMonoSolver results *\n"
129 << "***************************************************************\n";
130
131 if (Analysis.empty()) {
132 OS << "No results computed!" << '\n';
133 return;
134 }
135
136 std::vector<std::pair<n_t, mono_container_t>> Cells;
137 Cells.reserve(Analysis.size());
138 Cells.insert(Cells.end(), Analysis.begin(), Analysis.end());
139
140 std::sort(Cells.begin(), Cells.end(), [](const auto &Lhs, const auto &Rhs) {
141 if constexpr (std::is_same_v<n_t, const llvm::Instruction *>) {
142 return StringIDLess{}(getMetaDataID(Lhs.first),
143 getMetaDataID(Rhs.first));
144 } else {
145 // If non-LLVM IR is used
146 return Lhs.first < Rhs.first;
147 }
148 });
149
150 for (const auto &[Node, FlowFacts] : Cells) {
151 OS << "Instruction: " << NToString(Node);
152 OS << "\nFacts: ";
153 if (FlowFacts.empty()) {
154 OS << "EMPTY\n";
155 } else {
156 IMProblem.printContainer(OS, FlowFacts);
157 }
158 OS << "\n";
159 }
160 }
161
162 virtual void emitTextReport(llvm::raw_ostream &OS = llvm::outs()) {}
163
164 virtual void emitGraphicalReport(llvm::raw_ostream &OS = llvm::outs()) {}
165};
