25/// \brief Represent the starting points of the analysis.
26///
27/// The initial facts that should hold at the entry points.
28template <typename N, typename D, typename L> class InitialSeeds {
29public:
30 using GeneralizedSeeds = std::map<N, std::map<D, L>>;
31
32 using n_t = N;
33 using d_t = D;
34 using l_t = L;
35
36 InitialSeeds() = default;
37
38 template <typename LL = L,
39 typename = std::enable_if_t<std::is_same_v<LL, BinaryDomain>>>
40 InitialSeeds(const std::map<N, std::set<D>> &Seeds) {
41 for (const auto &[Node, Facts] : Seeds) {
42 for (const auto &Fact : Facts) {
43 this->Seeds[Node][Fact] = BinaryDomain::BOTTOM;
44 }
45 }
46 }
47
48 InitialSeeds(GeneralizedSeeds Seeds) : Seeds(std::move(Seeds)) {}
49
50 template <typename LL = L,
51 typename = std::enable_if_t<std::is_same_v<LL, BinaryDomain>>>
52 void addSeed(N Node, D Fact) {
53 addSeed(Node, Fact, BinaryDomain::BOTTOM);
54 }
55
56 void addSeed(N Node, D Fact, L Value) {
57 Seeds[std::move(Node)].insert_or_assign(std::move(Fact), std::move(Value));
58 }
59
60 [[nodiscard]] size_t countInitialSeeds() const {
61 size_t NumSeeds = 0;
62 for (const auto &[Node, Facts] : Seeds) {
63 NumSeeds += Facts.size();
64 }
65 return NumSeeds;
66 }
67
68 [[nodiscard]] size_t countInitialSeeds(N Node) const {
69 auto Search = Seeds.find(Node);
70 if (Search != Seeds.end()) {
71 return Search->second.size();
72 }
73 return 0;
74 }
75
76 [[nodiscard]] bool containsInitialSeedsFor(N Node) const {
77 return Seeds.count(Node);
78 }
79
80 [[nodiscard]] bool empty() const { return Seeds.empty(); }
81
82 [[nodiscard]] const GeneralizedSeeds &getSeeds() const & { return Seeds; }
83 [[nodiscard]] GeneralizedSeeds getSeeds() && { return std::move(Seeds); }
84
85 void dump(llvm::raw_ostream &OS = llvm::errs()) const {
86 OS << "======================== Initial Seeds ========================\n";
87 for (const auto &[Node, Facts] : Seeds) {
88 OS << "At " << NToString(Node) << '\n';
89 for (const auto &[Fact, Value] : Facts) {
90 OS << "> " << DToString(Fact) << " --> \\." << LToString(Value) << '\n';
91 }
92 OS << '\n';
93 }
94 OS << "========================== End Seeds ==========================\n";
95 }
96
97private:
98 GeneralizedSeeds Seeds;
99};
