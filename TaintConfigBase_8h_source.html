77 /// \brief Calls Handler for all operands of Inst (maybe including Inst
78 /// itself) that are generated unconditionally as tainted.
79 ///
80 /// If Inst is a function-call, the Callee function should be specified
81 /// explicitly.
82 template <typename HandlerFn>
83 void forAllGeneratedValuesAt(n_t Inst, Nullable<f_t> Callee,
84 HandlerFn &&Handler) const {
85 self().forAllGeneratedValuesAtImpl(std::move(Inst), std::move(Callee),
86 std::forward<HandlerFn>(Handler));
87 }
88
89 /// \brief Calls Handler for all operands of Inst that may generate a leak
90 /// when they are tainted.
91 ///
92 /// If Inst is a function-call, the Callee function should be specified
93 /// explicitly.
94 template <typename HandlerFn>
95 void forAllLeakCandidatesAt(n_t Inst, Nullable<f_t> Callee,
96 HandlerFn &&Handler) const {
97 self().forAllLeakCandidatesAtImpl(std::move(Inst), std::move(Callee),
98 std::forward<HandlerFn>(Handler));
99 }
100
101 /// \brief Calls Handler for all operands of Inst that become sanitized after
102 /// the instruction is completed.
103 ///
104 /// If Inst is a function-call, the Callee function should be specified
105 /// explicitly.
106 template <typename HandlerFn>
107 void forAllSanitizedValuesAt(n_t Inst, Nullable<f_t> Callee,
108 HandlerFn &&Handler) const {
109 self().forAllSanitizedValuesAtImpl(std::move(Inst), std::move(Callee),
110 std::forward<HandlerFn>(Handler));
111 }
112
113 [[nodiscard]] bool generatesValuesAt(n_t Inst, Nullable<f_t> Callee) const {
114 return self().generatesValuesAtImpl(std::move(Inst), std::move(Callee));
115 }

118 }
119 [[nodiscard]] bool sanitizesValuesAt(n_t Inst, Nullable<f_t> Callee) const {

121 }

125 }

150protected:

154 TaintDescriptionCallBackTy SinkCallBack{};
155 TaintDescriptionCallBackTy SanitizerCallBack{};
156};
