23/// An iterator that iterates over a numeric range, where the start value is
24/// always incremented by one.
25template <typename T>
26class IotaIterator
27 : public llvm::iterator_facade_base<IotaIterator<T>,
28 std::random_access_iterator_tag, T,
29 ptrdiff_t, const T *, T> {
30 using base_t = llvm::iterator_facade_base<IotaIterator<T>,
31 std::random_access_iterator_tag, T,
32 ptrdiff_t, const T *, T>;
33
34public:
35 using typename base_t::difference_type;
36 using typename base_t::iterator_category;
37 using typename base_t::pointer;
38 using typename base_t::reference;
39 using typename base_t::value_type;
40
41 constexpr reference operator*() const noexcept { return Elem; }
42 constexpr pointer operator->() const noexcept { return &Elem; }
43
44 constexpr IotaIterator &operator+=(difference_type N) noexcept {
45 Elem = T(difference_type(Elem) + N);
46 return *this;
47 }
48 constexpr IotaIterator &operator-=(difference_type N) noexcept {
49 Elem = T(difference_type(Elem) - N);
50 return *this;
51 }
52 constexpr bool operator<(const IotaIterator &Other) const noexcept {
53 return difference_type(Other.Elem) < difference_type(Elem);
54 }
55 constexpr bool operator==(const IotaIterator &Other) const noexcept {
56 return Other.Elem == Elem;
57 }
58 constexpr bool operator!=(const IotaIterator &Other) const noexcept {
59 return !(*this == Other);
60 }
61 constexpr difference_type
62 operator-(const IotaIterator &Other) const noexcept {
63 return difference_type(Elem) - difference_type(Other.Elem);
64 }
65
66 constexpr explicit IotaIterator(T Elem) noexcept : Elem(Elem) {}
67
68 constexpr IotaIterator() noexcept = default;
69
70private:
71 T Elem{};
72};

74template <typename IdT>
75using IotaRangeType = llvm::iterator_range<IotaIterator<IdT>>;
76
77template <typename IdT>
78[[nodiscard]] constexpr auto iota(IdT From, type_identity_t<IdT> To) noexcept {
79 static_assert(is_explicitly_convertible_to<size_t, IdT> &&
80 is_explicitly_convertible_to<IdT, size_t>,
81 "Iota only works on integers and integer-like types");
82 using iterator_type = IotaIterator<std::decay_t<IdT>>;
83 auto Ret = llvm::make_range(iterator_type(From), iterator_type(To));
84 return Ret;
85}
86
87template <typename IdT> [[nodiscard]] constexpr auto iota(size_t To) noexcept {
88 static_assert(is_explicitly_convertible_to<size_t, IdT> &&
89 is_explicitly_convertible_to<IdT, size_t>,
90 "Iota only works on integers and integer-like types");
91 using iterator_type = IotaIterator<std::decay_t<IdT>>;
92 return llvm::make_range(iterator_type(), iterator_type(IdT(To)));
93}
94
95static_assert(is_iterable_over_v<IotaRangeType<int>, int>);
