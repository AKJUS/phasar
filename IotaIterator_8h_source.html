22/// An iterator that iterates over the same value a specified number of times
23template <typename T> class IotaIterator {
24public:
25 using value_type = T;
26 using reference = T;
27 using pointer = const T *;
28 using difference_type = ptrdiff_t;
29 using iterator_category = std::forward_iterator_tag;
30
31 constexpr reference operator*() const noexcept { return Elem; }
32 constexpr pointer operator->() const noexcept { return &Elem; }
33
34 constexpr IotaIterator &operator++() noexcept {
35 ++Elem;
36 return *this;
37 }
38 constexpr IotaIterator operator++(int) noexcept {
39 auto Ret = *this;
40 ++*this;
41 return Ret;
42 }
43
44 constexpr bool operator==(const IotaIterator &Other) const noexcept {
45 return Other.Elem == Elem;
46 }
47 constexpr bool operator!=(const IotaIterator &Other) const noexcept {
48 return !(*this == Other);
49 }
50
51 constexpr explicit IotaIterator(T Elem) noexcept : Elem(Elem) {}
52
53 constexpr IotaIterator() noexcept = default;
54
55private:
56 T Elem{};
57};
58
59template <typename T>
60using IotaRangeType = llvm::iterator_range<IotaIterator<T>>;
61template <typename T>
62constexpr auto iota(T From, type_identity_t<T> To) noexcept {
63 static_assert(std::is_integral_v<T>, "Iota only works on integers");
64 using iterator_type = IotaIterator<std::decay_t<T>>;
65 auto Ret = llvm::make_range(iterator_type(From), iterator_type(To));
66 return Ret;
67}
68
69static_assert(is_iterable_over_v<IotaRangeType<int>, int>);
