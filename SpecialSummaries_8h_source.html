37template <typename D, typename V = BinaryDomain>
38class [[deprecated("This ancient API is not maintained and should not be used "
39 "anymore")]] SpecialSummaries {
40 using FlowFunctionType = FlowFunction<D>;
41 using FlowFunctionPtrType = std::shared_ptr<FlowFunction<D>>;
42
43private:
44 std::map<std::string, FlowFunctionPtrType> SpecialFlowFunctions;
45 std::map<std::string, EdgeFunction<V>> SpecialEdgeFunctions;
46 std::vector<std::string> SpecialFunctionNames;
47
48 /// Constructs the SpecialSummaryMap such that it contains all glibc,
49 /// llvm.intrinsics and C++'s new, new[], delete, delete[] with identity
50 /// flow functions.
51 SpecialSummaries() {
52 // insert default flow and edge functions
53 for (const auto &FunctionName :
54 PhasarConfig::getPhasarConfig().specialFunctionNames()) {
55 // SpecialFlowFunctions.insert(
56 // std::make_pair(FunctionName, Identity<D>::getInstance()));
57 SpecialEdgeFunctions.insert(
58 std::make_pair(FunctionName, EdgeIdentity<V>{}));
59 }
60 }
61
62public:
63 SpecialSummaries(const SpecialSummaries &) = delete;
64 SpecialSummaries &operator=(const SpecialSummaries &) = delete;
65 SpecialSummaries(SpecialSummaries &&) = delete;
66 SpecialSummaries &operator=(SpecialSummaries &&) = delete;
67 ~SpecialSummaries() = default;
68
69 static SpecialSummaries<D, V> &getInstance() {
70 static SpecialSummaries<D, V> Instance;
71 return Instance;
72 }
73
74 /// Returns true, when an existing function is overwritten, false otherwise.
75 bool provideSpecialSummary(const std::string &Name,
76 FlowFunctionPtrType FlowFunc) {
77 bool Override = containsSpecialSummary(Name);
78 SpecialFlowFunctions[Name] = FlowFunc;
79 return Override;
80 }
81
82 /// Returns true, when an existing function is overwritten, false otherwise.
83 bool provideSpecialSummary(const std::string &Name,
84 FlowFunctionPtrType FlowFunc,
85 std::shared_ptr<EdgeFunction<V>> EdgeFunc) {
86 bool Override = containsSpecialSummary(Name);
87 SpecialFlowFunctions[Name] = FlowFunc;
88 SpecialEdgeFunctions[Name] = EdgeFunc;
89 return Override;
90 }
91
92 bool containsSpecialSummary(const llvm::Function *Func) {
93 return containsSpecialSummary(Func->getName());
94 }
95
96 bool containsSpecialSummary(const std::string &Name) {
97 return SpecialFlowFunctions.count(Name);
98 }
99
100 FlowFunctionPtrType
101 getSpecialFlowFunctionSummary(const llvm::Function *Func) {
102 return getSpecialFlowFunctionSummary(Func->getName());
103 }
104
105 FlowFunctionPtrType getSpecialFlowFunctionSummary(const std::string &Name) {
106 return SpecialFlowFunctions[Name];
107 }
108
109 std::shared_ptr<EdgeFunction<V>>
110 getSpecialEdgeFunctionSummary(const llvm::Function *Func) {
111 return getSpecialEdgeFunctionSummary(Func->getName());
112 }
113
114 std::shared_ptr<EdgeFunction<V>>
115 getSpecialEdgeFunctionSummary(const std::string &Name) {
116 return SpecialEdgeFunctions[Name];
117 }
118
119 friend llvm::raw_ostream &
120 operator<<(llvm::raw_ostream &OS, const SpecialSummaries<D> &SpecialSumms) {
121 OS << "SpecialSummaries:\n";
122 for (auto &Entry : SpecialSumms.SpecialFunctionNames) {
123 OS << Entry.first << " ";
124 }
125 return OS;
126 }
127};
