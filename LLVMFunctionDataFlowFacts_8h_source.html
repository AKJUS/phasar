59 [[nodiscard]] const std::vector<DataFlowFact> &
60 getFacts(const llvm::Function *Fun, uint32_t Index) {
61 auto Iter = LLVMFdff.find(Fun);

63 return Iter->second[Index];

66 }

71

74 auto Iter = LLVMFdff.find(Fun);
75 if (Iter != LLVMFdff.end()) {

77 }

84private:
