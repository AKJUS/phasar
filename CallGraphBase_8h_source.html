22template <typename T>
23concept IsCallGraph =
24 requires(const T &CG, typename T::n_t Inst, typename T::f_t Fun) {
25 typename T::n_t;
26 typename T::f_t;
27
28 /// Returns an iterable range of all possible callee candidates at the
29 /// given call-site induced by the used call-graph.
30 ///
31 /// NOTE: This function is typically called in a hot part of the analysis
32 /// and should therefore be very fast
33 {
34 CG.getCalleesOfCallAt(Inst)
35 } -> psr::is_iterable_over_v<typename T::f_t>;
36
37 /// Returns an iterable range of all possible call-site candidates that
38 /// may call the given function induced by the used call-graph.
39 { CG.getCallersOf(Fun) } -> psr::is_iterable_over_v<typename T::n_t>;
40
41 /// A range of all functions that are vertices in the call-graph. The
42 /// number of vertex functions can be retrieved by
43 /// getNumVertexFunctions().
44 {
45 CG.getAllVertexFunctions()
46 } -> psr::is_iterable_over_v<typename T::f_t>;
47
48 /// A range of all call-sites that are vertices in the call-graph. The
49 /// number of vertex-callsites can be retrived by getNumVertexCallSites().
50 {
51 CG.getAllVertexCallSites()
52 } -> psr::is_iterable_over_v<typename T::n_t>;
53
54 { CG.getNumVertexFunctions() } -> std::convertible_to<size_t>;
55 { CG.getNumVertexCallSites() } -> std::convertible_to<size_t>;
56
57 /// Same as getNumVertexFunctions()
58 { CG.size() } noexcept -> std::convertible_to<size_t>;
59 { CG.empty() } noexcept -> std::convertible_to<bool>;
60 };

70/// Base class of all CallGraph implementations within phasar (currently only
71/// CallGraph<N, F>).
72/// Only represents the data, not how to create it.
73template <typename Derived> class CallGraphBase : public CRTPBase<Derived> {
74 friend Derived;
75 using CRTPBase<Derived>::self;
76
77public:
78 using n_t = typename CGTraits<Derived>::n_t;
79 using f_t = typename CGTraits<Derived>::f_t;
80
81 /// Returns an iterable range of all possible callee candidates at the given
82 /// call-site induced by the used call-graph.
83 ///
84 /// NOTE: This function is typically called in a hot part of the analysis and
85 /// should therefore be very fast
86 [[nodiscard]] constexpr decltype(auto)
87 getCalleesOfCallAt(ByConstRef<n_t> Inst) const
88 noexcept(noexcept(this->self().getCalleesOfCallAtImpl(Inst))) {
89 static_assert(
90 is_iterable_over_v<decltype(self().getCalleesOfCallAtImpl(Inst)), f_t>);
91 return self().getCalleesOfCallAtImpl(Inst);
92 }
93
94 /// Returns an iterable range of all possible call-site candidates that may
95 /// call the given function induced by the used call-graph.
96 [[nodiscard]] constexpr decltype(auto)
97 getCallersOf(ByConstRef<f_t> Fun) const {
98 static_assert(
99 is_iterable_over_v<decltype(this->self().getCallersOfImpl(Fun)), n_t>);
100 return self().getCallersOfImpl(Fun);
101 }
102
103 /// A range of all functions that are vertices in the call-graph. The number
104 /// of vertex functions can be retrieved by getNumVertexFunctions().
105 [[nodiscard]] constexpr decltype(auto)
106 getAllVertexFunctions() const noexcept {
107 return self().getAllVertexFunctionsImpl();
108 }
109
110 /// A range of all call-sites that are vertices in the call-graph. The number
111 /// of vertex-callsites can be retrived by getNumVertexCallSites().
112 [[nodiscard]] constexpr auto getAllVertexCallSites() const noexcept {
113 return self().getAllVertexCallSitesImpl();
114 }
115
116 [[nodiscard]] constexpr size_t getNumVertexFunctions() const noexcept {
117 return self().getNumVertexFunctionsImpl();
118 }
119 [[nodiscard]] constexpr size_t getNumVertexCallSites() const noexcept {
120 return self().getNumVertexCallSitesImpl();
121 }
122
123 /// The number of functions within this call-graph
124 [[nodiscard]] constexpr size_t size() const noexcept {
125 return getNumVertexFunctions();
126 }
127
128 [[nodiscard]] constexpr bool empty() const noexcept { return size() == 0; }
129};
