23/// Base class of all CallGraph implementations within phasar (currently only
24/// CallGraph<N, F>).
25/// Only represents the data, not how to create it.
26template <typename Derived> class CallGraphBase : public CRTPBase<Derived> {
27 friend Derived;
28 using CRTPBase<Derived>::self;
29
30public:
31 using n_t = typename CGTraits<Derived>::n_t;
32 using f_t = typename CGTraits<Derived>::f_t;
33
34 /// Returns an iterable range of all possible callee candidates at the given
35 /// call-site induced by the used call-graph.
36 ///
37 /// NOTE: This function is typically called in a hot part of the analysis and
38 /// should therefore be very fast
39 [[nodiscard]] constexpr decltype(auto)
40 getCalleesOfCallAt(ByConstRef<n_t> Inst) const
41 noexcept(noexcept(this->self().getCalleesOfCallAtImpl(Inst))) {
42 static_assert(
43 is_iterable_over_v<decltype(self().getCalleesOfCallAtImpl(Inst)), f_t>);
44 return self().getCalleesOfCallAtImpl(Inst);
45 }
46
47 /// Returns an iterable range of all possible call-site candidates that may
48 /// call the given function induced by the used call-graph.
49 [[nodiscard]] constexpr decltype(auto)
50 getCallersOf(ByConstRef<f_t> Fun) const {
51 static_assert(
52 is_iterable_over_v<decltype(this->self().getCallersOfImpl(Fun)), n_t>);
53 return self().getCallersOfImpl(Fun);
54 }
55
56 /// A range of all functions that are vertices in the call-graph. The number
57 /// of vertex functions can be retrieved by getNumVertexFunctions().
58 [[nodiscard]] constexpr decltype(auto)
59 getAllVertexFunctions() const noexcept {
60 return self().getAllVertexFunctionsImpl();
61 }
62
63 /// A range of all call-sites that are vertices in the call-graph. The number
64 /// of vertex-callsites can be retrived by getNumVertexCallSites().
65 [[nodiscard]] constexpr auto getAllVertexCallSites() const noexcept {
66 return self().getAllVertexCallSitesImpl();
67 }
68
69 [[nodiscard]] constexpr size_t getNumVertexFunctions() const noexcept {
70 return self().getNumVertexFunctionsImpl();
71 }
72 [[nodiscard]] constexpr size_t getNumVertexCallSites() const noexcept {
73 return self().getNumVertexCallSitesImpl();
74 }
75
76 /// The number of functions within this call-graph
77 [[nodiscard]] constexpr size_t size() const noexcept {
78 return getNumVertexFunctions();
79 }
80
81 [[nodiscard]] constexpr bool empty() const noexcept { return size() == 0; }
82};
83} // namespace psr
