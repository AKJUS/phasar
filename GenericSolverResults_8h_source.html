24/// \brief A type-erased version of the main functionality of SolverResults.
25/// Can be accepted by consumers that don't need deep access to the internals
26/// (so, the usual ones). As we have now two kinds of solver-results
27/// (SolverResults and IdBasedSolverResults), we need a common way of accessing
28/// them.

35 /// NOTE: Don't provide an operator= that constructs a GenericSolverResults
36 /// from a concrete SolverResults object to prevent issues with object
37 /// lifetime. Probably that can be mitigated with use of std::launder, but for
38 /// now, we don't need this complexity

40 template <typename T, typename = std::enable_if_t<!std::is_same_v<

42 GenericSolverResults(const T &SR) noexcept : VT(&VtableFor<std::decay_t<T>>) {

164template <typename SR1, typename SR2>
165bool ifdsEqual(const SR1 &LHS, const SR2 &RHS) {
166 if (LHS.size() != RHS.size()) {
167 return false;
168 }
169
170 for (const auto &[Row, ColVal] : LHS.rowMapView()) {
171 if (!RHS.containsNode(Row)) {
172 return false;
173 }
174
175 const auto &OtherColVal = LHS.row(Row);
176 if (ColVal.size() != OtherColVal.size()) {
177 return false;
178 }
179
180 for (const auto &[Col, Val] : ColVal) {
181 if (!OtherColVal.count(Col)) {
182 return false;
183 }
184 }
185 }
186 return true;
187}

189template <typename SR1, typename SR2>
190bool checkSREquality(const SR1 &LHS, const SR2 &RHS) {
191 bool HasError = false;
192 if (LHS.size() != RHS.size()) {
193 HasError = true;
194 llvm::errs() << "The results sizes do not match: " << LHS.size() << " vs "
195 << RHS.size() << '\n';
196 }
197
198 auto ToString = [](const auto &Fact) {
199 if constexpr (std::is_pointer_v<std::decay_t<decltype(Fact)>> &&
200 std::is_base_of_v<llvm::Value,
201 std::decay_t<std::remove_pointer_t<
202 std::decay_t<decltype(Fact)>>>>) {
203 return llvmIRToString(Fact);
204 } else {
205 std::string S;
206 llvm::raw_string_ostream ROS(S);
207 ROS << Fact;
208 ROS.flush();
209 return S;
210 }
211 };
212
213 for (const auto &[Row, ColVal] : LHS.rowMapView()) {
214 if (!RHS.containsNode(Row)) {
215 HasError = true;
216 llvm::errs() << "RHS does not contain row for Inst: " << ToString(Row)
217 << '\n';
218 }
219
220 auto RHSColVal = RHS.row(Row);
221
222 if (ColVal.size() != RHSColVal.size()) {
223 HasError = true;
224 llvm::errs() << "The number of facts at " << ToString(Row)
225 << " differ: " << ColVal.size() << " vs " << RHSColVal.size()
226 << '\n';
227 }
228
229 for (const auto &[Col, Val] : ColVal) {
230 auto It = RHSColVal.find(Col);
231
232 if (It == RHSColVal.end()) {
233 HasError = true;
234 llvm::errs() << "RHS does not contain fact " << ToString(Col)
235 << " at Inst " << ToString(Row) << '\n';
236 } else if (Val != It->second) {
237 HasError = true;
238 llvm::errs() << "The edge values at inst " << ToString(Row)
239 << " and fact " << ToString(Col)
240 << " differ: " << ToString(Val) << " vs "
241 << ToString(It->second) << '\n';
242 }
243 }
244 }
245 return !HasError;
246}
