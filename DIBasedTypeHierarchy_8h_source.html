28/// \brief Represents the type hierarchy of the target program.
29///
30/// \note This class only works, if the target program's IR was generated with
31/// debug information. Pass `-g` to the compiler to achieve this.
32class DIBasedTypeHierarchy
33 : public TypeHierarchy<const llvm::DIType *, const llvm::Function *> {
34public:
35 using ClassType = const llvm::DIType *;
36 using f_t = const llvm::Function *;
37
38 static inline constexpr llvm::StringLiteral StructPrefix = "struct.";
39 static inline constexpr llvm::StringLiteral ClassPrefix = "class.";
40 static inline constexpr llvm::StringLiteral VTablePrefix = "_ZTV";
41 static inline constexpr llvm::StringLiteral VTablePrefixDemang =
42 "vtable for ";
43 static inline constexpr llvm::StringLiteral PureVirtualCallName =
44 "__cxa_pure_virtual";
45
46 /// \brief Creates a type hierarchy based on an intermediate representation
47 /// database.
48 /// \param[in] IRDB The IR database of which the type hierarchy will be based
49 /// upon. This MUST contain debug information for the algorithm to work!
50 explicit DIBasedTypeHierarchy(const LLVMProjectIRDB &IRDB);
51
52 /// \brief Loads an already computed type hierarchy.
53 /// \param[in] IRDB The IR database of the type hierarchy.
54 /// \param[in] SerializedData The already existing type hierarchy, given by
55 /// the appropiate class DIBasedTypeHierarchyData, which contains all
56 /// neccesary information.
57 explicit DIBasedTypeHierarchy(const LLVMProjectIRDB *IRDB,
58 const DIBasedTypeHierarchyData &SerializedData);
59 ~DIBasedTypeHierarchy() override = default;
60
61 [[nodiscard]] bool hasType(ClassType Type) const override {
62 return TypeToVertex.count(Type);
63 }
64
65 [[nodiscard]] bool isSubType(ClassType Type,
66 ClassType SubType) const override {
67 return llvm::is_contained(subTypesOf(Type), SubType);
68 }
69
70 [[nodiscard]] std::set<ClassType> getSubTypes(ClassType Type) const override {
71 const auto &Range = subTypesOf(Type);
72 return {Range.begin(), Range.end()};
73 }
74
75 /// A more efficient version of getSubTypes()
76 [[nodiscard]] llvm::iterator_range<const ClassType *>
77 subTypesOf(ClassType Ty) const noexcept;
78
79 [[nodiscard]] ClassType
80 getType(llvm::StringRef TypeName) const noexcept override {
81 return NameToType.lookup(TypeName);
82 }
83
84 [[nodiscard]] std::vector<ClassType> getAllTypes() const override {
85 return {VertexTypes.begin(), VertexTypes.end()};
86 }
87
88 [[nodiscard]] const auto &getAllVTables() const noexcept { return VTables; }
89
90 [[nodiscard]] static llvm::StringRef typeName(ClassType Type) {
91 if (const auto *CompTy = llvm::dyn_cast<llvm::DICompositeType>(Type)) {
92 auto Ident = CompTy->getIdentifier();
93 return Ident.empty() ? CompTy->getName() : Ident;
94 }
95 return Type->getName();
96 }
97
98 [[nodiscard]] llvm::StringRef getTypeName(ClassType Type) const override {
99 return typeName(Type);
100 }
101
102 [[nodiscard]] size_t size() const noexcept override {
103 return VertexTypes.size();
104 }
105 [[nodiscard]] bool empty() const noexcept override {
106 return VertexTypes.empty();
107 }
108
109 void print(llvm::raw_ostream &OS = llvm::outs()) const override;
110
111 /// \brief Prints the class hierarchy to an ostream in dot format.
112 /// \param OS outputstream
113 void printAsDot(llvm::raw_ostream &OS = llvm::outs()) const;
114
115 /// \brief Prints the class hierarchy to an ostream in JSON format.
116 /// \param OS outputstream
117 void printAsJson(llvm::raw_ostream &OS = llvm::outs()) const override;
118
119private:
120 [[nodiscard]] DIBasedTypeHierarchyData getTypeHierarchyData() const;
121 [[nodiscard]] llvm::iterator_range<const ClassType *>
122 subTypesOf(size_t TypeIdx) const noexcept;
123
124 // ---
125
126 llvm::StringMap<ClassType> NameToType;
127 // Map each type to an integer index that is used by VertexTypes and
128 // DerivedTypesOf.
129 // Note: all the below arrays should always have the same size (except for
130 // Hierarchy)!
131 llvm::DenseMap<ClassType, size_t> TypeToVertex;
132 // The class types we care about ("VertexProperties")
133 std::vector<const llvm::DICompositeType *> VertexTypes;
134 std::vector<std::pair<uint32_t, uint32_t>> TransitiveDerivedIndex;
135 // The inheritance graph linearized as-if constructed by L2R pre-order
136 // traversal from the roots. Allows efficient access to the transitive closure
137 // without ever storing it explicitly. This only works, because the type-graph
138 // is known to never contain loops
139 std::vector<ClassType> Hierarchy;
140
141 // The VTables of the polymorphic types in the TH. default-constructed if not
142 // exists
143 std::deque<LLVMVFTable> VTables;
144};
145} // namespace psr
