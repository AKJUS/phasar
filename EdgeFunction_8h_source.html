39template <typename T>
40concept IsEdgeFunction =
41 requires(const T &EF, const EdgeFunction<typename T::l_t> &TEEF,
42 EdgeFunctionRef<T> CEF, typename T::l_t Src) {
43 typename T::l_t;
44 { EF.computeTarget(Src) } -> std::convertible_to<typename T::l_t>;
45 { T::compose(CEF, TEEF) } -> std::same_as<EdgeFunction<typename T::l_t>>;
46 { T::join(CEF, TEEF) } -> std::same_as<EdgeFunction<typename T::l_t>>;
47 };
48
49enum class EdgeFunctionAllocationPolicy {
50 SmallObjectOptimized,
51 DefaultHeapAllocated,
52 CustomHeapAllocated,
53};
54
55class EdgeFunctionBase {
56public:
57 template <typename ConcreteEF>
58 static constexpr bool IsSOOCandidate =
59 sizeof(ConcreteEF) <= sizeof(void *) && // NOLINT
60 alignof(ConcreteEF) <= alignof(void *) &&
61 std::is_trivially_copyable_v<ConcreteEF>;
62
63 using AllocationPolicy = EdgeFunctionAllocationPolicy;
64
65protected:
66 struct RefCountedBase {
67 mutable std::atomic_size_t Rc = 0;
68 };
69 template <typename T> struct RefCounted : RefCountedBase {
70 T Value;
71 };
72
73 template <typename T> struct CachedRefCounted : RefCounted<T> {
74 EdgeFunctionSingletonCache<T> *Cache{};
75 };
76
77 template <typename ConcreteEF>
78 constexpr static inline const ConcreteEF *
79 getPtr(const void *const &EF) noexcept {
80 if constexpr (!IsSOOCandidate<ConcreteEF>) {
81 return &static_cast<const RefCounted<ConcreteEF> *>(EF)->Value;
82 } else {
83 return static_cast<const ConcreteEF *>(static_cast<const void *>(&EF));
84 }
85 }
86 template <typename ConcreteEF>
87 constexpr static inline const ConcreteEF *
88 getPtr(const void *const &&EF) = delete; // NOLINT
89
90 template <typename ConcreteEF>
91 static constexpr AllocationPolicy DefaultAllocPolicy =
92 IsSOOCandidate<ConcreteEF> ? AllocationPolicy::SmallObjectOptimized
93 : AllocationPolicy::DefaultHeapAllocated;
94 template <typename ConcreteEF>
95 static constexpr AllocationPolicy CustomAllocPolicy =
96 IsSOOCandidate<ConcreteEF> ? AllocationPolicy::SmallObjectOptimized
97 : AllocationPolicy::CustomHeapAllocated;
98};

100/// \brief Non-null reference to an edge function that is guarenteed to be
101/// managed by an EdgeFunction object.
102template <typename EF>
103class [[clang::trivial_abi]] EdgeFunctionRef final : EdgeFunctionBase {
104 template <typename L> friend class EdgeFunction;
105
106public:
107 using l_t = typename EF::l_t;
108
109 EdgeFunctionRef(const EdgeFunctionRef &) noexcept = default;
110 EdgeFunctionRef &operator=(const EdgeFunctionRef &) noexcept = default;
111 ~EdgeFunctionRef() = default;
112
113 const EF *operator->() const noexcept { return getPtr<EF>(Instance); }
114 const EF *get() const noexcept { return getPtr<EF>(Instance); }
115 const EF &operator*() const noexcept { return *getPtr<EF>(Instance); }
116
117 [[nodiscard]] bool isCached() const noexcept {
118 if constexpr (IsSOOCandidate<EF>) {
119 return false;
120 } else {
121 return IsCached;
122 }
123 }
124
125 [[nodiscard]] EdgeFunctionSingletonCache<EF> *
126 getCacheOrNull() const noexcept {
127 if (isCached()) {
128 return static_cast<const CachedRefCounted<EF> *>(Instance)->Cache;
129 }
130
131 return nullptr;
132 }
133
134private:
135 explicit EdgeFunctionRef(const void *Instance, bool IsCached) noexcept
136 : Instance(Instance) {
137 if constexpr (!IsSOOCandidate<EF>) {
138 this->IsCached = IsCached;
139 }
140 }
141 const void *Instance{};
142 [[no_unique_address]] std::conditional_t<IsSOOCandidate<EF>, EmptyType, bool>
143 IsCached{};
144};

146/// \brief Ref-counted and type-erased edge function with small-object
147/// optimization. Supports caching.
148template <typename L>
149// -- combined copy and move assignment
150// NOLINTNEXTLINE(cppcoreguidelines-special-member-functions)
151class [[clang::trivial_abi]] EdgeFunction final : EdgeFunctionBase {
152public:
153 using l_t = L;
154
155 // --- Constructors
156
157 /// Default-initializes the edge-function with nullptr
158 EdgeFunction() noexcept = default;
159 /// Default-initializes the edge-function with nullptr
160 EdgeFunction(std::nullptr_t) noexcept : EdgeFunction() {}
161 /// Copy constructor. Increments the ref-count, if not small-object-optimized.
162 EdgeFunction(const EdgeFunction &Other) noexcept
163 : EdgeFunction(Other.EF, Other.VTAndHeapAlloc) {}
164 /// Move constructor. Does not increment the ref-count, but instead leaves the
165 /// moved-from edge function in the nullptr state.
166 EdgeFunction(EdgeFunction &&Other) noexcept
167 : EF(std::exchange(Other.EF, nullptr)),
168 VTAndHeapAlloc(
169 std::exchange(Other.VTAndHeapAlloc, decltype(VTAndHeapAlloc){})) {}
170
171 /// Standard swap; does not affect ref-counts
172 void swap(EdgeFunction &Other) noexcept {
173 std::swap(EF, Other.EF);
174 std::swap(VTAndHeapAlloc, Other.VTAndHeapAlloc);
175 }
176 /// Standard swap; does not affect ref-counts
177 friend void swap(EdgeFunction &LHS, EdgeFunction &RHS) noexcept {
178 LHS.swap(RHS);
179 }
180
181 /// Combined copy- and move assignment. If the assigned-to edge function is
182 /// not null, invokes the destructor on it, before overwriting its content.
183 EdgeFunction &operator=(EdgeFunction Other) noexcept {
184 std::destroy_at(this);
185 return *new (this) EdgeFunction(std::move(Other)); // NOLINT
186 }
187 /// Null-assignment operator. Decrements the ref-count if not
188 /// small-object-optimized or already null. Leaves the assigned-to edge
189 /// function in the nullptr state.
190 EdgeFunction &operator=(std::nullptr_t) noexcept {
191 std::destroy_at(this);
192 return *new (this) EdgeFunction(); // NOLINT
193 }
194
195 /// Destructor. Decrements the ref-count if not small-object-optimized and
196 /// destroyes the held edge function once the ref-count reaches 0.
197 ~EdgeFunction() noexcept {
198 AllocationPolicy Policy = VTAndHeapAlloc.getInt();
199 if (Policy != AllocationPolicy::SmallObjectOptimized) {
200 assert(VTAndHeapAlloc.getPointer() != nullptr && "Heap-alloc'd nullptr?");
201 // Note: Memory-order taken from llvm::ThreadSafeRefCountedBase
202 if (static_cast<const RefCountedBase *>(EF)->Rc.fetch_sub(
203 1, std::memory_order_acq_rel) == 1) {
204 VTAndHeapAlloc.getPointer()->destroy(EF, Policy);
205 }
206 }
207 }

209 /// Implicit-conversion constructor from EdgeFunctionRef. Increments the
210 /// ref-count if not small-object optimized
211 template <typename ConcreteEF>
212 requires(!std::is_same_v<EdgeFunction, std::decay_t<ConcreteEF>>)
213 EdgeFunction(EdgeFunctionRef<ConcreteEF> CEF) noexcept
214 : EdgeFunction(CEF.Instance,
215 {&VTableFor<ConcreteEF>, [CEF] {
216 if constexpr (IsSOOCandidate<ConcreteEF>) {
217 (void)CEF;
218 return AllocationPolicy::SmallObjectOptimized;
219 } else {
220 return CEF.IsCached
221 ? AllocationPolicy::CustomHeapAllocated
222 : AllocationPolicy::DefaultHeapAllocated;
223 }
224 }()}) {}
225
226 /// Conversion-constructor from any edge function (that satisfies the
227 /// IsEdgeFunction trait). Stores a type-erased copy of CEF and allocates
228 /// space for it on the heap if small-object-optimization cannot be applied.
229 template <IsEdgeFunction ConcreteEF>
230 requires(!std::is_same_v<EdgeFunction, std::decay_t<ConcreteEF>>)
231 EdgeFunction(ConcreteEF &&CEF) noexcept(
232 IsSOOCandidate<std::decay_t<ConcreteEF>>)
233 : EdgeFunction(std::in_place_type<std::decay_t<ConcreteEF>>,
234 std::forward<ConcreteEF>(CEF)) {}
235
236 /// Emplacement-constructor for any edge function. Constructs a new object of
237 /// type ConcreteEF with the given constructor arguments and allocates space
238 /// for it on the heap if small-object-optimization cannot be applied.
239 /// No extra copy- or move construction/assignment is performed. Use this ctor
240 /// if even moving is expensive.
241 template <IsEdgeFunction ConcreteEF, typename... ArgTys>
242 explicit EdgeFunction(
243 std::in_place_type_t<ConcreteEF> /*unused*/,
244 ArgTys &&...Args) noexcept(IsSOOCandidate<std::decay_t<ConcreteEF>> &&
245 std::is_nothrow_constructible_v<ConcreteEF,
246 ArgTys...>)
247 : EdgeFunction(
248 [](auto &&...Args) {
249 if constexpr (IsSOOCandidate<std::decay_t<ConcreteEF>>) {
250 void *Ret = nullptr;
251 new (&Ret) ConcreteEF(std::forward<ArgTys>(Args)...);
252 return Ret;
253 } else {
254 return new RefCounted<ConcreteEF>{
255 {}, {std::forward<ArgTys>(Args)...}};
256 }
257 }(std::forward<ArgTys>(Args)...),
258 {&VTableFor<ConcreteEF>, DefaultAllocPolicy<ConcreteEF>}) {
259 static_assert(std::is_same_v<l_t, typename ConcreteEF::l_t>,
260 "Cannot construct EdgeFunction with incompatible "
261 "lattice domain");
262 }
263
264 /// Conversion-constructor for any edge function with enabled caching. Stores
265 /// a type-erased copy of EF.EF and allocates space for it on the heap if
266 /// small-object-optimization cannot be applied. If a heap allocation is
267 /// requires, first consults the EF.Cache to check whether an equivalent edge
268 /// function is already allocated. If so, takes the one from the cache and
269 /// increases its ref-count. Otherwise, performs a fresh allocation and
270 /// inserts the edge function into EF.Cache.
271 /// When all references to this edge function are out-of-scope, the destructor
272 /// automatically removes the edge function from EF.Cache. Hence, make sure
273 /// that EF.Cache lives at least as long as the last edge function cached in
274 /// it.
275 template <IsEdgeFunction ConcreteEF>
276 requires std::is_move_constructible_v<ConcreteEF>
277 EdgeFunction(CachedEdgeFunction<ConcreteEF> EF)
278 : EdgeFunction(
279 [&EF] {
280 assert(EF.Cache != nullptr);
281 if constexpr (IsSOOCandidate<std::decay_t<ConcreteEF>>) {
282 void *Ret;
283 new (&Ret) ConcreteEF(std::move(EF.EF));
284 return Ret;
285 } else {
286 if (const auto *Mem = EF.Cache->lookup(EF.EF)) {
287 return static_cast<const RefCounted<ConcreteEF> *>(Mem);
288 }
289
290 auto Ret = new CachedRefCounted<ConcreteEF>{
291 {{}, {std::move(EF.EF)}}, EF.Cache};
292 EF.Cache->insert(&Ret->Value, Ret);
293 return static_cast<const RefCounted<ConcreteEF> *>(Ret);
294 }
295 }(),
296 {&VTableFor<ConcreteEF>, CustomAllocPolicy<ConcreteEF>}) {}

300 ///
301 /// This function describes the concrete value computation for its respective
302 /// exploded supergraph edge. The function(s) will be evaluated once the
303 /// exploded supergraph has been constructed and the concrete values of the
304 /// various value computation problems along the supergraph edges are
305 /// evaluated.
306 ///
307 /// Please also refer to the various edge function factories of the
308 /// EdgeFunctions interface: EdgeFunctions::get*EdgeFunction() for more
309 /// details.
310 ///
311 [[nodiscard]] l_t computeTarget(ByConstRef<l_t> Source) const {
312 assert(isValid() && "computeTarget() called on nullptr!");
313 return VTAndHeapAlloc.getPointer()->computeTarget(EF, Source);
314 }

316 ///
317 /// This function composes the two edge functions this and SecondEF. This
318 /// function is used to extend an edge function in order to construct
319 /// so-called jump functions that describe the effects of everlonger sequences
320 /// of code.
321 ///
322 /// Calls the static function EF::compose(*this, SecondEF) for your concrete
323 /// edge function EF.
324 ///
325 /// For semantic correctness, please make sure that for all inputs x in l_t,
326 /// it holds: this->composeWith(SecondEF).computeTarget(x) ==
327 /// SecondEF.computeTarget(this->computeTarget(x)).
328 [[nodiscard]] EdgeFunction composeWith(const EdgeFunction &SecondEF) const {
329 return compose(*this, SecondEF);
330 }
331
332 ///
333 /// This function composes the two edge functions this and SecondEF. This
334 /// function is used to extend an edge function in order to construct
335 /// so-called jump functions that describe the effects of everlonger sequences
336 /// of code.
337 ///
338 /// For semantic correctness, please make sure that for all inputs x in l_t,
339 /// it holds: compose(FirstEF, SecondEF).computeTarget(x) ==
340 /// SecondEF.computeTarget(FirstEF.computeTarget(x)).
341 [[nodiscard]] static EdgeFunction compose(const EdgeFunction &FirstEF,
342 const EdgeFunction &SecondEF) {
343 assert(FirstEF.isValid() && "compose() called on LHS nullptr!");
344 assert(SecondEF.isValid() && "compose() called on RHS nullptr!");
345 return FirstEF.VTAndHeapAlloc.getPointer()->compose(
346 FirstEF.EF, SecondEF, FirstEF.VTAndHeapAlloc.getInt());
347 }
348
349 ///
350 /// This function describes the join of the two edge functions this and
351 /// OtherEF. The function is called whenever two edge functions need to
352 /// be joined, for instance, when two branches lead to a common successor
353 /// instruction.
354 ///
355 /// Calls the static function EF::join(*this, OtherEF) for your concrete
356 /// edge function EF.
357 ///
358 /// For semantic correctness, please make sure that for all inputs x in l_t,
359 /// it holds (with join = JoinLatticeTraits<l_t>::join):
360 /// join(this->joinWith(OtherEF).computeTarget(x)),
361 /// join(this->computeTarget(x), OtherEF.computeTarget(x))) ==
362 /// this->joinWith(OtherEF).computeTarget(x)
363 /// i.e. that joining on edge functions only goes up the lattice that is
364 /// connected with the value-lattice on l_t
365 [[nodiscard]] EdgeFunction joinWith(const EdgeFunction &OtherEF) const {
366 return join(*this, OtherEF);
367 }
368
369 ///
370 /// This function describes the join of the two edge functions this and
371 /// OtherEF. The function is called whenever two edge functions need to
372 /// be joined, for instance, when two branches lead to a common successor
373 /// instruction.
374 ///
375 /// For semantic correctness, please make sure that for all inputs x in l_t,
376 /// it holds (with joinl = JoinLatticeTraits<l_t>::join):
377 /// joinl(join(FirstEF, SecondEF).computeTarget(x)),
378 /// joinl(FirstEF.computeTarget(x), SecondEF.computeTarget(x))) ==
379 /// this->joinWith(OtherEF).computeTarget(x)
380 /// i.e. that joining on edge functions only goes up the lattice that is
381 /// connected with the value-lattice on l_t
382 [[nodiscard]] static EdgeFunction join(const EdgeFunction &FirstEF,
383 const EdgeFunction &SecondEF) {
384 assert(FirstEF.isValid() && "join() called on LHS nullptr!");
385 assert(SecondEF.isValid() && "join() called on RHS nullptr!");
386 return FirstEF.VTAndHeapAlloc.getPointer()->join(
387 FirstEF.EF, SecondEF, FirstEF.VTAndHeapAlloc.getInt());
388 }

390 [[nodiscard]] bool
391 referenceEquals(const EdgeFunction<L> &Other) const noexcept {
392 return VTAndHeapAlloc.getPointer() == Other.VTAndHeapAlloc.getPointer() &&
393 EF == Other.EF;

396 /// Checks for equality of two edge functions. Equality requires exact
397 /// type-equality and value-equality based on operator== of the concrete edge
398 /// functions that are compared.
399 ///
400 /// If the concrete edge function has no nonstatic data members, the equality
401 /// is defaulted to only compare the types. However, by explicitly defining
402 /// operator==, this behavior can be overridden.
403 [[nodiscard]] friend bool operator==(const EdgeFunction &LHS,
404 const EdgeFunction &RHS) noexcept {
405 if (LHS.VTAndHeapAlloc.getPointer() != RHS.VTAndHeapAlloc.getPointer()) {
406 return false;
407 }
408 if (LHS.VTAndHeapAlloc.getOpaqueValue() == nullptr) {
409 return true;
410 }
411 return LHS.EF == RHS.EF ||
412 LHS.VTAndHeapAlloc.getPointer()->equals(LHS.EF, RHS.EF);
413 }

513 /// Performs a runtime-typecheck. True, if the concrete type of the held edge
514 /// function *exactly* equals ConcreteEF.

517 /// functions of a base class BaseEF and a derived class DerivedEF are

638 template <typename ConcreteEF>
639 static constexpr VTable VTableFor = {
640 [](const void *EF, ByConstRef<l_t> Source) {
641 return getPtr<ConcreteEF>(EF)->computeTarget(Source);
642 },
643 [](const void *EF, const EdgeFunction &SecondEF,
644 AllocationPolicy Policy) {
645 return ConcreteEF::compose(
646 EdgeFunctionRef<ConcreteEF>(
647 EF, Policy == AllocationPolicy::CustomHeapAllocated),

651 return ConcreteEF::join(
652 EdgeFunctionRef<ConcreteEF>(
653 EF, Policy == AllocationPolicy::CustomHeapAllocated),
654 OtherEF);
655 },
656 [](const void *EF1, const void *EF2) noexcept {
657 static_assert(IsEqualityComparable<ConcreteEF> ||
658 std::is_empty_v<ConcreteEF>,
659 "An EdgeFunction must be equality comparable with "
660 "operator==. Only if the type is empty, i.e. has no "
661 "members, the comparison can be inferred.");
662 if constexpr (IsEqualityComparable<ConcreteEF>) {
663 return *getPtr<ConcreteEF>(EF1) == *getPtr<ConcreteEF>(EF2);
664 } else {
665 return true;
666 }
667 },
668 [](const void *EF, llvm::raw_ostream &OS) {
669 if constexpr (is_llvm_printable_v<ConcreteEF>) {
670 OS << *getPtr<ConcreteEF>(EF);
671 } else {
672 OS << llvm::getTypeName<ConcreteEF>();
673 }
674 },
675 [](const void *EF) noexcept {
676 if constexpr (HasIsConstant<ConcreteEF>) {
677 static_assert(
678 std::is_nothrow_invocable_v<decltype(&ConcreteEF::isConstant),
679 const ConcreteEF &>,
680 "The function isConstant() must be noexcept!");
681 return getPtr<ConcreteEF>(EF)->isConstant();
682 } else {
683 return false;
684 }
685 },
686 [](const void *EF, AllocationPolicy Policy) noexcept {
687 if constexpr (!IsSOOCandidate<ConcreteEF>) {
688 if (Policy != AllocationPolicy::CustomHeapAllocated) {
689 assert(Policy == AllocationPolicy::DefaultHeapAllocated);
690 delete static_cast<const RefCounted<ConcreteEF> *>(EF);
691 } else {
692 auto CEF = static_cast<const CachedRefCounted<ConcreteEF> *>(EF);
693 CEF->Cache->erase(CEF->Value);
694 delete CEF;
695 }
696 }
697 },
698 [](const void *EF, const void *VT) noexcept -> size_t {
699 if constexpr (is_std_hashable_v<ConcreteEF>) {
700 return std::hash<ConcreteEF>{}(*getPtr<ConcreteEF>(EF));
701 } else if constexpr (is_llvm_hashable_v<ConcreteEF>) {
702 using llvm::hash_value;
703 return hash_value(*getPtr<ConcreteEF>(EF));
704 } else {
705 return llvm::hash_combine(EF, VT);
706 }
707 },
708 [](const void *EF) noexcept -> size_t {
709 if constexpr (HasDepth<ConcreteEF>) {
710 return getPtr<ConcreteEF>(EF)->depth();
711 } else {
712 return 1;
713 }
714 },
715 };
