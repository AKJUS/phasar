41template <typename T, typename = void>
42struct IsEdgeFunction : std::false_type {};
43template <typename T>
44struct IsEdgeFunction<
45 T, std::void_t<
46 typename T::l_t,
47 decltype(std::declval<const T &>().computeTarget(
48 std::declval<typename T::l_t>())),
49 decltype(T::compose(std::declval<EdgeFunctionRef<T>>(),
50 std::declval<EdgeFunction<typename T::l_t>>())),
51 decltype(T::join(std::declval<EdgeFunctionRef<T>>(),
52 std::declval<EdgeFunction<typename T::l_t>>()))>>
53 : std::true_type {};
54
55} // namespace detail
56template <typename T>
57static constexpr bool IsEdgeFunction = detail::IsEdgeFunction<T>::value;
58
59#else
60// clang-format off
61template <typename T>
62concept IsEdgeFunction = requires(const T &EF, const EdgeFunction<typename T::l_t>& TEEF, EdgeFunctionRef<T> CEF, typename T::l_t Src) {
63 typename T::l_t;
64 {EF.computeTarget(Src)} -> std::convertible_to<typename T::l_t>;
65 {T::compose(CEF, TEEF)} -> std::same_as<EdgeFunction<typename T::l_t>>;
66 {T::join(CEF, TEEF)} -> std::same_as<EdgeFunction<typename T::l_t>>;
67};

72enum class EdgeFunctionAllocationPolicy {
73 SmallObjectOptimized,
74 DefaultHeapAllocated,
75 CustomHeapAllocated,
76};

78class EdgeFunctionBase {
79public:
80 template <typename ConcreteEF>
81 static constexpr bool IsSOOCandidate =
82 sizeof(ConcreteEF) <= sizeof(void *) && // NOLINT
83 alignof(ConcreteEF) <= alignof(void *) &&
84 std::is_trivially_copyable_v<ConcreteEF>;
85
86 using AllocationPolicy = EdgeFunctionAllocationPolicy;
87
88protected:
89 struct RefCountedBase {
90 mutable std::atomic_size_t Rc = 0;
91 };
92 template <typename T> struct RefCounted : RefCountedBase {
93 T Value;
94 };
95
96 template <typename T> struct CachedRefCounted : RefCounted<T> {
97 EdgeFunctionSingletonCache<T> *Cache{};
98 };
99
100 template <typename ConcreteEF>
101 constexpr static inline const ConcreteEF *
102 getPtr(const void *const &EF) noexcept {
103 if constexpr (!IsSOOCandidate<ConcreteEF>) {
104 return &static_cast<const RefCounted<ConcreteEF> *>(EF)->Value;
105 } else {
106 return static_cast<const ConcreteEF *>(static_cast<const void *>(&EF));
107 }
108 }
109 template <typename ConcreteEF>
110 constexpr static inline const ConcreteEF *
111 getPtr(const void *const &&EF) = delete; // NOLINT
112
113 template <typename ConcreteEF>
114 static constexpr AllocationPolicy DefaultAllocPolicy =
115 IsSOOCandidate<ConcreteEF> ? AllocationPolicy::SmallObjectOptimized
116 : AllocationPolicy::DefaultHeapAllocated;
117 template <typename ConcreteEF>
118 static constexpr AllocationPolicy CustomAllocPolicy =
119 IsSOOCandidate<ConcreteEF> ? AllocationPolicy::SmallObjectOptimized
120 : AllocationPolicy::CustomHeapAllocated;
121};

123/// \brief Non-null reference to an edge function that is guarenteed to be
124/// managed by an EdgeFunction object.
125template <typename EF>
126class [[clang::trivial_abi]] EdgeFunctionRef final : EdgeFunctionBase {
127 template <typename L> friend class EdgeFunction;
128
129public:
130 using l_t = typename EF::l_t;
131
132 EdgeFunctionRef(const EdgeFunctionRef &) noexcept = default;
133 EdgeFunctionRef &operator=(const EdgeFunctionRef &) noexcept = default;
134 ~EdgeFunctionRef() = default;
135
136 const EF *operator->() const noexcept { return getPtr<EF>(Instance); }
137 const EF *get() const noexcept { return getPtr<EF>(Instance); }
138 const EF &operator*() const noexcept { return *getPtr<EF>(Instance); }
139
140 [[nodiscard]] bool isCached() const noexcept {
141 if constexpr (IsSOOCandidate<EF>) {
142 return false;
143 } else {
144 return IsCached;
145 }
146 }
147
148 [[nodiscard]] EdgeFunctionSingletonCache<EF> *
149 getCacheOrNull() const noexcept {
150 if (isCached()) {
151 return static_cast<const CachedRefCounted<EF> *>(Instance)->Cache;
152 }
153
154 return nullptr;
155 }
156
157private:
158 explicit EdgeFunctionRef(const void *Instance, bool IsCached) noexcept
159 : Instance(Instance) {
160 if constexpr (!IsSOOCandidate<EF>) {
161 this->IsCached = IsCached;
162 }
163 }
164 const void *Instance{};
165 [[no_unique_address]] std::conditional_t<IsSOOCandidate<EF>, EmptyType, bool>
166 IsCached{};
167};

178 // --- Constructors

180 /// Default-initializes the edge-function with nullptr
181 EdgeFunction() noexcept = default;
182 /// Default-initializes the edge-function with nullptr
183 EdgeFunction(std::nullptr_t) noexcept : EdgeFunction() {}
184 /// Copy constructor. Increments the ref-count, if not small-object-optimized.
185 EdgeFunction(const EdgeFunction &Other) noexcept
186 : EdgeFunction(Other.EF, Other.VTAndHeapAlloc) {}
187 /// Move constructor. Does not increment the ref-count, but instead leaves the
188 /// moved-from edge function in the nullptr state.
189 EdgeFunction(EdgeFunction &&Other) noexcept
190 : EF(std::exchange(Other.EF, nullptr)),
191 VTAndHeapAlloc(
192 std::exchange(Other.VTAndHeapAlloc, decltype(VTAndHeapAlloc){})) {}
193
194 /// Standard swap; does not affect ref-counts
195 void swap(EdgeFunction &Other) noexcept {
196 std::swap(EF, Other.EF);
197 std::swap(VTAndHeapAlloc, Other.VTAndHeapAlloc);
198 }
199 /// Standard swap; does not affect ref-counts
200 friend void swap(EdgeFunction &LHS, EdgeFunction &RHS) noexcept {
201 LHS.swap(RHS);
202 }
203
204 /// Combined copy- and move assignment. If the assigned-to edge function is
205 /// not null, invokes the destructor on it, before overwriting its content.
206 EdgeFunction &operator=(EdgeFunction Other) noexcept {
207 std::destroy_at(this);
208 return *new (this) EdgeFunction(std::move(Other)); // NOLINT
209 }
210 /// Null-assignment operator. Decrements the ref-count if not
211 /// small-object-optimized or already null. Leaves the assigned-to edge
212 /// function in the nullptr state.
213 EdgeFunction &operator=(std::nullptr_t) noexcept {
214 std::destroy_at(this);
215 return *new (this) EdgeFunction(); // NOLINT
216 }
217
218 /// Destructor. Decrements the ref-count if not small-object-optimized and
219 /// destroyes the held edge function once the ref-count reaches 0.
220 ~EdgeFunction() noexcept {
221 AllocationPolicy Policy = VTAndHeapAlloc.getInt();
222 if (Policy != AllocationPolicy::SmallObjectOptimized) {
223 assert(VTAndHeapAlloc.getPointer() != nullptr && "Heap-alloc'd nullptr?");
224 // Note: Memory-order taken from llvm::ThreadSafeRefCountedBase
225 if (static_cast<const RefCountedBase *>(EF)->Rc.fetch_sub(
226 1, std::memory_order_acq_rel) == 1) {
227 VTAndHeapAlloc.getPointer()->destroy(EF, Policy);
228 }
229 }
230 }

232 /// Implicit-conversion constructor from EdgeFunctionRef. Increments the
233 /// ref-count if not small-object optimized
234 template <typename ConcreteEF, typename = std::enable_if_t<!std::is_same_v<
235 EdgeFunction, std::decay_t<ConcreteEF>>>>
236 EdgeFunction(EdgeFunctionRef<ConcreteEF> CEF) noexcept
237 : EdgeFunction(CEF.Instance,
238 {&VTableFor<ConcreteEF>, [CEF] {
239 if constexpr (IsSOOCandidate<ConcreteEF>) {
240 (void)CEF;
241 return AllocationPolicy::SmallObjectOptimized;
242 } else {
243 return CEF.IsCached
244 ? AllocationPolicy::CustomHeapAllocated
245 : AllocationPolicy::DefaultHeapAllocated;
246 }
247 }()}) {}

249 /// Conversion-constructor from any edge function (that satisfies the
250 /// IsEdgeFunction trait). Stores a type-erased copy of CEF and allocates
251 /// space for it on the heap if small-object-optimization cannot be applied.
252 template <typename ConcreteEF,
253 typename = std::enable_if_t<
254 !std::is_same_v<EdgeFunction, std::decay_t<ConcreteEF>> &&
255 IsEdgeFunction<ConcreteEF>>>
256 EdgeFunction(ConcreteEF &&CEF) noexcept(
257 IsSOOCandidate<std::decay_t<ConcreteEF>>)
258 : EdgeFunction(std::in_place_type<std::decay_t<ConcreteEF>>,
259 std::forward<ConcreteEF>(CEF)) {}
260
261 /// Emplacement-constructor for any edge function. Constructs a new object of
262 /// type ConcreteEF with the given constructor arguments and allocates space
263 /// for it on the heap if small-object-optimization cannot be applied.
264 /// No extra copy- or move construction/assignment is performed. Use this ctor
265 /// if even moving is expensive.
266 template <typename ConcreteEF, typename... ArgTys>
267 explicit EdgeFunction(
268 std::in_place_type_t<ConcreteEF> /*unused*/,
269 ArgTys &&...Args) noexcept(IsSOOCandidate<std::decay_t<ConcreteEF>> &&
270 std::is_nothrow_constructible_v<ConcreteEF,
271 ArgTys...>)
272 : EdgeFunction(
273 [](auto &&...Args) {
274 if constexpr (IsSOOCandidate<std::decay_t<ConcreteEF>>) {
275 void *Ret = nullptr;
276 new (&Ret) ConcreteEF(std::forward<ArgTys>(Args)...);
277 return Ret;
278 } else {
279 return new RefCounted<ConcreteEF>{
280 {}, {std::forward<ArgTys>(Args)...}};
281 }
282 }(std::forward<ArgTys>(Args)...),
283 {&VTableFor<ConcreteEF>, DefaultAllocPolicy<ConcreteEF>}) {
284 static_assert(std::is_same_v<l_t, typename ConcreteEF::l_t>,
285 "Cannot construct EdgeFunction with incompatible "
286 "lattice domain");
287 }
288
289 /// Conversion-constructor for any edge function with enabled caching. Stores
290 /// a type-erased copy of EF.EF and allocates space for it on the heap if
291 /// small-object-optimization cannot be applied. If a heap allocation is
292 /// requires, first consults the EF.Cache to check whether an equivalent edge
293 /// function is already allocated. If so, takes the one from the cache and
294 /// increases its ref-count. Otherwise, performs a fresh allocation and
295 /// inserts the edge function into EF.Cache.
296 /// When all references to this edge function are out-of-scope, the destructor
297 /// automatically removes the edge function from EF.Cache. Hence, make sure
298 /// that EF.Cache lives at least as long as the last edge function cached in
299 /// it.
300 template <typename ConcreteEF, typename = std::enable_if_t<
301 IsEdgeFunction<ConcreteEF> &&
302 std::is_move_constructible_v<ConcreteEF>>>
303 EdgeFunction(CachedEdgeFunction<ConcreteEF> EF)
304 : EdgeFunction(
305 [&EF] {
306 assert(EF.Cache != nullptr);
307 if constexpr (IsSOOCandidate<std::decay_t<ConcreteEF>>) {
308 void *Ret;
309 new (&Ret) ConcreteEF(std::move(EF.EF));
310 return Ret;
311 } else {
312 if (const auto *Mem = EF.Cache->lookup(EF.EF)) {
313 return static_cast<const RefCounted<ConcreteEF> *>(Mem);
314 }
315
316 auto Ret = new CachedRefCounted<ConcreteEF>{
317 {{}, {std::move(EF.EF)}}, EF.Cache};
318 EF.Cache->insert(&Ret->Value, Ret);
319 return static_cast<const RefCounted<ConcreteEF> *>(Ret);
320 }
321 }(),
322 {&VTableFor<ConcreteEF>, CustomAllocPolicy<ConcreteEF>}) {}

358 ///
359 /// This function composes the two edge functions this and SecondEF. This
360 /// function is used to extend an edge function in order to construct
361 /// so-called jump functions that describe the effects of everlonger sequences
362 /// of code.
363 ///
364 /// For semantic correctness, please make sure that for all inputs x in l_t,

366 /// SecondEF.computeTarget(FirstEF.computeTarget(x)).
367 [[nodiscard]] static EdgeFunction compose(const EdgeFunction &FirstEF,

375 ///
376 /// This function describes the join of the two edge functions this and
377 /// OtherEF. The function is called whenever two edge functions need to
378 /// be joined, for instance, when two branches lead to a common successor
379 /// instruction.
380 ///
381 /// Calls the static function EF::join(*this, OtherEF) for your concrete
382 /// edge function EF.
383 ///
384 /// For semantic correctness, please make sure that for all inputs x in l_t,
385 /// it holds (with join = JoinLatticeTraits<l_t>::join):
386 /// join(this->joinWith(OtherEF).computeTarget(x)),
387 /// join(this->computeTarget(x), OtherEF.computeTarget(x))) ==
388 /// this->joinWith(OtherEF).computeTarget(x)
389 /// i.e. that joining on edge functions only goes up the lattice that is
390 /// connected with the value-lattice on l_t
391 [[nodiscard]] EdgeFunction joinWith(const EdgeFunction &OtherEF) const {
392 return join(*this, OtherEF);
393 }

395 ///
396 /// This function describes the join of the two edge functions this and
397 /// OtherEF. The function is called whenever two edge functions need to
398 /// be joined, for instance, when two branches lead to a common successor
399 /// instruction.
400 ///
401 /// For semantic correctness, please make sure that for all inputs x in l_t,
402 /// it holds (with joinl = JoinLatticeTraits<l_t>::join):
403 /// joinl(join(FirstEF, SecondEF).computeTarget(x)),
404 /// joinl(FirstEF.computeTarget(x), SecondEF.computeTarget(x))) ==
405 /// this->joinWith(OtherEF).computeTarget(x)
406 /// i.e. that joining on edge functions only goes up the lattice that is
407 /// connected with the value-lattice on l_t
408 [[nodiscard]] static EdgeFunction join(const EdgeFunction &FirstEF,
409 const EdgeFunction &SecondEF) {
410 assert(FirstEF.isValid() && "join() called on LHS nullptr!");
411 assert(SecondEF.isValid() && "join() called on RHS nullptr!");
412 return FirstEF.VTAndHeapAlloc.getPointer()->join(
413 FirstEF.EF, SecondEF, FirstEF.VTAndHeapAlloc.getInt());
414 }

422 /// Checks for equality of two edge functions. Equality requires exact
423 /// type-equality and value-equality based on operator== of the concrete edge
424 /// functions that are compared.
425 ///
426 /// If the concrete edge function has no nonstatic data members, the equality
427 /// is defaulted to only compare the types. However, by explicitly defining
428 /// operator==, this behavior can be overridden.
429 [[nodiscard]] friend bool operator==(const EdgeFunction &LHS,
430 const EdgeFunction &RHS) noexcept {
431 if (LHS.VTAndHeapAlloc.getPointer() != RHS.VTAndHeapAlloc.getPointer()) {
432 return false;
433 }
434 if (LHS.VTAndHeapAlloc.getOpaqueValue() == nullptr) {
435 return true;
436 }
437 return LHS.EF == RHS.EF ||
438 LHS.VTAndHeapAlloc.getPointer()->equals(LHS.EF, RHS.EF);
439 }

636 [[nodiscard]] bool isCached() const noexcept {

648 /// Gets the cache where this edge function is being cached in. If this edge
649 /// function is not cached (i.e., isCached() returns false), returns nullptr.
650 /// Assumes that the held edge function is *exactly* of type ConcreteEF. In
651 /// debug builds, asserts isa<ConcreteEF>.
652 ///
653 /// Use with caution!
654 template <typename ConcreteEF>
655 [[nodiscard]] EdgeFunctionSingletonCache<ConcreteEF> *
656 getCacheOrNull() const noexcept {
657 assert(isa<ConcreteEF>());

659 VTAndHeapAlloc.getInt() == AllocationPolicy::DefaultHeapAllocated) {
660 return nullptr;
661 }
662 return static_cast<const CachedRefCounted<ConcreteEF> *>(EF)->Cache;
663 }

666 if (!VTAndHeapAlloc.getOpaqueValue()) {
667 return 0;
668 }
669
670 return VTAndHeapAlloc.getPointer()->getHashCode(
671 EF, VTAndHeapAlloc.getPointer());
672 }

675 assert(isValid() && "depth() called on nullptr!");
676 return VTAndHeapAlloc.getPointer()->depth(EF);

711 template <typename ConcreteEF>
712 static constexpr VTable VTableFor = {
713 [](const void *EF, ByConstRef<l_t> Source) {
714 return getPtr<ConcreteEF>(EF)->computeTarget(Source);
715 },
716 [](const void *EF, const EdgeFunction &SecondEF,
717 AllocationPolicy Policy) {
718 return ConcreteEF::compose(
719 EdgeFunctionRef<ConcreteEF>(
720 EF, Policy == AllocationPolicy::CustomHeapAllocated),
721 SecondEF);
722 },
723 [](const void *EF, const EdgeFunction &OtherEF, AllocationPolicy Policy) {
724 return ConcreteEF::join(
725 EdgeFunctionRef<ConcreteEF>(
726 EF, Policy == AllocationPolicy::CustomHeapAllocated),
727 OtherEF);
728 },
729 [](const void *EF1, const void *EF2) noexcept {
730 static_assert(IsEqualityComparable<ConcreteEF> ||
731 std::is_empty_v<ConcreteEF>,
732 "An EdgeFunction must be equality comparable with "
733 "operator==. Only if the type is empty, i.e. has no "
734 "members, the comparison can be inferred.");
735 if constexpr (IsEqualityComparable<ConcreteEF>) {
736 return *getPtr<ConcreteEF>(EF1) == *getPtr<ConcreteEF>(EF2);
737 } else {
738 return true;
739 }
740 },
741 [](const void *EF, llvm::raw_ostream &OS) {
742 if constexpr (is_llvm_printable_v<ConcreteEF>) {
743 OS << *getPtr<ConcreteEF>(EF);
744 } else {
745 OS << llvm::getTypeName<ConcreteEF>();
746 }
747 },
748 [](const void *EF) noexcept {
749 if constexpr (HasIsConstant<ConcreteEF>) {
750 static_assert(
751 std::is_nothrow_invocable_v<decltype(&ConcreteEF::isConstant),
752 const ConcreteEF &>,
753 "The function isConstant() must be noexcept!");
754 return getPtr<ConcreteEF>(EF)->isConstant();
755 } else {
756 return false;
757 }
758 },
759 [](const void *EF, AllocationPolicy Policy) noexcept {
760 if constexpr (!IsSOOCandidate<ConcreteEF>) {
761 if (Policy != AllocationPolicy::CustomHeapAllocated) {
762 assert(Policy == AllocationPolicy::DefaultHeapAllocated);
763 delete static_cast<const RefCounted<ConcreteEF> *>(EF);
764 } else {
765 auto CEF = static_cast<const CachedRefCounted<ConcreteEF> *>(EF);
766 CEF->Cache->erase(CEF->Value);
767 delete CEF;
768 }
769 }
770 },
771 [](const void *EF, const void *VT) noexcept -> size_t {
772 if constexpr (is_std_hashable_v<ConcreteEF>) {
773 return std::hash<ConcreteEF>{}(*getPtr<ConcreteEF>(EF));
774 } else if constexpr (is_llvm_hashable_v<ConcreteEF>) {
775 using llvm::hash_value;
776 return hash_value(*getPtr<ConcreteEF>(EF));
777 } else {
778 return llvm::hash_combine(EF, VT);
779 }
780 },
781 [](const void *EF) noexcept -> size_t {
782 if constexpr (HasDepth<ConcreteEF>) {
783 return getPtr<ConcreteEF>(EF)->depth();
784 } else {
785 return 1;
786 }
787 },
788 };
