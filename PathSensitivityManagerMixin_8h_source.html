70 /// Reconstruct the combined control- and data-flow paths the lead to any of
71 /// the given data-flow facts in FactsRange holding right after Inst.
72 ///
73 /// The result is given as graph, where cycles are unrolled once in an
74 /// implementation-defined way.
75 template <
76 typename FactsRangeTy, typename ConfigTy,
77 typename Filter = DefaultPathTracingFilter,
78 typename = std::enable_if_t<is_pathtracingfilter_for_v<Filter, NodeRef>>>
79 [[nodiscard]] GraphType
80 pathsDagToAll(n_t Inst, FactsRangeTy FactsRange,
81 const PathSensitivityConfigBase<ConfigTy> &Config,
82 const Filter &PFilter = {}) const {
83 graph_type Dag;
84 PathsToContext Ctx;
85
86 for (const d_t &Fact : FactsRange) {
87 auto Nod = ESG.getNodeOrNull(Inst, std::move(Fact));
88
89 if (LLVM_UNLIKELY(!Nod)) {
90 llvm::errs() << "Invalid Instruction-FlowFact pair. Only use those "
91 "pairs that are part of the IDE analysis results!\n";
92 llvm::errs() << "Fatal error occurred. Printing ESG ...\n";
93
94 ESG.printAsDot(llvm::errs());
95 llvm::errs().flush();
96
97 abort();
98 }
99
100 /// NOTE: We don't need to check that Nod has not been processed yet,
101 /// because in the ESG construction we only merge nodes with the same flow
102 /// fact. Here, the flow fact for each node differs (assuming FactsRage
103 /// does not contain duplicates)
104
105 auto Rt = pathsToImpl(Inst, Nod, Dag, Ctx, PFilter);
106 if (Rt != GraphTraits<GraphType>::Invalid) {
107 graph_traits_t::addRoot(Dag, Rt);
108 }
109 }
110
111#ifndef NDEBUG
112 if (!static_cast<const Derived *>(this)->assertIsDAG(Dag)) {
113 llvm::report_fatal_error("Invariant violated: DAG has a circle in it!");
114 } else {
115 PHASAR_LOG_LEVEL_CAT(DEBUG, "PathSensitivityManager",
116 "The DAG indeed has no circles");
117 }
118
119#endif
120
121 if (Config.DAGDepthThreshold != SIZE_MAX) {
122 Dag = Derived::reverseDAG(std::move(Dag), Config.DAGDepthThreshold);
123 } else {
124 Dag = reverseGraph(std::move(Dag));
125 }
126
127 if (Config.MinimizeDAG) {
128
129 auto Equiv = minimizeGraph(Dag);
130
131 Dag = createEquivalentGraphFrom(std::move(Dag), Equiv);
132
133#ifndef NDEBUG
134 if (!static_cast<const Derived *>(this)->assertIsDAG(Dag)) {
135 llvm::report_fatal_error("Invariant violated: DAG has a circle in it!");
136 } else {
137 PHASAR_LOG_LEVEL_CAT(DEBUG, "PathSensitivityManager",
138 "The DAG indeed has no circles");
139 }
140#endif
141 }
142
143 return Dag;
144 }

146 /// Reconstruct the combined control- and data-flow paths the lead to any of
147 /// the given data-flow facts holding right after Inst.

149 /// The result is given as graph, where cycles are unrolled once in an

152 typename ConfigTy, typename L, typename Filter = DefaultPathTracingFilter,

158 auto Res = SR.resultsAt(Inst);

180 /// Reconstruct the combined control- and data-flow paths the lead to any of
181 /// the given data-flow facts in FactsRange holding at Inst.
182 ///
183 /// The result is given as graph, where cycles are unrolled once in an
184 /// implementation-defined way.
185 template <
186 typename ConfigTy, typename Filter = DefaultPathTracingFilter,
187 typename = std::enable_if_t<is_pathtracingfilter_for_v<Filter, NodeRef>>>
188 [[nodiscard]] GraphType
189 pathsDagToInLLVMSSA(n_t Inst, d_t Fact,
190 const PathSensitivityConfigBase<ConfigTy> &Config,
191 const Filter &PFilter = {}) const {
192 // Temporary code to bridge the time until merging f-IDESolverStrategy
193 // into development
194 if (Inst->getType()->isVoidTy()) {
195 return pathsDagToAll(Inst, llvm::ArrayRef(&Fact, 1), Config, PFilter);
196 }
197
198 if (auto Next = Inst->getNextNonDebugInstruction()) {
199 return pathsDagToAll(Next, llvm::ArrayRef(&Fact, 1), Config, PFilter);
200 }
201
202 PHASAR_LOG_LEVEL(WARNING, "[pathsDagToInLLVMSSA]: Cannot precisely "
203 "determine the ESG node for inst-flowfact-pair ("
204 << NToString(Inst) << ", " << DToString(Fact)
205 << "). Fall-back to an approximation");
206
207 for (const auto *BB : llvm::successors(Inst)) {
208 const auto *First = &BB->front();
209 if (llvm::isa<llvm::DbgInfoIntrinsic>(First)) {
210 First = First->getNextNonDebugInstruction();
211 }
212 if (ESG.getNodeOrNull(First, Fact)) {
213 return pathsDagToAll(First, llvm::ArrayRef(&Fact, 1), Config, PFilter);
214 }
215 }
216
217 llvm::report_fatal_error("Could not determine any ESG node corresponding "
218 "to the inst-flowfact-pair (" +
219 llvm::Twine(NToString(Inst)) + ", " +
220 DToString(Fact) + ")");
221
222 return {};
223 }

226 template <typename Filter>
227 bool pathsToImplLAInvoke(vertex_t Ret, NodeRef Vtx, PathsToContext &Ctx,
228 graph_type &RetDag, const Filter &PFilter) const {
229 NodeRef Prev;
230 bool IsEnd = false;
231 bool IsError = false;
232
233 do {
234 Prev = Vtx;
235 graph_traits_t::node(RetDag, Ret).push_back(Vtx.source());
236
237 Vtx = Vtx.predecessor();
238
239 if (PFilter.HasReachedEnd(Prev, Vtx)) {
240 IsEnd = true;
241 } else if (PFilter.IsErrorneousTransition(Prev, Vtx)) {
242 IsError = true;
243 }
244
245 if (!Vtx) {
246 return !IsError;
247 }
248
249 if (IsEnd || IsError) {
250 break;
251 }
252
253 } while (!Vtx.hasNeighbors());
254
255 if (!Ctx.CurrPath.insert(Ret)) {
256 PHASAR_LOG_LEVEL(ERROR, "Node " << Ret << " already on path");
257 return false;
258 }
259 scope_exit PopRet = [&Ctx] { Ctx.CurrPath.pop_back(); };
260
261 // NOLINTNEXTLINE(readability-identifier-naming)
262 auto traverseNext = [&Ctx, this, Ret, &RetDag, &PFilter](NodeRef Nxt) {
263 auto Succ = pathsToImplLA(Nxt, Ctx, RetDag, PFilter);
264 if (Succ != graph_traits_t::Invalid && !Ctx.CurrPath.contains(Succ)) {
265 graph_traits_t::addEdge(RetDag, Ret, Succ);
266 }
267 };
268
269 if (!IsEnd && !IsError) {
270 traverseNext(Vtx);
271 }
272
273 for (auto Nxt : Vtx.neighbors()) {
274 assert(Nxt != nullptr);
275 if (PFilter.IsErrorneousTransition(Prev, Nxt)) {
276 continue;
277 }
278 if (PFilter.HasReachedEnd(Prev, Nxt)) {
279 IsEnd = true;
280 continue;
281 }
282 traverseNext(Nxt);
283 }
284
285 graph_traits_t::dedupOutEdges(RetDag, Ret);
286
287 return IsEnd || graph_traits_t::outDegree(RetDag, Ret) != 0;
288 }

290 template <typename Filter>
291 vertex_t pathsToImplLA(NodeRef Vtx, PathsToContext &Ctx, graph_type &RetDag,
292 const Filter &PFilter) const {
293 /// Idea: Treat the graph as firstChild-nextSibling notation and always
294 /// traverse with one predecessor lookAhead
295
296 auto [It, Inserted] =
297 Ctx.Cache.try_emplace(Vtx.id(), graph_traits_t::Invalid);
298 if (!Inserted) {
299 return It->second;
300 }
301
302 auto Ret =
303 graph_traits_t::addNode(RetDag, typename graph_traits_t::value_type());
304 // auto Ret = RetDag.addNode();
305 It->second = Ret;
306
307 if (!pathsToImplLAInvoke(Ret, Vtx, Ctx, RetDag, PFilter)) {
308 /// NOTE: Don't erase Vtx from Cache to guarantee termination
309 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic) -- fp
310 Ctx.Cache[Vtx.id()] = graph_traits_t::Invalid;
311
312 if (Ctx.CurrPath.contains(Ret) || !graph_traits_t::pop(RetDag, Ret)) {
313 PHASAR_LOG_LEVEL(WARNING, "Cannot remove invalid path at: " << Ret);
314 graph_traits_t::node(RetDag, Ret).clear();
315 }
316
317 return graph_traits_t::Invalid;
318 }
319 return Ret;
320 }

322 template <typename Filter>
323 vertex_t pathsToImpl(n_t QueryInst, NodeRef Vtx, graph_type &RetDag,
324 PathsToContext &Ctx, const Filter &PFilter) const {
325
326 auto Ret =
327 graph_traits_t::addNode(RetDag, typename graph_traits_t::value_type());
328 graph_traits_t::node(RetDag, Ret).push_back(QueryInst);
329
330 for (auto NB : Vtx.neighbors()) {
331 auto NBNode = pathsToImplLA(NB, Ctx, RetDag, PFilter);
332 if (NBNode != graph_traits_t::Invalid) {
333 graph_traits_t::addEdge(RetDag, Ret, NBNode);
334 }
335 }
336 auto VtxNode = pathsToImplLA(Vtx, Ctx, RetDag, PFilter);
337 if (VtxNode != graph_traits_t::Invalid) {
338 graph_traits_t::addEdge(RetDag, Ret, VtxNode);
339 }
340
341 graph_traits_t::dedupOutEdges(RetDag, Ret);
342
343 return Ret;
344 }
