29 * @brief Holds the results of a query to the Hexastore.
30 */
31struct HSResult {
32 /// Used for the source node.
33 std::string Subject;
34 /// Used for the edge.
35 std::string Predicate;
36 /// Used for the destination node.
37 std::string Object;
38 HSResult() = default;
39 HSResult(const std::string Subject, // NOLINT
40 std::string Predicate, // NOLINT
41 std::string Object) // NOLINT
42 : Subject(Subject), Predicate(std::move(Predicate)),
43 Object(std::move(Object)) {}
44 /// Prints an entry of the results to the command-line
45 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
46 const HSResult &Result) {
47 return OS << "[ subject: " << Result.Subject
48 << " | predicate: " << Result.Predicate
49 << " | object: " << Result.Object << " ]";
50 }
51
52 friend bool operator==(const HSResult &LHS, const HSResult &RHS) {
53 return LHS.Subject == RHS.Subject && LHS.Predicate == RHS.Predicate &&
54 LHS.Object == RHS.Object;
55 }
56};

58/**
59 * A Hexastore is an efficient approach to store large graphs.
60 * This approach is based on the paper "Database-Backed Program Analysis
61 * for Scalable Error Propagation" by Weiss, Rubio-González and Libit.
62 *
63 * To store a graph into a database, we represent the graph by a set of
64 * string 3-tuples by the form:
65 *
66 * (source node, edge, destination node)
67 *
68 * A Hexastore indexes and saves graphs six-fold, according to each of
69 * the permutations of source, edge and destination label. For example,
70 * this allows to quickly find all edges for a given source node with one
71 * look-up. In general, given one or two fixed elements of a (source, edge,
72 * destination) tuple, Hexastore can quickly access the related information.
73 *
74 * @brief Efficient data structure for holding graphs in databases.
75 */
76class [[deprecated("This ancient API is not maintained for long and should not "
77 "be used anymore!")]] Hexastore {
78private:
79 sqlite3 *HSInternalDB{};
80 static int callback(void * /*NotUsed*/, int Argc, char **Argv,
81 char **AzColName);
82 void doPut(const std::string &Query, std::array<std::string, 3> Edge);
83
84public:
85 /**
86 * If the given filename matches an already created Hexastore, no
87 * new Hexastore will be created. Instead the already created Hexastore
88 * will be used.
89 *
90 * @brief Constructs a Hexastore under the given filename.
91 * @param filename Filename of the Hexastore.
92 */
93 Hexastore(const std::string &FileName);
94 Hexastore(const Hexastore &) = delete;
95 Hexastore &operator=(const Hexastore &) = delete;
96
97 /**
98 * Destructor.
99 */
100 ~Hexastore();
101
102 /**
103 * Adds the given tuple as a new entry to the Hexastore. It is not
104 * possible to have duplicate entries in the Hexastore and
105 * duplicate put queries will be silently ignored by the Hexastore.
106 *
107 * @brief Creates a new entry in the Hexastore.
108 * @note To mitigate certain compiler warnings, it is advised to use
109 * double braces, e.g.:
110 * hexastore.put({{"subject", "predicate", "object"}});
111 * @param edge New entry in the form of a 3-tuple.
112 */
113 void put(const std::array<std::string, 3> &Edge);
114
115 /**
116 * A query is always in the form of a 3-tuple (source, edge, destination)
117 * where
118 * none/one/two or all three elements of the tuple are given, while the not
119 * fixed
120 * elements are represented by a "?". E.g. to query all destination nodes
121 * for a certain source node and edge, the following query is used:
122 *
123 * ("node_19", "edge_3", "?")
124 *
125 * @brief Query information from the Hexastore.
126 * @param edge_query Query in the form of a 3-tuple.
127 * @param result_size_hint Used for possible optimization.
128 * @return An object of hs_result, holding the queried information.
129 */
130 std::vector<HSResult> get(std::array<std::string, 3> EdgeQuery,
131 size_t ResultSizeHint = 0);
132};
