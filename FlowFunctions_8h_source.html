39///
40/// This class models a flow function for distributive data-flow problems.
41///
42template <typename D, typename Container = std::set<D>> class FlowFunction {
43 static_assert(std::is_same<typename Container::value_type, D>::value,
44 "Container values needs to be the same as D");
45
46public:
47 using FlowFunctionType = FlowFunction<D, Container>;
48 using FlowFunctionPtrType = std::shared_ptr<FlowFunctionType>;
49
50 using container_type = Container;
51 using value_type = typename container_type::value_type;
52
53 virtual ~FlowFunction() = default;
54
55 ///
56 /// This function is called for each data-flow fact Source that holds before
57 /// the instruction under analysis. The return value is a (potentially empty)
58 /// set of data-flow facts that are generated from Source and hold after the
59 /// instruction under analysis. In other words: the function describes what
60 /// exploded supergraph edges have to be "drawn".
61 ///
62 /// Please also refer to the various flow function factories of the
63 /// FlowFunctions interface: FlowFunctions::get*FlowFunction() for more
64 /// details.
65 ///
66 virtual container_type computeTargets(D Source) = 0;
67};
68
69/// Helper template to check at compile-time whether a type implements the
70/// FlowFunction interface, no matter which data-flow fact type it uses.
71///
72/// Use is_flowfunction_v instead.
73template <typename FF> struct IsFlowFunction {
74 template <typename D, typename Container>
75 static std::true_type test(const FlowFunction<D, Container> &);
76 static std::false_type test(...);
77
78 static constexpr bool value = // NOLINT
79 std::is_same_v<std::true_type,
80 decltype(test(std::declval<const FF &>()))>;
81};
82
83/// Helper template to check at compile-time whether a type implements the
84/// FlowFunction interface, no matter which data-flow fact type it uses.
85template <typename FF>
86concept is_flowfunction_v = IsFlowFunction<FF>::value; // NOLINT

88/// Given a flow-function type FF, returns a (smart) pointer type pointing to FF
89template <is_flowfunction_v FF>
90using FlowFunctionPtrTypeOf = std::shared_ptr<FF>;

92/// Given a dataflow-fact type and optionally a container-type, returns a
93/// (smart) pointer type pointing to a FlowFunction with the specified
94/// flow-fact- and container type.
95///
96/// Equivalent to FlowFunctionPtrTypeOf<FlowFunction<D, Container>>
97template <typename D, typename Container = std::set<D>>
98using FlowFunctionPtrType = FlowFunctionPtrTypeOf<FlowFunction<D, Container>>;

100/// Wrapper flow function that is automatically used by the IDESolver if the
101/// autoAddZero configuration option is set to true (default).
102/// Ensures that the tautological zero-flow fact (Λ) does not get killed.
103template <typename D, typename Container = std::set<D>>
104class ZeroedFlowFunction : public FlowFunction<D, Container> {
105 using typename FlowFunction<D, Container>::container_type;
106 using typename FlowFunction<D, Container>::FlowFunctionPtrType;
107
108public:
109 ZeroedFlowFunction(FlowFunctionPtrType FF, D ZV)
110 : Delegate(std::move(FF)), ZeroValue(std::move(ZV)) {}
111 container_type computeTargets(D Source) override {
112 if (Source == ZeroValue) {
113 container_type Result = Delegate->computeTargets(Source);
114 Result.insert(ZeroValue);
115 return Result;
116 }
117 return Delegate->computeTargets(std::move(Source));
118 }
119
120private:
121 FlowFunctionPtrType Delegate;
122 D ZeroValue;
123};

126template <typename Container, typename Range>
127Container makeContainer(Range &&Rng) {
128 if constexpr (std::is_convertible_v<std::decay_t<Range>, Container>) {
129 return std::forward<Range>(Rng);
130 } else {
131 Container C;
132 reserveIfPossible(C, Rng.size());
133 for (auto &&Fact : Rng) {
134 C.insert(PSR_FWD(Fact));
135 }
136 return C;
137 }

150 /// A flow function that propagates all incoming facts unchanged.
151 ///
152 /// Given a flow-function f = identityFlow(), then for all incoming
153 /// dataflow-facts x, f(x) = {x}.
154 ///
155 /// In the exploded supergraph it may look as follows:
156 /// \code
157 /// x1 x1 x3 ...
158 /// | | | ...
159 /// id-instruction | | | ...
160 /// v v v ...
161 /// x1 x2 x3 ...
162 /// \endcode
163 static auto identityFlow() {
164 struct IdFF final : public FlowFunction<d_t, container_type> {
165 container_type computeTargets(d_t Source) override {
166 return {std::move(Source)};
167 }
168 };
169 static auto TheIdentity = std::make_shared<IdFF>();
170
171 return TheIdentity;
172 }
173
174 /// The most generic flow function. Invokes the passed function object F to
175 /// retrieve the desired data-flows.
176 ///
177 /// So, given a flow function f = lambdaFlow(F), then for all incoming
178 /// dataflow-facts x, f(x) = F(x).
179 ///
180 /// In the exploded supergraph it may look as follows:
181 /// \code
182 /// x
183 /// |
184 /// inst F
185 /// / / | \ \ ...
186 /// v v v v v
187 /// x1 x2 x x3 x4
188 /// \endcode
189 template <typename Fn> static auto lambdaFlow(Fn &&F) {
190 struct LambdaFlow final : public FlowFunction<d_t, container_type> {
191 LambdaFlow(Fn &&F) : Flow(std::forward<Fn>(F)) {}
192 container_type computeTargets(d_t Source) override {
193 return std::invoke(Flow, std::move(Source));
194 }
195
196 [[no_unique_address]] std::decay_t<Fn> Flow;
197 };
198
199 return std::make_shared<LambdaFlow>(std::forward<Fn>(F));
200 }

205 /// A flow function that generates a new dataflow fact (FactToGenerate) if
206 /// called with an already known dataflow fact (From). All other facts are
207 /// propagated like with the identityFlow.
208 ///
209 /// Given a flow function f = generateFlow(v, w), then for all incoming
210 /// dataflow facts x:
211 /// \code
212 /// f(w) = {v, w},
213 /// f(x) = {x}.
214 /// \endcode
215 /// In the exploded supergraph it may look as follows:
216 /// \code
217 /// x w u ...
218 /// | |\ | ...
219 /// inst | | \ | ...
220 /// v v v v ...
221 /// x w v u
222 /// \endcode
223 /// \note If the FactToGenerate already holds at the beginning of the
224 /// statement, this flow function does not kill it. For IFDS analysis it makes
225 /// no difference, but in the case of IDE, the corresponding edge functions
226 /// are being joined together potentially lowering precision. If that is an
227 /// issue, use transferFlow instead.
228 static auto generateFlow(d_t FactToGenerate, d_t From) {
229 struct GenFrom final : public FlowFunction<d_t, container_type> {
230 GenFrom(d_t GenValue, d_t FromValue)
231 : GenValue(std::move(GenValue)), FromValue(std::move(FromValue)) {}
232
233 container_type computeTargets(d_t Source) override {
234 if (Source == FromValue) {
235 return {std::move(Source), GenValue};
236 }
237 return {std::move(Source)};
238 }
239
240 d_t GenValue;
241 d_t FromValue;
242 };
243
244 return std::make_shared<GenFrom>(std::move(FactToGenerate),
245 std::move(From));
246 }

248 /// A flow function similar to generateFlow, that generates a new dataflow
249 /// fact (FactToGenerate), if the given Predicate evaluates to true on an
250 /// incoming dataflow fact
251 ///
252 /// So, given a flow function f = generateFlowIf(v, p), for all incoming
253 /// dataflow facts x:
254 /// \code
255 /// f(x) = {v, x} if p(x) == true
256 /// f(x) = {x} else.
257 /// \endcode
258 template <typename Fn = psr::TrueFn>
259 requires std::is_invocable_r_v<bool, Fn, d_t>
260 static auto generateFlowIf(d_t FactToGenerate, Fn Predicate) {
261 struct GenFlowIf final : public FlowFunction<d_t, container_type> {
262 GenFlowIf(d_t GenValue, Fn &&Predicate)
263 : GenValue(std::move(GenValue)),
264 Predicate(std::forward<Fn>(Predicate)) {}
265
266 container_type computeTargets(d_t Source) override {
267 if (std::invoke(Predicate, Source)) {
268 return {std::move(Source), GenValue};
269 }
270 return {std::move(Source)};
271 }
272
273 d_t GenValue;
274 [[no_unique_address]] std::decay_t<Fn> Predicate;
275 };
276
277 return std::make_shared<GenFlowIf>(std::move(FactToGenerate),
278 std::forward<Fn>(Predicate));
279 }

281 /// A flow function that generates multiple new dataflow facts
282 /// (FactsToGenerate) if called from an already known dataflow fact (From).
283 ///
284 /// Given a flow function f = generateManyFlows({v1, v2, ..., vN}, w), for all
285 /// incoming dataflow facts x:
286 /// \code
287 /// f(w) = {v1, v2, ..., vN, w}
288 /// f(x) = {x}.
289 /// \endcode
290 /// In the exploded supergraph it may look as follows:
291 /// \code
292 /// x w u ...
293 /// | |\ \ ... \ | ...
294 /// inst | | \ \ ... \ | ...
295 /// v v v v ... \ v ...
296 /// x w v1 v2 ... vN u
297 /// \endcode
298 template <is_iterable_over_v<d_t> Range = std::initializer_list<d_t>>
299 static auto generateManyFlows(Range &&FactsToGenerate, d_t From) {
300 struct GenMany final : public FlowFunction<d_t, container_type> {
301 GenMany(container_type &&GenValues, d_t FromValue)
302 : GenValues(std::move(GenValues)), FromValue(std::move(FromValue)) {}
303
304 container_type computeTargets(d_t Source) override {
305 if (Source == FromValue) {
306 auto Ret = GenValues;
307 Ret.insert(std::move(Source));
308 return Ret;
309 }
310 return {std::move(Source)};
311 }
312
313 container_type GenValues;
314 d_t FromValue;
315 };
316
317 return std::make_shared<GenMany>(detail::makeContainer<container_type>(
318 std::forward<Range>(FactsToGenerate)),
319 std::move(From));
320 }

532 /// A flow function that, similar to generateFlow, generates a new dataflow
533 /// fact (FactsToGenerate) when called with a specific dataflow fact (From).
534 /// Unlike generateFlow, it kills FactToGenerate if it is part of the incoming
535 /// facts. THis has no additional effect for IFDS analyses (which in fact
536 /// should use generateFlow instead), but for IDE analyses it may avoid
537 /// joining the edge functions reaching the FactToGenerate together which may
538 /// improve the analysis' precision.
539 ///
540 /// Given a flow function f = transferFlow(v, w), for all incoming dataflow
541 /// facts x:
542 /// \code
543 /// f(v) = {}
544 /// f(w) = {v, w}
545 /// f(x) = {x}.
546 /// \endcode
547 /// In the exploded supergraph it may look as follows:
548 /// \code
549 /// x w v u ...
550 /// | |\ | | ...
551 /// | | \ | ...
552 /// inst | | \ | ...
553 /// v v v v ...
554 /// x w v u
555 /// \endcode
556 static auto transferFlow(d_t FactToGenerate, d_t From) {
557 struct TransferFlow final : public FlowFunction<d_t, container_type> {
558 TransferFlow(d_t GenValue, d_t FromValue)
559 : GenValue(std::move(GenValue)), FromValue(std::move(FromValue)) {}
560
561 container_type computeTargets(d_t Source) override {
562 if (Source == FromValue) {
563 return {std::move(Source), GenValue};
564 }
565 if (Source == GenValue) {
566 return {};
567 }
568 return {std::move(Source)};
569 }
570
571 d_t GenValue;
572 d_t FromValue;
573 };
574
575 return std::make_shared<TransferFlow>(std::move(FactToGenerate),
576 std::move(From));
577 }

579 /// A flow function that takes two other flow functions OneFF and OtherFF and
580 /// applies both flow functions on each input merging the results together
581 /// with set-union.
582 ///
583 /// Given a flow function f = unionFlows(g, h), for all incoming dataflow
584 /// facts x:
585 /// \code
586 /// f(x) = g(x) u h(x). (where u denotes set-union)
587 /// \endcode
588 template <typename F1, typename F2>
589 requires(std::is_same_v<d_t, typename F2::value_type> &&
590 std::is_same_v<container_type, typename F2::container_type> &&
591 std::is_same_v<typename F1::container_type,
592 typename F2::container_type>)
593 auto unionFlows(FlowFunctionPtrTypeOf<F1> OneFF,
594 FlowFunctionPtrTypeOf<F2> OtherFF) {
595 struct UnionFlow final : public FlowFunction<d_t, container_type> {
596 UnionFlow(FlowFunctionPtrTypeOf<F1> OneFF,
597 FlowFunctionPtrTypeOf<F2> OtherFF) noexcept
598 : OneFF(std::move(OneFF)), OtherFF(std::move(OtherFF)) {}
599
600 container_type computeTargets(d_t Source) override {
601 auto OneRet = OneFF->computeTargets(Source);
602 auto OtherRet = OtherFF->computeTargets(std::move(Source));
603 if (OneRet.size() < OtherRet.size()) {
604 std::swap(OneRet, OtherRet);
605 }
606
607 OneRet.insert(std::make_move_iterator(OtherRet.begin()),
608 std::make_move_iterator(OtherRet.end()));
609 return OneRet;
610 }
611
612 FlowFunctionPtrTypeOf<F1> OneFF;
613 FlowFunctionPtrTypeOf<F2> OtherFF;
614 };
615
616 return std::make_shared<UnionFlow>(std::move(OneFF), std::move(OtherFF));
617 }

624template <typename AnalysisDomainTy, typename Container>
625class FlowFunctions
626 : protected FlowFunctionTemplates<typename AnalysisDomainTy::d_t,
627 Container> {
628 static_assert(
629 std::is_same<typename Container::value_type,
630 typename AnalysisDomainTy::d_t>::value,
631 "Value type of the container needs to match the type parameter D");
632
633public:
634 using d_t = typename AnalysisDomainTy::d_t;
635 using f_t = typename AnalysisDomainTy::f_t;
636 using n_t = typename AnalysisDomainTy::n_t;
637
638 using FlowFunctionType = FlowFunction<d_t, Container>;
639 using FlowFunctionPtrType = typename FlowFunctionType::FlowFunctionPtrType;
640
641 using container_type = typename FlowFunctionType::container_type;
642
643 virtual ~FlowFunctions() = default;
644
645 ///
646 /// Describes the effects of the current instruction, i.e. data-flows, along
647 /// normal (non-call, non-return) instructions. Analysis writers are free to
648 /// inspect the successor instructions, too, as a lookahead.
649 ///
650 /// Let instruction_1 := Curr, instruction_2 := Succ, and 0 the tautological
651 /// lambda fact.
652 ///
653 /// The returned flow function implementation f
654 /// (FlowFunction::computeTargets()) is applied to each data-flow fact d_i
655 /// that holds before the current statement under analysis. f's return type is
656 /// a set of (target) facts that have to be generated from the source fact d_i
657 /// by the data-flow solver. Each combination of input fact d_i (given as an
658 /// input to f) and respective output facts (f(d_i)) represents an edge that
659 /// must be "drawn" to construct the exploded supergraph for the analysis
660 /// problem to be solved.
661 ///
662 /// The concrete implementation of f is depending on the analysis problem. In
663 /// the following, we present a brief, contrived example:
664 ///
665 /// f is applied to each data-flow fact d_i that holds before instruction_1.
666 /// We assume that f is implemented to produce the following outputs.
667 /// \code
668 /// f(0) -> {0} // pass the lambda (or zero fact) as identity
669 /// f(o) -> {o, x} // generate a new fact x from o
670 /// f(.) -> {.} // pass all other facts that hold before
671 /// // instruction_1 as identity
672 /// \endcode
673 /// The above implementation corresponds to the following edges in the
674 /// exploded supergraph.
675 /// \code
676 /// 0 o ...
677 /// | |\ ...
678 /// x = instruction_1 o p | | \ ...
679 /// | | | ...
680 /// v v v ...
681 /// 0 o x ...
682 ///
683 /// y = instruction_2 q r
684 /// \endcode
685 virtual FlowFunctionPtrType getNormalFlowFunction(n_t Curr, n_t Succ) = 0;
686
687 ///
688 /// Handles call flows: describes the effects of a function call at callInst
689 /// to the callee target destFun. If a call instruction has multiple callee
690 /// targets, for instance, because it is an indirect function call that cannot
691 /// be analyzed precisely in a static manner, the call flow function will be
692 /// queried for each callee target.
693 ///
694 /// This flow function usually handles parameter passing and maps actual to
695 /// formal parameters. If an analysis writer does not wish to analyze a given
696 /// callee target they can return a flow function implementation that kills
697 /// all data-flow facts (e.g. KillAll) such that call is not followed. A
698 /// commonly used trick to model the effects of functions that are not present
699 /// (e.g. library functions such as malloc(), free(), etc.) is to kill all
700 /// facts at the call to the respective target and plugin the semantics in the
701 /// call-to-return flow function. In the call-to-return flow function, an
702 /// analysis writer can check if the function of interest is one of the
703 /// possible targets and then, return a flow function implementation that
704 /// describes the special semantics of that function call.
705 ///
706 /// Let start_point be the starting point of the callee target CalleeFun.
707 ///
708 /// The returned flow function implementation f
709 /// (FlowFunction::computeTargets()) is applied to each data-flow fact d_i
