20template <typename T>
21concept ProjectSymbolTable = requires(const T &ST, llvm::StringRef Name,
22 size_t InstId, typename T::n_t Inst) {
23 /// Reference to a Instruction/Statement
24 typename T::n_t;
25 /// Reference to a function
26 typename T::f_t;
27 /// Reference to a program module (potentially containing the whole
28 /// program)
29 typename T::m_t;
30 /// Reference to a Global Variable
31 typename T::g_t;
32
33 /// Returns the function (declaration or definition) if available,
34 /// nullptr/nullopt otherwise.
35 { ST.getFunction(Name) } -> std::convertible_to<Nullable<typename T::f_t>>;
36
37 /// Returns the function's definition if available, null otherwise.
38 {
39 ST.getFunctionDefinition(Name)
40 } -> std::convertible_to<Nullable<typename T::f_t>>;
41
42 /// Returns whether the IRDB contains a function (declaration or
43 /// definition) with the given name.
44 { ST.hasFunction(Name) } -> std::convertible_to<bool>;
45
46 /// Returns the global variable (declaration or definition) if available,
47 /// nullptr/nullopt otherwise.
48 {
49 ST.getGlobalVariable(Name)
50 } -> std::convertible_to<Nullable<typename T::g_t>>;
51
52 /// Returns the global variable's definition if available, nullptr/nullopt
53 /// otherwise.
54 {
55 ST.getGlobalVariableDefinition(Name)
56 } -> std::convertible_to<Nullable<typename T::g_t>>;
57
58 /// Returns the instruction to the corresponding Id. Returns
59 /// nullptr/nullopt, if there is no instruction for this Id
60 {
61 ST.getInstruction(InstId)
62 } -> std::convertible_to<Nullable<typename T::n_t>>;
63
64 /// Returns an instruction's Id. The instruction must belong to the managed
65 /// module for this function to work.
66 ///
67 /// You can expect the instruction-ids to be sequential in the interval [0,
68 /// N), where N is the total number of managed instructions.
69 { ST.getInstructionId(Inst) } -> std::convertible_to<size_t>;
70};

72/// This concept describes the minimum requirements for a project-IR-database.
73/// Each phasar-analysis works on a program that is managed by a ProjectIRDB.
74template <typename T>
75concept ProjectIRDB =
76 requires(const T &DB, typename T::n_t Inst, typename T::f_t Fun) {
77 /// Returns the managed module
78 { DB.getModule() } noexcept -> std::convertible_to<typename T::m_t>;
79
80 /// Check if debug information are available.
81 { DB.debugInfoAvailable() } -> std::convertible_to<bool>;
82
83 /// Returns a range of all function definitions and declarations available
84 { DB.getAllFunctions() } -> psr::is_iterable_over_v<typename T::f_t>;
85
86 /// Returns the number of functions in the managed module.
87 { DB.getNumFunctions() } -> std::convertible_to<size_t>;
88
89 /// Returns a range of all global variables (and global constants, e.g,
90 /// string literals) in the managed module
91 { DB.getAllGlobals() } -> psr::is_iterable_over_v<typename T::g_t>;
92
93 /// Returns the number of global variables (and global constants) in the
94 /// managed module.
95 { DB.getNumGlobals() } -> std::convertible_to<size_t>;
96
97 /// Returns a range of all instructions available.
98 { DB.getAllInstructions() } -> psr::is_iterable_over_v<typename T::n_t>;
99
100 /// Returns the number of instruction in the managed module.
101 { DB.getNumInstructions() } -> std::convertible_to<size_t>;
102
103 /// Returns the function that contains the given instruction Inst.
104 { DB.getFunctionOf(Inst) } -> std::convertible_to<typename T::f_t>;
105
106 /// Returns an iterable range of all instructions of the given function
107 /// that are part of the control-flow graph.
108 {
109 DB.getAllInstructionsOf(Fun)
110 } -> psr::is_iterable_over_v<typename T::n_t>;
111
112 requires ProjectSymbolTable<T>;
113 };
