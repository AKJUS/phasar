39template <typename AliasSetTy> class AliasSetOwner {
40public:
41 using allocator_type =
42#if HAS_MEMORY_RESOURCE
43 std::pmr::polymorphic_allocator<AliasSetTy>
44#else
45 llvm::RecyclingAllocator<llvm::BumpPtrAllocator, AliasSetTy> *
46#endif
47 ;
48
49 using memory_resource_type =
50#if HAS_MEMORY_RESOURCE
51 std::pmr::unsynchronized_pool_resource
52#else
53 llvm::RecyclingAllocator<llvm::BumpPtrAllocator, AliasSetTy>
54#endif
55 ;
56
57 AliasSetOwner(allocator_type Alloc) noexcept : Alloc(Alloc) {
58 if constexpr (std::is_pointer_v<allocator_type>) {
59 assert(Alloc != nullptr);
60 }
61 }
62 AliasSetOwner(AliasSetOwner &&) noexcept = default;
63
64 AliasSetOwner(const AliasSetOwner &) = delete;
65 AliasSetOwner &operator=(const AliasSetOwner &) = delete;
66 AliasSetOwner &operator=(AliasSetOwner &&) = delete;
67
68 ~AliasSetOwner() {
69 for (auto *PTS : OwnedPTS) {
70 std::destroy_at(PTS);
71#if HAS_MEMORY_RESOURCE
72 Alloc.deallocate(PTS, 1);
73#else
74 Alloc->Deallocate(PTS);
75#endif
76 }
77 OwnedPTS.clear();
78 }
79
80 BoxedPtr<AliasSetTy> acquire() {
81 auto RawMem =
82#if HAS_MEMORY_RESOURCE
83 Alloc.allocate(1)
84#else
85 Alloc->Allocate()
86#endif
87 ;
88 auto Ptr = new (RawMem) AliasSetTy();
89 OwnedPTS.insert(Ptr);
90 return &AllPTS.emplace_back(Ptr);
91 }
92
93 void release(AliasSetTy *PTS) noexcept {
94 if (LLVM_UNLIKELY(!OwnedPTS.erase(PTS))) {
95 llvm::report_fatal_error(
96 "ERROR: release AliasSet that was either already "
97 "freed, or never allocated with this AliasSetOwner!");
98 }
99 std::destroy_at(PTS);
100#if HAS_MEMORY_RESOURCE
101 Alloc.deallocate(PTS, 1);
102#else
103 Alloc->Deallocate(PTS);
104#endif
105 /// NOTE: Do not delete from AllPTS!
106 }
107
108 void reserve(size_t Capacity) { OwnedPTS.reserve(Capacity); }
109
110 [[nodiscard]] auto getAllAliasSets() const noexcept {
111 return llvm::make_range(OwnedPTS.begin(), OwnedPTS.end());
112 }
113
114private:
115 allocator_type Alloc{};
116 llvm::DenseSet<AliasSetTy *> OwnedPTS;
117 StableVector<AliasSetTy *> AllPTS;
118};
