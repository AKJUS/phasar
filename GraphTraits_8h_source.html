37template <typename Edge>
38concept is_graph_edge = requires(const Edge e1, Edge e2) {
39 { e1 == e2 } -> std::convertible_to<bool>;
40 { e1 != e2 } -> std::convertible_to<bool>;
41 { e1 < e2 } -> std::convertible_to<bool>;
42};

44template <typename GraphTrait>

90template <typename Graph>
91concept is_graph = requires(Graph g) {
92 typename GraphTraits<std::decay_t<Graph>>;
93 requires is_graph_trait<GraphTraits<std::decay_t<Graph>>>;
94};

96template <typename GraphTrait>
97concept is_reservable_graph_trait_v =
98 is_graph_trait<GraphTrait> && requires(typename GraphTrait::graph_type &g) {
99 { GraphTrait::reserve(g, size_t(0)) };
100 };

102template <typename GraphTrait>
103concept is_removable_graph_trait_v =
104 is_graph_trait<GraphTrait> &&
105 requires(typename GraphTrait::graph_type &g,
106 typename GraphTrait::vertex_t vtx,
107 typename GraphTrait::edge_iterator edge_it,
108 typename GraphTrait::roots_iterator root_it) {
109 typename GraphTrait::edge_iterator;
110 typename GraphTrait::roots_iterator;
111 { GraphTrait::removeEdge(g, vtx, edge_it) };
112 { GraphTrait::removeRoot(g, root_it) };
113 };

156template <typename GraphTy>
157std::decay_t<GraphTy> reverseGraph(GraphTy &&G)
158#if __cplusplus >= 202002L
159 requires is_graph<GraphTy>
160#endif
161{
162 std::decay_t<GraphTy> Ret;
163 using traits_t = GraphTraits<std::decay_t<GraphTy>>;
164 if constexpr (is_reservable_graph_trait_v<traits_t>) {
165 traits_t::reserve(Ret, traits_t::size(G));
166 }
167
168 for (auto &Nod : traits_t::nodes(G)) {
169 traits_t::addNode(Ret, forward_like<GraphTy>(Nod));
170 }
171 for (auto I : traits_t::vertices(G)) {
172 for (auto Child : traits_t::outEdges(G, I)) {
173 traits_t::addEdge(Ret, traits_t::target(Child),
174 traits_t::withEdgeTarget(Child, I));
175 }
176 if (traits_t::outDegree(G, I) == 0) {
177 traits_t::addRoot(Ret, I);
178 }
179 }
180 return Ret;
181}

183struct DefaultNodeTransform {
184 template <typename N> std::string operator()(const N &Nod) const {
185 std::string Buf;
186 llvm::raw_string_ostream ROS(Buf);
187 ROS << Nod;
188 ROS.flush();
189 return Buf;
190 }
191};

193template <typename GraphTy, typename NodeTransform = DefaultNodeTransform>
194void printGraph(const GraphTy &G, llvm::raw_ostream &OS,
195 llvm::StringRef Name = "", NodeTransform NodeToString = {})
196#if __cplusplus >= 202002L
197 requires is_graph<GraphTy>
198#endif
199{
200 using traits_t = GraphTraits<GraphTy>;
201
202 OS << "digraph " << Name << " {\n";
203 psr::scope_exit CloseBrace = [&OS] { OS << "}\n"; };
204
205 auto Sz = traits_t::size(G);
206
207 for (size_t I = 0; I < Sz; ++I) {
208 OS << I;
209 if constexpr (!std::is_same_v<llvm::NoneType,
210 typename traits_t::value_type>) {
211 OS << "[label=\"";
212 OS.write_escaped(std::invoke(NodeToString, traits_t::node(G, I)));
213 OS << "\"]";
214 }
215 OS << ";\n";
216 for (const auto &Edge : traits_t::outEdges(G, I)) {
217 OS << I << "->" << Edge << ";\n";
218 }
219 }
220}
