32/// \brief Similar to SparseLLVMBasedICFG; the only difference is that this one
33/// *is* no LLVMBasedICFG -- it contains a pointer to an already existing one.
34/// It still owns the sparse value-flow graphs.
35///
36/// Use this in the IDESolver or IFDSSolver to profit from the SparseIFDS or
37/// SparseIDE optimization after Karakaya et al. "Symbol-Specific Sparsification
38/// of Interprocedural Distributive Environment Problems"
39/// <https://doi.org/10.48550/arXiv.2401.14813>
40class SparseLLVMBasedICFGView
41 : public LLVMBasedCFG,
42 public ICFGBase<SparseLLVMBasedICFGView>,
43 public SparseLLVMBasedCFGProvider<SparseLLVMBasedICFGView> {
44 friend ICFGBase;
45 friend SparseLLVMBasedCFGProvider<SparseLLVMBasedICFGView>;
46
47public:
48 explicit SparseLLVMBasedICFGView(const LLVMBasedICFG *ICF,
49 LLVMAliasInfoRef PT);
50
51 ~SparseLLVMBasedICFGView();
52
53 // To make the IDESolver happy...
54 operator const LLVMBasedICFG &() const noexcept { return *ICF; }
55
56private:
57 [[nodiscard]] FunctionRange getAllFunctionsImpl() const;
58 [[nodiscard]] f_t getFunctionImpl(llvm::StringRef Fun) const;
59
60 [[nodiscard]] bool isIndirectFunctionCallImpl(n_t Inst) const;
61 [[nodiscard]] bool isVirtualFunctionCallImpl(n_t Inst) const;
62 [[nodiscard]] std::vector<n_t> allNonCallStartNodesImpl() const;
63 [[nodiscard]] llvm::SmallVector<n_t> getCallsFromWithinImpl(f_t Fun) const;

65 getReturnSitesOfCallAtImpl(n_t Inst) const;
66 void printImpl(llvm::raw_ostream &OS) const;
67 [[nodiscard]] const CallGraph<n_t, f_t> &getCallGraphImpl() const noexcept;
68
69 [[nodiscard]] const SparseLLVMBasedCFG &
70 getSparseCFGImpl(const llvm::Function *Fun, const llvm::Value *Val) const;
71
72 const LLVMBasedICFG *ICF{};
73 std::unique_ptr<SVFGCache> SparseCFGCache;
74 LLVMAliasInfoRef AliasAnalysis;
75};
