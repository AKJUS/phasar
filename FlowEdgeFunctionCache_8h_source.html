39template <typename KeyT> class DefaultMapKeyCompressor {
40public:
41 using KeyType = KeyT;
42 using CompressedType = KeyT;
43
44 [[nodiscard]] inline CompressedType getCompressedID(KeyT Key) { return Key; }
45};

47template <typename... Ts> class MapKeyCompressorCombinator : public Ts... {
48public:
49 using Ts::getCompressedID...;
50};

52class LLVMMapKeyCompressor {
53public:
54 using KeyType = const llvm::Value *;
55 using CompressedType = uint32_t;
56
57 [[nodiscard]] inline CompressedType getCompressedID(KeyType Key) {
58 auto Search = Map.find(Key);
59 if (Search == Map.end()) {
60 return Map.insert(std::make_pair(Key, Map.size() + 1)).first->getSecond();
61 }
62 return Search->getSecond();
63 }
64
65private:
66 llvm::DenseMap<KeyType, CompressedType> Map{};
67};

69/**
70 * This class caches flow and edge functions to avoid their reconstruction.
71 * When a flow or edge function must be applied to multiple times, a cached
72 * version is used if existend, otherwise a new one is created and inserted
73 * into the cache.
74 */
75template <typename AnalysisDomainTy,
76 typename Container = std::set<typename AnalysisDomainTy::d_t>>
77class FlowEdgeFunctionCache {
78 using IDEProblemType = IDETabulationProblem<AnalysisDomainTy, Container>;
79 using FlowFunctionPtrType = typename IDEProblemType::FlowFunctionPtrType;
80
81 using n_t = typename AnalysisDomainTy::n_t;
82 using d_t = typename AnalysisDomainTy::d_t;
83 using f_t = typename AnalysisDomainTy::f_t;
84 using t_t = typename AnalysisDomainTy::t_t;
85 using l_t = typename AnalysisDomainTy::l_t;
86
87 using DTKeyCompressorType = std::conditional_t<
88 std::is_base_of_v<llvm::Value, std::remove_pointer_t<d_t>>,
89 LLVMMapKeyCompressor, DefaultMapKeyCompressor<d_t>>;
90 using NTKeyCompressorType = std::conditional_t<
91 std::is_base_of_v<llvm::Value, std::remove_pointer_t<n_t>>,
92 LLVMMapKeyCompressor, DefaultMapKeyCompressor<n_t>>;
93
94 using MapKeyCompressorType = std::conditional_t<
95 std::is_same_v<NTKeyCompressorType, DTKeyCompressorType>,
96 NTKeyCompressorType,
97 MapKeyCompressorCombinator<NTKeyCompressorType, DTKeyCompressorType>>;
98
99private:
100 MapKeyCompressorType KeyCompressor;
101
102 using EdgeFuncInstKey = uint64_t;
103 using EdgeFuncNodeKey = std::conditional_t<
104 std::is_base_of_v<llvm::Value, std::remove_pointer_t<d_t>>, uint64_t,
105 std::pair<d_t, d_t>>;
106 using InnerEdgeFunctionMapType =
107 EquivalenceClassMap<EdgeFuncNodeKey, EdgeFunction<l_t>>;
108
109 IDETabulationProblem<AnalysisDomainTy, Container> &Problem;
110 // Auto add zero
111 bool AutoAddZero;
112 d_t ZV;

114 struct NormalEdgeFlowData {
115 NormalEdgeFlowData(FlowFunctionPtrType Val)
116 : FlowFuncPtr(std::move(Val)), EdgeFunctionMap{} {}
117 NormalEdgeFlowData(InnerEdgeFunctionMapType Map)
118 : FlowFuncPtr(nullptr), EdgeFunctionMap{std::move(Map)} {}
119
120 FlowFunctionPtrType FlowFuncPtr;
121 InnerEdgeFunctionMapType EdgeFunctionMap;
122 };

124 // Caches for the flow/edge functions
125 std::map<EdgeFuncInstKey, NormalEdgeFlowData> NormalFunctionCache;
126
127 // Caches for the flow functions
128 std::map<std::tuple<n_t, f_t>, FlowFunctionPtrType> CallFlowFunctionCache;
129 std::map<std::tuple<n_t, f_t, n_t, n_t>, FlowFunctionPtrType>
130 ReturnFlowFunctionCache;
131 std::map<std::tuple<n_t, n_t>, FlowFunctionPtrType>
132 CallToRetFlowFunctionCache;
133 // Caches for the edge functions
134 std::map<std::tuple<n_t, d_t, f_t, d_t>, EdgeFunction<l_t>>
135 CallEdgeFunctionCache;
136 std::map<std::tuple<n_t, f_t, n_t, d_t, n_t, d_t>, EdgeFunction<l_t>>
137 ReturnEdgeFunctionCache;
138 std::map<EdgeFuncInstKey, InnerEdgeFunctionMapType>
139 CallToRetEdgeFunctionCache;
140 std::map<std::tuple<n_t, d_t, n_t, d_t>, EdgeFunction<l_t>>
141 SummaryEdgeFunctionCache;

192 FlowFunctionPtrType getNormalFlowFunction(n_t Curr, n_t Succ) {
193 assertNotNull(Curr);
194 assertNotNull(Succ);
195 PAMM_GET_INSTANCE;
196 IF_LOG_ENABLED(
197 PHASAR_LOG_LEVEL(DEBUG, "Normal flow function factory call");
198 PHASAR_LOG_LEVEL(DEBUG, "(N) Curr Inst : " << NToString(Curr));
199 PHASAR_LOG_LEVEL(DEBUG, "(N) Succ Inst : " << NToString(Succ)));
200 auto Key = createEdgeFunctionInstKey(Curr, Succ);
201 auto SearchNormalFlowFunction = NormalFunctionCache.find(Key);
202 if (SearchNormalFlowFunction != NormalFunctionCache.end()) {
203 PHASAR_LOG_LEVEL(DEBUG, "Flow function fetched from cache");
204 INC_COUNTER("Normal-FF Cache Hit", 1, Full);
205 if (SearchNormalFlowFunction->second.FlowFuncPtr != nullptr) {
206 return SearchNormalFlowFunction->second.FlowFuncPtr;
207 }
208 auto FF = (AutoAddZero)
209 ? std::make_shared<ZeroedFlowFunction<d_t, Container>>(
210 Problem.getNormalFlowFunction(Curr, Succ), ZV)
211 : Problem.getNormalFlowFunction(Curr, Succ);
212 SearchNormalFlowFunction->second.FlowFuncPtr = FF;
213 return FF;
214 }
215 INC_COUNTER("Normal-FF Construction", 1, Full);
216 auto FF = (AutoAddZero)
217 ? std::make_shared<ZeroedFlowFunction<d_t, Container>>(
218 Problem.getNormalFlowFunction(Curr, Succ), ZV)
219 : Problem.getNormalFlowFunction(Curr, Succ);
220 NormalFunctionCache.insert(std::make_pair(Key, NormalEdgeFlowData(FF)));
221 PHASAR_LOG_LEVEL(DEBUG, "Flow function constructed");
222
223 return FF;
224 }

332 EdgeFunction<l_t> getNormalEdgeFunction(n_t Curr, d_t CurrNode, n_t Succ,
333 d_t SuccNode) {
334 assertNotNull(Curr);
335 assertNotNull(Succ);
336
337 PAMM_GET_INSTANCE;
338 IF_LOG_ENABLED(
339 PHASAR_LOG_LEVEL(DEBUG, "Normal edge function factory call");
340 PHASAR_LOG_LEVEL(DEBUG, "(N) Curr Inst : " << NToString(Curr));
341 PHASAR_LOG_LEVEL(DEBUG, "(D) Curr Node : " << DToString(CurrNode));
342 PHASAR_LOG_LEVEL(DEBUG, "(N) Succ Inst : " << NToString(Succ));
343 PHASAR_LOG_LEVEL(DEBUG, "(D) Succ Node : " << DToString(SuccNode)));
344
345 EdgeFuncInstKey OuterMapKey = createEdgeFunctionInstKey(Curr, Succ);
346 auto SearchInnerMap = NormalFunctionCache.find(OuterMapKey);
347 if (SearchInnerMap != NormalFunctionCache.end()) {
348 auto SearchEdgeFunc = SearchInnerMap->second.EdgeFunctionMap.find(
349 createEdgeFunctionNodeKey(CurrNode, SuccNode));
350 if (SearchEdgeFunc != SearchInnerMap->second.EdgeFunctionMap.end()) {
351 INC_COUNTER("Normal-EF Cache Hit", 1, Full);
352 PHASAR_LOG_LEVEL(DEBUG, "Edge function fetched from cache");
353 PHASAR_LOG_LEVEL(DEBUG,
354 "Provide Edge Function: " << SearchEdgeFunc->second);
355 return SearchEdgeFunc->second;
356 }
357 INC_COUNTER("Normal-EF Construction", 1, Full);
358 auto EF = Problem.getNormalEdgeFunction(Curr, CurrNode, Succ, SuccNode);
359
360 SearchInnerMap->second.EdgeFunctionMap.insert(
361 createEdgeFunctionNodeKey(CurrNode, SuccNode), EF);
362
363 PHASAR_LOG_LEVEL(DEBUG, "Edge function constructed");
364 PHASAR_LOG_LEVEL(DEBUG, "Provide Edge Function: " << EF);
365 return EF;
366 }
367 INC_COUNTER("Normal-EF Construction", 1, Full);
368 auto EF = Problem.getNormalEdgeFunction(Curr, CurrNode, Succ, SuccNode);
369
370 NormalFunctionCache.try_emplace(
371 OuterMapKey, NormalEdgeFlowData(InnerEdgeFunctionMapType{std::make_pair(
372 createEdgeFunctionNodeKey(CurrNode, SuccNode), EF)}));
373
374 PHASAR_LOG_LEVEL(DEBUG, "Edge function constructed");
375 PHASAR_LOG_LEVEL(DEBUG, "Provide Edge Function: " << EF);
376 return EF;
377 }
