62/// \brief A type-erased reference to any object implementing the
63/// IsAliasIterator interface. Use this, if your alias-aware analysis just needs
64/// a minimal interface to work with aliases and does not require the
65/// versatility of AliasInfoRef.
66///
67/// This is a *non-owning* reference similar to std::string_view and
68/// llvm::ArrayRef. Pass values of this type by value.
69///
70/// Example:
71/// \code
72/// LLVMAliasSet ASet(...);
73/// LLVMAliasIteratorRef AA = &ASet;
74/// \endcode
75template <typename V, typename N>
76class [[gsl::Pointer]] AliasIteratorRef : private TypeErasureUtils {
77public:
78 using n_t = N;
79 using v_t = V;
80
81 using ForallAliasesOfFn = void (*)(void *, ByConstRef<V>, ByConstRef<N>,
82 llvm::function_ref<void(V)>);
83
84 template <typename ConcreteAA,
85 typename = std::enable_if_t<
86 !std::is_base_of_v<AliasIteratorRef, ConcreteAA> &&
87 (detail::IsAliasIteratorFor<ConcreteAA, V, N>::value ||
88 detail::HasGetAliasSet<ConcreteAA>::value)>>
89 constexpr AliasIteratorRef(ConcreteAA *AA) noexcept
90 : AA(getOpaquePtr(psr::assertNotNull(AA))), Fn(TypeErase<ConcreteAA>) {
91 static_assert(IsAliasIterator<AliasIteratorRef>);
92 }
93 template <typename ConcreteAA,
94 typename = std::enable_if_t<
95 !std::is_base_of_v<AliasIteratorRef, ConcreteAA> &&
96 (detail::IsAliasIteratorFor<ConcreteAA, V, N>::value ||
97 detail::HasGetAliasSet<ConcreteAA>::value) &&
98 CanSSO<ConcreteAA>>>
99 constexpr AliasIteratorRef(ConcreteAA AA) noexcept
100 : AA(getOpaquePtr(AA)), Fn(TypeErase<ConcreteAA>) {
101 static_assert(IsAliasIterator<AliasIteratorRef>);
102 }
103
104 constexpr explicit AliasIteratorRef(void *AA, ForallAliasesOfFn Fn) noexcept
105 : AA(AA), Fn(Fn) {
106 assert(Fn != nullptr);
107 }

109 constexpr AliasIteratorRef(const AliasIteratorRef &) noexcept = default;

114 /// \brief Invokes the callback WithAlias for all aliases of Of at the
115 /// instruction At.
116 ///
117 /// Note: The alias-relation is reflexive, so WithAlias is also called with
118 /// Of.
119 ///
120 /// \param Of The pointer, for which the aliases should be iterated
121 /// \param At The instruction, where the alias-query is raised.
122 /// Implementations may ignore this parameter
123 /// \param WithAlias Callback to invoke for each alias of Of
124 void forallAliasesOf(ByConstRef<v_t> Of, ByConstRef<n_t> At,
125 llvm::function_ref<void(v_t)> WithAlias) {
126 assert(Fn != nullptr);
127 Fn(AA, Of, At, WithAlias);
128 }
129
130 /// \brief Convenience function to aggregate all aliases of Of in a set.
131 ///
132 /// \param Of The pointer, for which the aliases should be iterated
133 /// \param At The instruction, where the alias-query is raised.
134 /// Implementations may ignore this parameter
135 /// \tparam SetT The set-type of the set to create
136 /// \returns A set of type SetT containing all aliases of Of
137 template <typename SetT = std::set<v_t>>
138 [[nodiscard]] SetT asSet(ByConstRef<v_t> Of, ByConstRef<n_t> At) {
139 SetT Set;
140 forallAliasesOf(Of, At,
141 [&Set](v_t Alias) { Set.insert(std::move(Alias)); });
142 return Set;
143 }
144
145private:
146 template <typename ConcreteAA>
147 static void aliasesOfThunk(void *AA, ByConstRef<v_t> Of, ByConstRef<n_t> At,
148 llvm::function_ref<void(v_t)> WithAlias) {
149 auto *CAA = fromOpaquePtr<ConcreteAA>(AA);
150 if constexpr (detail::IsAliasIteratorFor<ConcreteAA, V, N>::value) {
151 return (void)CAA->forallAliasesOf(Of, At, WithAlias);
152 } else {
153 auto AliasSetPtr = CAA->getAliasSet(Of, At);
154 for (auto &&Alias : *AliasSetPtr) {
155 WithAlias(PSR_FWD(Alias));
156 }
157 }
158 }
159
160 template <typename ConcreteAA>
161 static constexpr ForallAliasesOfFn TypeErase = &aliasesOfThunk<ConcreteAA>;
162
163 void *AA{};
164 ForallAliasesOfFn Fn{};
165}; // namespace psr

167} // namespace psr
