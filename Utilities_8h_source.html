38template <typename T>
39std::set<std::set<T>> computePowerSet(const std::set<T> &S) {
40 // compute all subsets of {a, b, c, d}
41 // bit-pattern - {d, c, b, a}
42 // 0000 {}
43 // 0001 {a}
44 // 0010 {b}
45 // 0011 {a, b}
46 // 0100 {c}
47 // 0101 {a, c}
48 // 0110 {b, c}
49 // 0111 {a, b, c}
50 // 1000 {d}
51 // 1001 {a, d}
52 // 1010 {b, d}
53 // 1011 {a, b, d}
54 // 1100 {c, d}
55 // 1101 {a, c, d}
56 // 1110 {b, c, d}
57 // 1111 {a, b, c, d}
58 std::set<std::set<T>> Powerset;
59 for (std::size_t I = 0; I < (1 << S.size()); ++I) {
60 std::set<T> Subset;
61 for (std::size_t J = 0; J < S.size(); ++J) {
62 if ((I & (1 << J)) > 0) {
63 auto It = S.begin();
64 advance(It, J);
65 Subset.insert(*It);
66 }
67 Powerset.insert(Subset);
68 }
69 }
70 return Powerset;
71}

73/// \brief Computes the set-intersection of the potentially unordered sets
74/// Dest and Src and stores the result back in Dest.
75///
76/// This function should work on all types of sets as long as they provide the
77/// type value_type specifying, which type their elements have.
78/// By-reference iteration is required, but the elements do not have any
79/// requirements, although the performance is probably higher for small
80/// elements that are trivially copyable.
81template <typename ContainerTy, typename OtherContainerTy>
82std::enable_if_t<!has_erase_iterator_v<ContainerTy>>
83intersectWith(ContainerTy &Dest, const OtherContainerTy &Src) {
84 static_assert(std::is_same_v<typename ContainerTy::value_type,
85 typename OtherContainerTy::value_type>,
86 "The containers Src and Dest must be compatible");
87 using ValueTy = typename ContainerTy::value_type;
88 using ElementTy =
89 std::conditional_t<std::is_trivially_copy_constructible_v<ValueTy> &&
90 sizeof(ValueTy) <= sizeof(void *),
91 ValueTy, ValueTy *>;
92
93 auto removeFrom = [](auto &Dst, auto &&Elem) { // NOLINT
94 if constexpr (std::is_same_v<ValueTy, ElementTy>) {
95 Dst.erase(Elem);
96 } else {
97 Dst.erase(*Elem);
98 }
99 };
100
101 /// This whole functionality is only for computing the set-intersection of
102 /// Dest and Src storing the result in-place in Dest. It would be a big win,
103 /// if the STL would provide us with in-place set operations and those that
104 /// do not require the sets to be sorted...
105
106 llvm::SmallVector<ElementTy, 16> Buffer;
107
108 if constexpr (std::is_same_v<ValueTy, ElementTy>) {
109 for (auto &&Elem : Dest) {
110 if (!Src.count(Elem)) {
111 Buffer.push_back(Elem);
112 }
113 }
114 } else {
115 for (auto &Elem : Dest) {
116 if (!Src.count(Elem)) {
117 Buffer.insert(&Elem);
118 }
119 }
120 }
121
122 for (auto &&Elem : Buffer) {
123 removeFrom(Dest, Elem);
124 }
125}

127template <typename ContainerTy, typename OtherContainerTy>
128std::enable_if_t<has_erase_iterator_v<ContainerTy>>
129intersectWith(ContainerTy &Dest, const OtherContainerTy &Src) {
130 static_assert(std::is_same_v<typename ContainerTy::value_type,
131 typename OtherContainerTy::value_type>,
132 "The containers Src and Dest must be compatible");
133
134 for (auto It = Dest.begin(), End = Dest.end(); It != End;) {
135 if (Src.count(*It)) {
136 ++It;
137 } else {
138 It = Dest.erase(It);
139 }
140 }
141}

179/// Based on the reference implementation of std::remove_if
180/// <https://en.cppreference.com/w/cpp/algorithm/remove> and optimized for the
181/// case that a sorted list of indices is given instead of an unary predicate
182/// specifying the elements to be removed.
183template <typename It, typename EndIt, typename IdxIt,
184 typename IdxEndIt> // NOLINTNEXTLINE(readability-identifier-naming)
185It remove_by_index(It First, EndIt Last, IdxIt FirstIndex, IdxEndIt LastIndex) {
186 if (FirstIndex == LastIndex || First == Last) {
187 return Last;
188 }
189
190 if constexpr (std::is_same_v<It, EndIt> &&
191 std::is_same_v<
192 std::random_access_iterator_tag,
193 typename std::iterator_traits<It>::iterator_category>) {
194 /// Random-access version inspired from
195 /// "https://codereview.stackexchange.com/a/207056" and slightly changed
196
197 auto Bounds = size_t(std::distance(First, Last));
198 auto Out = std::next(First, *FirstIndex);
199 auto In = std::next(Out);
200 while (++FirstIndex != LastIndex) {
201 auto CurrIdx = *FirstIndex;
202 if (*std::prev(FirstIndex) + 1 == CurrIdx) {
203 ++In;
204 continue;
205 }
206 if (LLVM_UNLIKELY(CurrIdx >= Bounds)) {
207 break;
208 }
209
210 auto Tar = std::next(First, CurrIdx);
211 Out = std::move(In, Tar, Out);
212 In = std::next(Tar);
213 }
214 return std::move(In, Last, Out);
215 } else {
216 auto CurrIdx = *FirstIndex;
217 First = std::next(First, CurrIdx);
218 if (First == Last) {
219 return First;
220 }
221 for (auto I = First; I != Last; ++CurrIdx, ++I) {
222 if (CurrIdx != *FirstIndex) {
223 *First++ = std::move(*I);
224 if (++FirstIndex == LastIndex) {
225 return std::move(std::next(I), Last, First);
226 }
227 }
228 }
229 return First;
230 }
231}

252/// See <https://en.cppreference.com/w/cpp/utility/forward_like>
253template <class T, class U>
254[[nodiscard]] LLVM_ATTRIBUTE_ALWAYS_INLINE constexpr auto &&
255forward_like(U &&X) noexcept { // NOLINT
256 // NOLINTNEXTLINE
257 constexpr bool is_adding_const = std::is_const_v<std::remove_reference_t<T>>;
258 if constexpr (std::is_lvalue_reference_v<T &&>) {
259 if constexpr (is_adding_const) {
260 return std::as_const(X);
261 } else {
262 return static_cast<U &>(X);
263 }
264 } else {
265 if constexpr (is_adding_const) {
266 return std::move(std::as_const(X));
267 } else {
268 return std::move(X); // NOLINT
269 }
270 }
