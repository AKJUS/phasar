template<typename T, typename IdT>
class psr::Compressor< T, IdT, std::enable_if_t<!CanEfficientlyPassByValue< T > > >

A utility class that assigns a sequential Id to every inserted object.

This specialization handles types that cannot be efficiently passed by value
