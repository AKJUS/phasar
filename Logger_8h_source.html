86// For performance reason, we want to disable any
87// formatting computation that would go straight into
88// logs if logs are deactivated. This macro does just
89// that
90#define IF_LOG_ENABLED_BOOL(condition, computation) \
91 if (LLVM_UNLIKELY(condition)) { \
92 computation; \
93 }
94
95#define IS_LOG_ENABLED ::psr::Logger::isLoggingEnabled()
96#define IF_LOG_ENABLED(computation) \
97 IF_LOG_ENABLED_BOOL(::psr::Logger::isLoggingEnabled(), computation)
98
99#define IS_LOG_LEVEL_ENABLED(level) \
100 (::psr::Logger::isLoggingEnabled() && \
101 (::psr::SeverityLevel::level) >= ::psr::Logger::getLoggerFilterLevel())
102#define IF_LOG_LEVEL_ENABLED(level, computation) \
103 IF_LOG_ENABLED_BOOL(IS_LOG_LEVEL_ENABLED(level), computation)
104
105#define PHASAR_LOG_LEVEL(level, message) \
106 do { \
107 IF_LOG_ENABLED_BOOL(IS_LOG_LEVEL_ENABLED(level), { \
108 auto &Stream = ::psr::Logger::getLogStreamWithLinePrefix( \
109 ::psr::SeverityLevel::level, std::nullopt); \
110 /* NOLINTNEXTLINE(bugprone-macro-parentheses) */ \
111 Stream << message << '\n'; \
112 }) \
113 } while (false)
