27/// \brief A set-type that can compactly store sets of sequential integer-like
28/// types.
29///
30/// Use this type for sequential (unsigned) integers and ids that can convert
31/// from and to uint32_t.
32///
33/// \tparam IdT The type of elements to store in this set. Must be losslessly
34/// convertible from and to uint32_t.
35/// \tparam BitVectorTy The underlying bit-vector to use. Must be either
36/// llvm::BitVector or llvm::SmallBitVector.
37template <typename IdT, typename BitVectorTy = llvm::BitVector> class BitSet {
38 static llvm::ArrayRef<uintptr_t> getWords(const llvm::BitVector &BV,
39 uintptr_t & /*Store*/) {
40 return BV.getData();

42 static llvm::ArrayRef<uintptr_t> getWords(const llvm::SmallBitVector &BV,
43 uintptr_t &Store) {

147 [[nodiscard]] friend bool operator==(const BitSet &Lhs,
148 const BitSet &Rhs) noexcept {
149 // Check, whether Lhs and Rhs actually have the same bits set and not
150 // whether their internal representation is exactly identitcal
151 uintptr_t LhsStore{};
152 uintptr_t RhsStore{};
153
154 auto LhsWords = getWords(Lhs.Bits, LhsStore);
155 auto RhsWords = getWords(Rhs.Bits, RhsStore);
156 if (LhsWords.size() == RhsWords.size()) {
157 return LhsWords == RhsWords;
158 }
159 auto MinSize = std::min(LhsWords.size(), RhsWords.size());
160 if (LhsWords.slice(0, MinSize) != RhsWords.slice(0, MinSize)) {
161 return false;
162 }
163 auto Rest = (LhsWords.size() > RhsWords.size() ? LhsWords : RhsWords)
164 .slice(MinSize);
165 return std::all_of(Rest.begin(), Rest.end(),
166 [](auto Word) { return Word == 0; });
167 }

181 /// Calls the given handler function for each sert bit in the bitset.
182 ///
183 /// This is likely faster than using iterators.
184 template <std::invocable<IdT> HandlerFn>
185 void foreach (HandlerFn Handler) const
186 noexcept(std::is_nothrow_invocable_v<HandlerFn &, IdT>) {
187 uintptr_t Store{};
188 auto Words = getWords(Bits, Store);
189 uint32_t Offset = 0;
190 for (auto W : Words) {
191 while (W) {
192 auto Curr = std::countr_zero(W) + Offset;
193 W &= W - 1;
194 std::invoke(Handler, IdT(Curr));
195 }
196
197 Offset += sizeof(W) * CHAR_BIT;
198 }
199 }

224 [[nodiscard]] bool isSubsetOf(const BitSet &Of) const {
225 uintptr_t Buf = 0;
226 uintptr_t OfBuf = 0;
227
228 auto Words = getWords(Bits, Buf);
229 auto OfWords = getWords(Of.Bits, OfBuf);
230 if (Words.size() > OfWords.size()) {
231 if (llvm::any_of(Words.drop_front(OfWords.size()),
232 [](uintptr_t W) { return W != 0; })) {
233 return false;
234 }
235 }
236
237 for (auto [W, OfW] : llvm::zip(Words, OfWords)) {
238 if ((W & ~OfW) != 0) {
239 return false;
240 }
241 }
242
243 return true;
244 }
245
246 [[nodiscard]] bool isSupersetOf(const BitSet &Of) const {
247 return Of.isSubsetOf(*this);
248 }
