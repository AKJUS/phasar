30/// \brief A utility class that assigns a sequential Id to every inserted
31/// object.
32///
33/// This specialization handles types that can be efficiently passed by value
34template <typename T, typename IdT>
35class Compressor<T, IdT, std::enable_if_t<CanEfficientlyPassByValue<T>>> {
36public:
37 void reserve(size_t Capacity) {
38 assert(Capacity <= UINT32_MAX);
39 ToInt.reserve(Capacity);
40 FromInt.reserve(Capacity);
41 }
42
43 IdT getOrInsert(T Elem) {
44 auto [It, Inserted] = ToInt.try_emplace(Elem, IdT(ToInt.size()));
45 if (Inserted) {
46 FromInt.push_back(Elem);
47 }
48 return It->second;
49 }
50
51 std::pair<IdT, bool> insert(T Elem) {
52 auto [It, Inserted] = ToInt.try_emplace(Elem, IdT(ToInt.size()));
53 if (Inserted) {
54 FromInt.push_back(Elem);
55 }
56 return {It->second, Inserted};
57 }
58
59 [[nodiscard]]
60 std::optional<IdT> getOrNull(T Elem) const {
61 if (auto It = ToInt.find(Elem); It != ToInt.end()) {
62 return It->second;
63 }
64 return std::nullopt;
65 }
66
67 [[nodiscard]] IdT get(T Elem) const {
68 auto It = ToInt.find(Elem);
69 assert(It != ToInt.end());
70 return It->second;
71 }
72
73 [[nodiscard]] bool inbounds(IdT Idx) const noexcept {
74 return FromInt.inbounds(Idx);
75 }
76
77 [[nodiscard]] T operator[](IdT Idx) const noexcept {
78 assert(inbounds(Idx));
79 return FromInt[Idx];
80 }
81
82 [[nodiscard]] size_t size() const noexcept { return FromInt.size(); }
83 [[nodiscard]] size_t capacity() const noexcept {
84 return FromInt.capacity() +
85 ToInt.getMemorySize() / sizeof(typename decltype(ToInt)::value_type);
86 }
87
88 [[nodiscard]] auto begin() const noexcept { return FromInt.begin(); }
89 [[nodiscard]] auto end() const noexcept { return FromInt.end(); }
90
91 [[nodiscard]] auto enumerate() const noexcept { return FromInt.enumerate(); }
92
93 void clear() noexcept {
94 ToInt.clear();
95 FromInt.clear();
96 }
97
98private:
99 llvm::DenseMap<T, IdT> ToInt;
100 TypedVector<IdT, T, 0> FromInt;
101};

103/// \brief A utility class that assigns a sequential Id to every inserted
104/// object.
105///
106/// This specialization handles types that cannot be efficiently passed by value
107template <typename T, typename IdT>
108class Compressor<T, IdT, std::enable_if_t<!CanEfficientlyPassByValue<T>>> {
109public:
110 void reserve(size_t Capacity) {
111 assert(Capacity <= UINT32_MAX);
112 ToInt.reserve(Capacity);
113 }
114
115 /// Returns the index of the given element in the compressors storage. If
116 /// the element isn't present yet, it will be added first and its index will
117 /// then be returned.
118 IdT getOrInsert(const T &Elem) {
119 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
120 return It->second;
121 }
122 auto Ret = IdT(FromInt.size());
123 auto *Ins = &FromInt.emplace_back(Elem);
124 ToInt[Ins] = Ret;
125 return Ret;
126 }
127
128 /// Returns the index of the given element in the compressors storage. If
129 /// the element isn't present yet, it will be added first and its index will
130 /// then be returned.
131 IdT getOrInsert(T &&Elem) {
132 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
133 return It->second;
134 }
135 auto Ret = IdT(FromInt.size());
136 auto *Ins = &FromInt.emplace_back(std::move(Elem));
137 ToInt[Ins] = Ret;
138 return Ret;
139 }
140
141 std::pair<IdT, bool> insert(const T &Elem) {
142 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
143 return {It->second, false};
144 }
145 auto Ret = Id(FromInt.size());
146 auto *Ins = &FromInt.emplace_back(Elem);
147 ToInt[Ins] = Ret;
148 return {Ret, true};
149 }
150
151 std::pair<IdT, bool> insert(T &&Elem) {
152 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
153 return {It->second, false};
154 }
155 auto Ret = Id(FromInt.size());
156 auto *Ins = &FromInt.emplace_back(std::move(Elem));
157 ToInt[Ins] = Ret;
158 return {Ret, true};
159 }
160
161 /// Returns the index of the given element in the compressors storage. If
162 /// the element isn't present, std::nullopt will be returned
163 [[nodiscard]] std::optional<IdT> getOrNull(const T &Elem) const {
164 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
165 return It->second;
166 }
167 return std::nullopt;
168 }
169
170 [[nodiscard]] IdT get(const T &Elem) const {
171 auto It = ToInt.find(&Elem);
172 assert(It != ToInt.end());
173 return It->second;
174 }
175
176 [[nodiscard]] bool inbounds(IdT Idx) const noexcept {
177 return size_t(Idx) < FromInt.size();
178 }
179
180 [[nodiscard]] const T &operator[](IdT Idx) const noexcept {
181 assert(inbounds(Idx));
182 return FromInt[size_t(Idx)];
183 }
184
185 [[nodiscard]] size_t size() const noexcept { return FromInt.size(); }
186 [[nodiscard]] size_t capacity() const noexcept {
187 return FromInt.size() +
188 ToInt.getMemorySize() / sizeof(typename decltype(ToInt)::value_type);
189 }
190
191 auto begin() const noexcept { return FromInt.begin(); }
192 auto end() const noexcept { return FromInt.end(); }
193
194 [[nodiscard]] auto enumerate() const noexcept {
195 return llvm::map_range(llvm::enumerate(FromInt),
196 [](const auto &IndexAndVal) {
197 return std::pair<IdT, const T &>{
198 IdT(IndexAndVal.index()), IndexAndVal.value()};
199 });
200 }
201
202 void clear() noexcept {
203 ToInt.clear();
204 FromInt.clear();
205 }
206
207private:
208 struct DSI : llvm::DenseMapInfo<const T *> {
209 static auto getHashValue(const T *Elem) noexcept {
210 assert(Elem != nullptr);
211 if constexpr (has_llvm_dense_map_info<T>) {
212 return llvm::DenseMapInfo<T>::getHashValue(*Elem);
213 } else {
214 return std::hash<T>{}(*Elem);
215 }
216 }
217 static auto isEqual(const T *LHS, const T *RHS) noexcept {
218 if (LHS == RHS) {
219 return true;
220 }
221 if (LHS == DSI::getEmptyKey() || LHS == DSI::getTombstoneKey() ||
222 RHS == DSI::getEmptyKey() || RHS == DSI::getTombstoneKey()) {
223 return false;
224 }
225 if constexpr (has_llvm_dense_map_info<T>) {
226 return llvm::DenseMapInfo<T>::isEqual(*LHS, *RHS);
227 } else {
228 return *LHS == *RHS;
229 }
230 }
231 };
232
233 std::deque<T> FromInt;
234 llvm::DenseMap<const T *, IdT, DSI> ToInt;
235};
