29/// \brief A utility class that assigns a sequential Id to every inserted
30/// object.
31///
32/// This specialization handles types that can be efficiently passed by value
33template <typename T, typename IdT>
34 requires CanEfficientlyPassByValue<T>
35class Compressor<T, IdT> {
36public:
37 void reserve(size_t Capacity) {
38 assert(Capacity <= UINT32_MAX);
39 ToInt.reserve(Capacity);
40 FromInt.reserve(Capacity);
41 }
42
43 IdT getOrInsert(T Elem) {
44 auto [It, Inserted] = ToInt.try_emplace(Elem, IdT(ToInt.size()));
45 if (Inserted) {
46 FromInt.push_back(Elem);
47 }
48 return It->second;
49 }
50
51 std::pair<IdT, bool> insert(T Elem) {
52 auto [It, Inserted] = ToInt.try_emplace(Elem, IdT(ToInt.size()));
53 if (Inserted) {
54 FromInt.push_back(Elem);
55 }
56 return {It->second, Inserted};
57 }
58
59 [[nodiscard]]
60 std::optional<IdT> getOrNull(T Elem) const {
61 if (auto It = ToInt.find(Elem); It != ToInt.end()) {
62 return It->second;
63 }
64 return std::nullopt;
65 }
66
67 [[nodiscard]] IdT get(T Elem) const {
68 auto It = ToInt.find(Elem);
69 assert(It != ToInt.end());
70 return It->second;
71 }
72
73 [[nodiscard]] bool inbounds(IdT Idx) const noexcept {
74 return FromInt.inbounds(Idx);
75 }
76
77 [[nodiscard]] T operator[](IdT Idx) const noexcept {
78 assert(inbounds(Idx));
79 return FromInt[Idx];
80 }
81
82 [[nodiscard]] size_t size() const noexcept { return FromInt.size(); }
83 [[nodiscard]] size_t capacity() const noexcept {
84 return FromInt.capacity() +
85 ToInt.getMemorySize() / sizeof(typename decltype(ToInt)::value_type);
86 }
87
88 [[nodiscard]] auto begin() const noexcept { return FromInt.begin(); }
89 [[nodiscard]] auto end() const noexcept { return FromInt.end(); }
90
91 [[nodiscard]] auto enumerate() const noexcept { return FromInt.enumerate(); }
92
93 void clear() noexcept {
94 ToInt.clear();
95 FromInt.clear();
96 }
97
98private:
99 llvm::DenseMap<T, IdT> ToInt;
100 TypedVector<IdT, T, 0> FromInt;
101};

103/// \brief A utility class that assigns a sequential Id to every inserted
104/// object.
105///
106/// This specialization handles types that cannot be efficiently passed by value
107template <typename T, typename IdT>
108 requires(!CanEfficientlyPassByValue<T>)
109class Compressor<T, IdT> {
110public:
111 void reserve(size_t Capacity) {
112 assert(Capacity <= UINT32_MAX);
113 ToInt.reserve(Capacity);
114 }
115
116 /// Returns the index of the given element in the compressors storage. If
117 /// the element isn't present yet, it will be added first and its index will
118 /// then be returned.
119 IdT getOrInsert(const T &Elem) {
120 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
121 return It->second;
122 }
123 auto Ret = IdT(FromInt.size());
124 auto *Ins = &FromInt.emplace_back(Elem);
125 ToInt[Ins] = Ret;
126 return Ret;
127 }
128
129 /// Returns the index of the given element in the compressors storage. If
130 /// the element isn't present yet, it will be added first and its index will
131 /// then be returned.
132 IdT getOrInsert(T &&Elem) {
133 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
134 return It->second;
135 }
136 auto Ret = IdT(FromInt.size());
137 auto *Ins = &FromInt.emplace_back(std::move(Elem));
138 ToInt[Ins] = Ret;
139 return Ret;
140 }
141
142 std::pair<IdT, bool> insert(const T &Elem) {
143 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
144 return {It->second, false};
145 }
146 auto Ret = Id(FromInt.size());
147 auto *Ins = &FromInt.emplace_back(Elem);
148 ToInt[Ins] = Ret;
149 return {Ret, true};
150 }
151
152 std::pair<IdT, bool> insert(T &&Elem) {
153 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
154 return {It->second, false};
155 }
156 auto Ret = Id(FromInt.size());
157 auto *Ins = &FromInt.emplace_back(std::move(Elem));
158 ToInt[Ins] = Ret;
159 return {Ret, true};
160 }
161
162 /// Returns the index of the given element in the compressors storage. If
163 /// the element isn't present, std::nullopt will be returned
164 [[nodiscard]] std::optional<IdT> getOrNull(const T &Elem) const {
165 if (auto It = ToInt.find(&Elem); It != ToInt.end()) {
166 return It->second;
167 }
168 return std::nullopt;
169 }
170
171 [[nodiscard]] IdT get(const T &Elem) const {
172 auto It = ToInt.find(&Elem);
173 assert(It != ToInt.end());
174 return It->second;
175 }
176
177 [[nodiscard]] bool inbounds(IdT Idx) const noexcept {
178 return size_t(Idx) < FromInt.size();
179 }
180
181 [[nodiscard]] const T &operator[](IdT Idx) const noexcept {
182 assert(inbounds(Idx));
183 return FromInt[size_t(Idx)];
184 }
185
186 [[nodiscard]] size_t size() const noexcept { return FromInt.size(); }
187 [[nodiscard]] size_t capacity() const noexcept {
188 return FromInt.size() +
189 ToInt.getMemorySize() / sizeof(typename decltype(ToInt)::value_type);
190 }
191
192 auto begin() const noexcept { return FromInt.begin(); }
193 auto end() const noexcept { return FromInt.end(); }
194
195 [[nodiscard]] auto enumerate() const noexcept {
196 return llvm::map_range(llvm::enumerate(FromInt),
197 [](const auto &IndexAndVal) {
198 return std::pair<IdT, const T &>{
199 IdT(IndexAndVal.index()), IndexAndVal.value()};
200 });
201 }
202
203 void clear() noexcept {
204 ToInt.clear();
205 FromInt.clear();
206 }
207
208private:
209 struct DSI : llvm::DenseMapInfo<const T *> {
210 static auto getHashValue(const T *Elem) noexcept {
211 assert(Elem != nullptr);
212 if constexpr (has_llvm_dense_map_info<T>) {
213 return llvm::DenseMapInfo<T>::getHashValue(*Elem);
214 } else {
215 return std::hash<T>{}(*Elem);
216 }
217 }
218 static auto isEqual(const T *LHS, const T *RHS) noexcept {
219 if (LHS == RHS) {
220 return true;
221 }
222 if (LHS == DSI::getEmptyKey() || LHS == DSI::getTombstoneKey() ||
223 RHS == DSI::getEmptyKey() || RHS == DSI::getTombstoneKey()) {
224 return false;
225 }
226 if constexpr (has_llvm_dense_map_info<T>) {
227 return llvm::DenseMapInfo<T>::isEqual(*LHS, *RHS);
228 } else {
229 return *LHS == *RHS;
230 }
231 }
232 };
233
234 std::deque<T> FromInt;
235 llvm::DenseMap<const T *, IdT, DSI> ToInt;
236};
