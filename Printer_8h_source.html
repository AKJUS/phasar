29template <typename T>
30PSR_CONCEPT IsSomehowPrintable =
31 has_str_v<T> || is_llvm_printable_v<T> || has_adl_to_string_v<T>;
32
33template <typename T> decltype(auto) printSomehow(const T &Val) {
34 if constexpr (has_str_v<T>) {
35 return Val.str();
36 } else if constexpr (has_adl_to_string_v<T>) {
37 return adl_to_string(Val);
38 } else if constexpr (is_llvm_printable_v<T>) {
39 std::string Str;
40 llvm::raw_string_ostream ROS(Str);
41 ROS << Val;
42 return Str;
43 } else {
44 llvm_unreachable(
45 "All compilable cases should be handled in the if-chain above");
46 }
47}
48} // namespace detail
