112 /// Returns the data-flow results at the given statement while respecting
113 /// LLVM's SSA semantics.
114 ///
115 /// An example: when a value is loaded and the location loaded from, here
116 /// variable 'i', is a data-flow fact that holds, then the loaded value '%0'
117 /// will usually be generated and also holds. However, due to the underlying
118 /// theory (and respective implementation) this load instruction causes the
119 /// loaded value to be generated and thus, it will be valid only AFTER the
120 /// load instruction, i.e., at the successor instruction.
121 ///
122 /// %0 = load i32, i32* %i, align 4
123 ///
124 /// This result accessor function returns the results at the successor
125 /// instruction(s) reflecting that the expression on the left-hand side holds
126 /// if the expression on the right-hand side holds.
127 [[nodiscard]] std::unordered_map<d_t, l_t>
128 resultsAtInLLVMSSA(ByConstRef<n_t> Stmt, bool AllowOverapproximation = false,
129 bool StripZero = false) const
130 requires same_as_decay<std::remove_pointer_t<n_t>, llvm::Instruction>;

158 template <typename ICFGTy>
159 void dumpResults(const ICFGTy &ICF,
160 llvm::raw_ostream &OS = llvm::outs()) const {
161 using f_t = typename ICFGTy::f_t;
162
163 PAMM_GET_INSTANCE;
164 START_TIMER("DFA IDE Result Dumping", Full);
165 OS << "\n***************************************************************\n"
166 << "* Raw IDESolver results *\n"
167 << "***************************************************************\n";
168 auto Cells = self().Results.cellVec();
169 if (Cells.empty()) {
170 OS << "No results computed!" << '\n';
171 } else {
172 std::sort(
173 Cells.begin(), Cells.end(), [](const auto &Lhs, const auto &Rhs) {
174 if constexpr (std::is_same_v<n_t, const llvm::Instruction *>) {
175 return StringIDLess{}(getMetaDataID(Lhs.getRowKey()),
176 getMetaDataID(Rhs.getRowKey()));
177 } else {
178 // If non-LLVM IR is used
179 return Lhs.getRowKey() < Rhs.getRowKey();
180 }
181 });
182 n_t Prev = n_t{};
183 n_t Curr = n_t{};
184 f_t PrevFn = f_t{};
185 f_t CurrFn = f_t{};
186 for (unsigned I = 0; I < Cells.size(); ++I) {
187 Curr = Cells[I].getRowKey();
188 CurrFn = ICF.getFunctionOf(Curr);
189 if (PrevFn != CurrFn) {
190 PrevFn = CurrFn;
191 OS << "\n\n============ Results for function '" +
192 ICF.getFunctionName(CurrFn) + "' ============\n";
193 }
194 if (Prev != Curr) {
195 Prev = Curr;
196 std::string NString = NToString(Curr);
197 std::string Line(NString.size(), '-');
198 OS << "\n\nN: " << NString << "\n---" << Line << '\n';
199 }
200 OS << "\tD: " << DToString(Cells[I].getColumnKey())
201 << " | V: " << LToString(Cells[I].getValue()) << '\n';
202 }
203 }
204 OS << '\n';
205 STOP_TIMER("DFA IDE Result Dumping", Full);
206 }

208 template <typename HandlerFn>
209 void foreachResultEntry(HandlerFn Handler) const {
210 for (const auto &[Row, RowMap] : rowMapView()) {
211 for (const auto &[Col, Val] : RowMap) {
212 std::invoke(Handler, std::make_tuple(Row, Col, Val));
213 }
214 }
215 }
