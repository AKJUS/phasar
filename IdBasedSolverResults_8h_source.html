90 [[nodiscard]] l_t resultAt(ByConstRef<n_t> Stmt, ByConstRef<d_t> Node) const {
91 auto NodeId = Results->NodeCompressor.getOrNull(Stmt);
92 auto FactId = Results->FactCompressor.getOrNull(Node);
93
94 if (!NodeId || !FactId) {
95 return l_t{};
96 }
97
98 const auto &Entry = Results->ValTab[size_t(*NodeId)];
99 auto RetIt = Entry.find(*FactId);
100 if (RetIt == Entry.cells().end()) {
101 return l_t{};
102 }
103
104 return Results->ValCompressor[RetIt->second];
105 }

107 [[nodiscard]] std::unordered_map<d_t, l_t>
108 resultsAt(ByConstRef<n_t> Stmt, bool StripZero = false) const {
109 auto NodeId = Results->NodeCompressor.getOrNull(Stmt);
110 if (!NodeId) {
111 return {};
112 }
113
114 std::unordered_map<d_t, l_t> Result;
115 Result.reserve(Results->ValTab[size_t(*NodeId)].size());
116 for (auto [Fact, Value] : Results->ValTab[size_t(*NodeId)].cells()) {
117 /// In the IterativeIDESolver, we have made sure that the zero flow-fact
118 /// always has the Id 0
119 if (StripZero && Fact == 0) {
120 continue;
121 }
122 Result.try_emplace(Results->FactCompressor[Fact],
123 Results->ValCompressor[Value]);
124 }
125
126 return Result;
127 }

129 [[nodiscard]] std::set<d_t> ifdsResultsAt(ByConstRef<n_t> Stmt) const {

131 if (!NodeId) {

135 std::set<d_t> Result;
136 for (auto [Fact, Unused] : Results->ValTab[size_t(*NodeId)]) {
