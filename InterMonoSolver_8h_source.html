30/// \brief A solver class for interprocedual monotone problems (derived from
31/// InterMonoProblem). To solve the problem, call solve().
32///
33/// \tparam AnalysisDomainTy type of the analysis domain.
34/// \tparam K An unsigned integer used as the maximum length for call-string
35/// contexts.
36template <typename AnalysisDomainTy, unsigned K> class InterMonoSolver {
37public:
38 using ProblemTy = InterMonoProblem<AnalysisDomainTy>;
39 using n_t = typename AnalysisDomainTy::n_t;
40 using d_t = typename AnalysisDomainTy::d_t;
41 using f_t = typename AnalysisDomainTy::f_t;
42 using t_t = typename AnalysisDomainTy::t_t;
43 using v_t = typename AnalysisDomainTy::v_t;
44 using i_t = typename AnalysisDomainTy::i_t;
45 using mono_container_t = typename AnalysisDomainTy::mono_container_t;

47protected:
48 ProblemTy &IMProblem;
49 std::deque<std::pair<n_t, n_t>> Worklist;
50 std::unordered_map<
51 n_t, std::unordered_map<CallStringCTX<n_t, K>, mono_container_t>>
52 Analysis;
53 std::unordered_set<f_t> AddedFunctions;
54 const i_t *ICF;

56 void initialize() {
57 for (auto &[Node, FlowFacts] : IMProblem.initialSeeds()) {
58 auto ControlFlowEdges =
59 ICF->getAllControlFlowEdges(ICF->getFunctionOf(Node));
60 Worklist.insert(Worklist.begin(), ControlFlowEdges.begin(),
61 ControlFlowEdges.end());
62 // Initialize with empty context and empty data-flow set such that the
63 // flow functions are at least called once per instruction
64 for (auto &[Src, Dst] : ControlFlowEdges) {
65 Analysis[Src][CallStringCTX<n_t, K>()] = IMProblem.allTop();
66 }
67 // Initialize last
68 if (!ControlFlowEdges.empty()) {
69 Analysis[ControlFlowEdges.back().second][CallStringCTX<n_t, K>()] =
70 IMProblem.allTop();
71 }
72 // Additionally, insert the initial seeds
73 Analysis[Node][CallStringCTX<n_t, K>()].insert(FlowFacts.begin(),
74 FlowFacts.end());
75 }
76 }

78 bool isIntraEdge(std::pair<n_t, n_t> Edge) {
79 return ICF->getFunctionOf(Edge.first) == ICF->getFunctionOf(Edge.second);
80 }
81
82 bool isCallEdge(std::pair<n_t, n_t> Edge) {
83 return !isIntraEdge(Edge) && ICF->isCallSite(Edge.first);
84 }
85
86 bool isReturnEdge(std::pair<n_t, n_t> Edge) {
87 return !isIntraEdge(Edge) && ICF->isExitInst(Edge.first);
88 }

98 void addCalleesToWorklist(std::pair<n_t, n_t> Edge) {
99 auto Src = Edge.first;
100 // auto Dst = Edge.second;
101 // Add inter- and intra-edges of callee(s)
102 for (auto Callee : ICF->getCalleesOfCallAt(Src)) {
103 if (AddedFunctions.find(Callee) != AddedFunctions.end()) {
104 break;
105 }
106 AddedFunctions.insert(Callee);
107 // Add call Edge(s)
108 for (auto StartPoint : ICF->getStartPointsOf(Callee)) {
109 Worklist.push_back({Src, StartPoint});
110 }
111 // Add intra edges of callee
112 auto Edges = ICF->getAllControlFlowEdges(Callee);
113 Worklist.insert(Worklist.begin(), Edges.begin(), Edges.end());
114 // Initialize with empty context and empty data-flow set such that the
115 // flow functions are at least called once per instruction
116 for (auto &[Src, Dst] : Edges) {
117 Analysis[Src][CallStringCTX<n_t, K>()] = IMProblem.allTop();
118 }
119 // Initialize last
120 if (!Edges.empty()) {
121 Analysis[Edges.back().second][CallStringCTX<n_t, K>()] =
122 IMProblem.allTop();
123 }
124 // Add return Edge(s)
125 for (auto Ret : ICF->getExitPointsOf(Callee)) {
126 for (auto RetSite : ICF->getReturnSitesOfCallAt(Src)) {
127 Worklist.push_back({Ret, RetSite});
128 }
129 }
130 }
131 // The (intra-procedural) call-to-return Edge of the caller is already in
132 // the worklist
133 }
134
135 void addToWorklist(std::pair<n_t, n_t> Edge) {
136 auto Src = Edge.first;
137 auto Dst = Edge.second;
138 Worklist.push_back({Src, Dst});
139 // add intra-procedural edges again
140 for (auto Nprimeprime : ICF->getSuccsOf(Dst)) {
141 Worklist.push_back({Dst, Nprimeprime});
142 }
143 // add inter-procedural call edges again
144 if (ICF->isCallSite(Dst)) {
145 for (auto Callee : ICF->getCalleesOfCallAt(Dst)) {
146 for (auto StartPoint : ICF->getStartPointsOf(Callee)) {
147 Worklist.push_back({Dst, StartPoint});
148 }
149 }
150 }
151 // add inter-procedural return edges again
152 if (ICF->isExitInst(Dst)) {
153 for (const auto *Caller : ICF->getCallersOf(ICF->getFunctionOf(Dst))) {
154 for (const auto *Nprimeprime : ICF->getSuccsOf(Caller)) {
155 Worklist.push_back({Dst, Nprimeprime});
156 }
157 }
158 }
159 }

181 void processNormal(std::pair<n_t, n_t> Edge) {
182 llvm::outs() << "Handle normal flow\n";
183 auto Src = Edge.first;
184 auto Dst = Edge.second;
185 llvm::outs() << "Src: " << NToString(Src) << '\n';
186 llvm::outs() << "Dst: " << NToString(Dst) << '\n';
187 std::unordered_map<CallStringCTX<n_t, K>, mono_container_t> Out;
188 for (auto &[Ctx, Facts] : Analysis[Src]) {
189 Out[Ctx] = IMProblem.normalFlow(Src, Analysis[Src][Ctx]);
190 // need to merge if Dst is a branch target
191 if (ICF->isBranchTarget(Src, Dst)) {
192 llvm::outs() << "Num preds: " << ICF->getPredsOf(Dst).size() << '\n';
193 for (auto Pred : ICF->getPredsOf(Dst)) {
194 if (Pred != Src) {
195 // we need to compute the out set of Pred and merge it with the
196 // out set of Src on-the-fly as we do not have a dedicated
197 // storage for merge points (otherwise we run into trouble with
198 // merge operator such as set union)
199 auto OtherPredOut = IMProblem.normalFlow(Pred, Analysis[Pred][Ctx]);
200 Out[Ctx] = IMProblem.merge(Out[Ctx], OtherPredOut);
201 }
202 }
203 }
204 // Check if data-flow facts have changed and if so, add Edge(s) to
205 // worklist again.
206 llvm::outs() << "\nNormal Out[Ctx]:\n";
207 IMProblem.printContainer(llvm::outs(), Out[Ctx]);
208 llvm::outs() << "\nAnalysis[Dst][Ctx]:\n";
209 IMProblem.printContainer(llvm::outs(), Analysis[Dst][Ctx]);
210 bool FlowFactStabilized =
211 IMProblem.equal_to(Out[Ctx], Analysis[Dst][Ctx]);
212 if (!FlowFactStabilized) {
213 llvm::outs() << "\nNormal stabilized? --> " << FlowFactStabilized
214 << '\n';
215 auto Merged = Out[Ctx];
216 llvm::outs() << "Normal merged:\n";
217 IMProblem.printContainer(llvm::outs(), Merged);
218 llvm::outs() << '\n';
219 Analysis[Dst][Ctx] = Merged;
220 addToWorklist({Src, Dst});
221 }
222 }
223 }

225 void processCall(std::pair<n_t, n_t> Edge) {
226 auto Src = Edge.first;
227 auto Dst = Edge.second;
228 std::unordered_map<CallStringCTX<n_t, K>, mono_container_t> Out;
229 if (!isIntraEdge(Edge)) {
230 llvm::outs() << "Handle call flow\n";
231 llvm::outs() << "Src: " << NToString(Src) << '\n';
232 llvm::outs() << "Dst: " << NToString(Dst) << '\n';
233 for (auto &[Ctx, Facts] : Analysis[Src]) {
234 auto CTXAdd(Ctx);
235 CTXAdd.push_back(Src);
236 Out[CTXAdd] = IMProblem.callFlow(Src, ICF->getFunctionOf(Dst),
237 Analysis[Src][Ctx]);
238 bool FlowFactStabilized =
239 IMProblem.equal_to(Out[CTXAdd], Analysis[Dst][CTXAdd]);
240 llvm::outs() << "Call Out[CTXAdd]:\n";
241 IMProblem.printContainer(llvm::outs(), Out[CTXAdd]);
242 llvm::outs() << '\n';
243 llvm::outs() << "Call Analysis[Dst][CTXAdd]:\n";
244 IMProblem.printContainer(llvm::outs(), Analysis[Dst][CTXAdd]);
245 llvm::outs() << '\n';
246 llvm::outs() << "Call stabilized? --> " << FlowFactStabilized << '\n';
247 if (!FlowFactStabilized) {
248 // auto merge = IMProblem.merge(Analysis[Dst][CTXAdd], Out[CTXAdd]);
249 auto Merge = Out[CTXAdd];
250 llvm::outs() << "Call merge:\n";
251 IMProblem.printContainer(llvm::outs(), Merge);
252 llvm::outs() << '\n';
253 Analysis[Dst][CTXAdd] = Merge;
254 addToWorklist({Src, Dst});
255 }
256 }
257 } else {
258 // Handle call-to-ret flow
259 llvm::outs() << "Handle call to ret flow\n";
260 llvm::outs() << "Src: " << NToString(Src) << '\n';
261 llvm::outs() << "Dst: " << NToString(Dst) << '\n';
262 for (auto &[Ctx, Facts] : Analysis[Src]) {
263 // call-to-ret flow does not modify contexts
264 Out[Ctx] = IMProblem.callToRetFlow(
265 Src, Dst, ICF->getCalleesOfCallAt(Src), Analysis[Src][Ctx]);
266 bool FlowFactStabilized =
267 IMProblem.equal_to(Out[Ctx], Analysis[Dst][Ctx]);
268 llvm::outs() << "Call to ret stabilized? --> " << FlowFactStabilized
269 << '\n';
270 llvm::outs() << "Call Out[Ctx]:\n";
271 IMProblem.printContainer(llvm::outs(), Out[Ctx]);
272 llvm::outs() << '\n';
273 llvm::outs() << "Call Analysis[Dst][CTX]:\n";
274 IMProblem.printContainer(llvm::outs(), Analysis[Dst][Ctx]);
275 llvm::outs() << '\n';
276 if (!FlowFactStabilized) {
277 auto Merge = Out[Ctx];
278 llvm::outs() << "Call to ret merge:\n";
279 IMProblem.printContainer(llvm::outs(), Merge);
280 llvm::outs() << '\n';
281 Analysis[Dst][Ctx] =
282 Merge; // IMProblem.merge(Analysis[Dst][Ctx], Out[Ctx]);
283 addToWorklist({Src, Dst});
284 }
285 }
286 }

289 void processExit(std::pair<n_t, n_t> Edge) {
290 auto Src = Edge.first;
291 auto Dst = Edge.second;
292 std::unordered_map<CallStringCTX<n_t, K>, mono_container_t> Out;
293 llvm::outs() << "\nHandle ret flow in: "
294 << ICF->getFunctionName(ICF->getFunctionOf(Src)) << '\n';
295 llvm::outs() << "Src: " << NToString(Src) << '\n';
296 llvm::outs() << "Dst: " << NToString(Dst) << '\n';
297 for (auto &[Ctx, Facts] : Analysis[Src]) {
298 auto CTXRm(Ctx);
299 CTXRm.print(llvm::outs() << "CTXRm: ") << '\n';
300 // we need to use several call- and retsites if the context is empty
301 llvm::SmallVector<n_t> CallSites;
302
303 // handle empty context
304 if (Ctx.empty()) {
305 const auto &Callers = ICF->getCallersOf(ICF->getFunctionOf(Src));
306 CallSites.append(Callers.begin(), Callers.end());
307 } else {
308 // handle context containing at least one element
309 CallSites.push_back(CTXRm.pop_back());
310 }
311
312 std::set<n_t> RetSites;
313 // retrieve the possible return sites for each call
314 for (auto CallSite : CallSites) {
315 auto RetSitesPerCall = ICF->getReturnSitesOfCallAt(CallSite);
316 RetSites.insert(RetSitesPerCall.begin(), RetSitesPerCall.end());
317 }
318 for (auto CallSite : CallSites) {
319 auto RetFactsPerCall = IMProblem.returnFlow(
320 CallSite, ICF->getFunctionOf(Src), Src, Dst, Analysis[Src][Ctx]);
321 Out[CTXRm].insert(RetFactsPerCall.begin(), RetFactsPerCall.end());
322 }
323 // TODO!
324 llvm::outs() << "ResSites.size(): " << RetSites.size() << '\n';
325 for (auto RetSite : RetSites) {
326 llvm::outs() << "RetSite: " << NToString(RetSite) << '\n';
327 llvm::outs() << "Return facts: ";
328 IMProblem.printContainer(llvm::outs(), Out[CTXRm]);
329 llvm::outs() << '\n';
330 llvm::outs() << "RetSite facts: ";
331 IMProblem.printContainer(llvm::outs(), Analysis[RetSite][CTXRm]);
332 llvm::outs() << '\n';
333 bool FlowFactStabilized =
334 IMProblem.equal_to(Out[CTXRm], Analysis[RetSite][CTXRm]);
335 llvm::outs() << "Ret stabilized? --> " << FlowFactStabilized << '\n';
336 if (!FlowFactStabilized) {
337 mono_container_t Merge;
338 Merge.insert(Analysis[RetSite][CTXRm].begin(),
339 Analysis[RetSite][CTXRm].end());
340 Merge.insert(Out[CTXRm].begin(), Out[CTXRm].end());
341 Analysis[RetSite][CTXRm] = Merge;
342 Analysis[Dst][CTXRm] = Merge;
343 // IMProblem.merge(Analysis[RetSite][CTXRm], Out[CTXRm]);
344 llvm::outs() << "Merged to: ";
345 IMProblem.printContainer(llvm::outs(), Merge);
346 llvm::outs() << '\n';
347 // addToWorklist({Src, RetSite});
348 }
349 }
350 }

364 virtual void solve() {
365 initialize();
366 while (!Worklist.empty()) {
367 std::pair<n_t, n_t> Edge = Worklist.front();
368 Worklist.pop_front();
369 auto Src = Edge.first;
370 // auto Dst = Edge.second;
371 if (ICF->isCallSite(Src)) {
372 addCalleesToWorklist(Edge);
373 }
374 // Compute the data-flow facts using the respective kind of flows
375 if (ICF->isCallSite(Src)) {
376 // Handle call flow(s) and call-to-return flow
377 processCall(Edge);
378 } else if (ICF->isExitInst(Src)) {
379 // Handle return flow
380 processExit(Edge);
381 } else {
382 // Handle normal flow
383 processNormal(Edge);
384 }
385 }
386 }
