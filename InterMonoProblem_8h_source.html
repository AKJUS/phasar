31template <typename T>
32concept InterMonoAnalysisDomain = MonoAnalysisDomain<T> && requires() {
33 typename T::i_t;
34 requires ICFG<typename T::i_t>;
35};

37/// \brief The analysis problem interface for interprocedural monotone problems
38/// (solvable by the InterMonoSolver). Create a subclass from this and override
39/// all pure-virtual functions to create your own inter-mono analysis.
40template <InterMonoAnalysisDomain AnalysisDomainTy>
41class InterMonoProblem : public IntraMonoProblem<AnalysisDomainTy> {
42public:
43 using n_t = typename AnalysisDomainTy::n_t;
44 using d_t = typename AnalysisDomainTy::d_t;
45 using f_t = typename AnalysisDomainTy::f_t;
46 using t_t = typename AnalysisDomainTy::t_t;
47 using v_t = typename AnalysisDomainTy::v_t;
48 using i_t = typename AnalysisDomainTy::i_t;
49 using db_t = typename AnalysisDomainTy::db_t;
50 using mono_container_t = typename AnalysisDomainTy::mono_container_t;
51
52protected:
53 const i_t *ICF;
54
55public:
56 /// An interprocedural monotone problem generated from an intermediate
57 /// representation, a type hierarchy of said representation, a control flow
58 /// graph, points-to information and optionally a vector of entry points.
59 /// @param[in] IRDB A project IR database.
60 /// @param[in] TH A type hierarchy based on the given IRDB.
61 /// @param[in] CF A control flow graph based on the given IRDB.
62 /// @param[in] PT Points-to information based on the given IRDB.
63 /// @param[in] EntryPoints A vector of entry points. Provide at least one.
64 InterMonoProblem(const db_t *IRDB, const i_t *ICF, AliasInfoRef<v_t, n_t> PT,
65 std::vector<std::string> EntryPoints = {})
66 : IntraMonoProblem<AnalysisDomainTy>(IRDB, ICF, PT, EntryPoints),
67 ICF(ICF) {}
68
69 ~InterMonoProblem() override = default;
70 InterMonoProblem(const InterMonoProblem &Other) = delete;
71 InterMonoProblem(InterMonoProblem &&Other) = delete;
72 InterMonoProblem &operator=(const InterMonoProblem &Other) = delete;
73 InterMonoProblem &operator=(InterMonoProblem &&Other) = delete;
74
75 virtual mono_container_t callFlow(n_t CallSite, f_t Callee,
76 const mono_container_t &In) = 0;
77
78 virtual mono_container_t returnFlow(n_t CallSite, f_t Callee, n_t ExitStmt,
79 n_t RetSite,
80 const mono_container_t &In) = 0;
81
82 virtual mono_container_t callToRetFlow(n_t CallSite, n_t RetSite,
83 llvm::ArrayRef<f_t> Callees,
84 const mono_container_t &In) = 0;
85
86 [[nodiscard]] const i_t *getICFG() const { return ICF; }
87};
