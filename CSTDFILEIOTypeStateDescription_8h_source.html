38template <> struct JoinLatticeTraits<CSTDFILEIOState> {

45 static constexpr CSTDFILEIOState join(CSTDFILEIOState L,
46 CSTDFILEIOState R) noexcept {
47 if (L == top() || R == bottom()) {
