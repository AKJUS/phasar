28class EdgeDomain final {
29public:
30 enum Kind {
31 Bot,
32 Top,
33 Sanitized,
34 WithSanitizer,
35 };
36
37private:
38 llvm::PointerIntPair<const llvm::Instruction *, 2, Kind> Value;
39
40public:
41 inline explicit EdgeDomain() noexcept : Value(nullptr, Kind::Bot) {}
42
43 inline EdgeDomain(std::nullptr_t) noexcept
44 : Value(nullptr, Kind::WithSanitizer) {}
45
46 inline EdgeDomain(psr::Bottom /*unused*/) noexcept
47 : Value(nullptr, Kind::Bot) {}
48
49 inline EdgeDomain(psr::Top /*unused*/) noexcept : Value(nullptr, Kind::Top) {}
50
51 inline EdgeDomain(psr::XTaint::Sanitized /*unused*/) noexcept
52 : Value(nullptr, Kind::Sanitized) {}
53
54 inline EdgeDomain(const llvm::Instruction *Sani) noexcept
55 : Value(Sani, Kind::WithSanitizer) {}
56
57 [[nodiscard]] inline bool isBottom() const { return getKind() == Bot; }
58 [[nodiscard]] inline bool isTop() const { return getKind() == Top; }
59 [[nodiscard]] inline bool isSanitized() const {
60 return getKind() == Sanitized;
61 }
62 [[nodiscard]] inline bool isNotSanitized() const {
63 return getKind() == WithSanitizer && getSanitizer() == nullptr;
64 }
65 [[nodiscard]] inline bool hasSanitizer() const {
66 return getSanitizer() != nullptr;
67 }
68 [[nodiscard]] inline const llvm::Instruction *getSanitizer() const {
69 return Value.getPointer();
70 }
71 [[nodiscard]] inline bool mayBeSanitized() const {
72 return isSanitized() || hasSanitizer();
73 }
74
75 [[nodiscard]] inline bool operator==(const EdgeDomain &Other) const {
76 return Other.Value == Value;
77 }
78 [[nodiscard]] inline bool operator!=(const EdgeDomain &Other) const {
79 return !(*this == Other);
80 }
81
82 /// An arbitrary ordering to be able to insert EdgeDomain values into std::set
83 /// and as keys into std::map
84 [[nodiscard]] inline bool operator<(const EdgeDomain &Other) const {
85 return Other.Value.getOpaqueValue() < Value.getOpaqueValue();
86 }
87 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
88 const EdgeDomain &ED);
89 EdgeDomain join(const EdgeDomain &Other,
90 BasicBlockOrdering *BBO = nullptr) const;
91
92 [[nodiscard]] inline Kind getKind() const { return Value.getInt(); }
93};
94} // namespace psr::XTaint
