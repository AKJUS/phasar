45enum class TAGNodeId : uint32_t {};

47struct Variable {
48 const llvm::Value *Val;
49};
50
51struct Field {
52 const llvm::DIType *Base;
53 size_t ByteOffset;
54};
55
56struct Return {
57 const llvm::Function *Fun;
58};
59
60struct TAGNode {
61 std::variant<Variable, Field, Return> Label;
62};
63
64constexpr bool operator==(Variable L, Variable R) noexcept {
65 return L.Val == R.Val;
66}
67constexpr bool operator==(Field L, Field R) noexcept {
68 return L.Base == R.Base && L.ByteOffset == R.ByteOffset;
69}
70constexpr bool operator==(Return L, Return R) noexcept {
71 return L.Fun == R.Fun;
72}
73constexpr bool operator==(TAGNode L, TAGNode R) noexcept {
74 return L.Label == R.Label;
75}
76}; // namespace psr::vta

79template <> struct DenseMapInfo<psr::vta::TAGNode> {
80 using TAGNode = psr::vta::TAGNode;
81 using Variable = psr::vta::Variable;
82 using Field = psr::vta::Field;
83 using Return = psr::vta::Return;
84
85 inline static TAGNode getEmptyKey() noexcept {
86 return {Variable{llvm::DenseMapInfo<const llvm::Value *>::getEmptyKey()}};
87 }
88 inline static TAGNode getTombstoneKey() noexcept {
89 return {
90 Variable{llvm::DenseMapInfo<const llvm::Value *>::getTombstoneKey()}};
91 }
92 inline static bool isEqual(TAGNode L, TAGNode R) noexcept { return L == R; }
93 inline static auto getHashValue(TAGNode TN) noexcept {
94 if (const auto *Var = std::get_if<Variable>(&TN.Label)) {
95 return llvm::hash_combine(0, Var->Val);
96 }
97 if (const auto *Fld = std::get_if<Field>(&TN.Label)) {
98 return llvm::hash_combine(1, Fld->Base, Fld->ByteOffset);
99 }
100 if (const auto *Ret = std::get_if<Return>(&TN.Label)) {
101 return llvm::hash_combine(2, Ret->Fun);
102 }
103 llvm_unreachable("All TAGNode variants should be handled already");
104 }
105};

125struct TypeAssignmentGraph {
126 using GraphNodeId = TAGNodeId;
127 using TypeInfoTy =
128 llvm::PointerUnion<const llvm::Function *, const llvm::DIType *>;
129 Compressor<TAGNode, GraphNodeId> Nodes;
130
131 TypedVector<TAGNodeId, llvm::SmallDenseSet<TAGNodeId>> Adj;
132 llvm::SmallDenseMap<TAGNodeId, llvm::SmallDenseSet<TypeInfoTy>>
133 TypeEntryPoints;
134
135 [[nodiscard]] inline std::optional<TAGNodeId> get(TAGNode TN) const noexcept {
136 return Nodes.getOrNull(TN);
137 }
138
139 [[nodiscard]] inline TAGNode operator[](TAGNodeId Id) const noexcept {
140 return Nodes[Id];
141 }
142
143 inline void addEdge(TAGNodeId From, TAGNodeId To) {
144 assert(size_t(From) < Adj.size());
145 assert(size_t(To) < Adj.size());
146
147 if (From == To) {
148 return;
149 }
150
151 Adj[From].insert(To);
152 }
153
154 void print(llvm::raw_ostream &OS);
155};
