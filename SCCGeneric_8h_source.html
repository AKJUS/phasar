35struct SCCIdBase {
36 uint32_t Value{};
37
38 constexpr SCCIdBase() noexcept = default;
39
40 explicit constexpr SCCIdBase(uint32_t Val) noexcept : Value(Val) {}
41
42 explicit constexpr operator uint32_t() const noexcept { return Value; }
43 template <typename T = size_t>
44 explicit constexpr operator size_t() const noexcept
45 requires(!std::is_same_v<uint32_t, T>)
46 {
47 return Value;
48 }
49
50 explicit constexpr operator ptrdiff_t() const noexcept {
51 return ptrdiff_t(Value);
52 }
53
54 constexpr uint32_t operator+() const noexcept { return Value; }
55
56 friend constexpr bool operator==(SCCIdBase L, SCCIdBase R) noexcept {
57 return L.Value == R.Value;
58 }
59 friend constexpr bool operator!=(SCCIdBase L, SCCIdBase R) noexcept {
60 return !(L == R);
61 }

95/// \brief Holds the SCCs of a given graph. Each SCC is assigned a unique
96/// sequential id.
97template <typename GraphNodeId> struct SCCHolder {
98 TypedVector<GraphNodeId, SCCId<GraphNodeId>> SCCOfNode;
99 TypedVector<SCCId<GraphNodeId>, llvm::SmallVector<GraphNodeId, 1>>
100 NodesInSCC{};
101
102 /// Number of SCCs
103 [[nodiscard]] size_t size() const noexcept { return NodesInSCC.size(); }
104 [[nodiscard]] bool empty() const noexcept { return NodesInSCC.empty(); }

106 /// \brief Prints the given Graph as dot, highlighting the SCCs in the graph.
107 ///
108 /// \param Graph The graph to print
109 /// \param OS The output-stream, where to print into
110 /// \param Name The name of the graph
111 /// \param NodeToString If the graph has node-labels, convert a node-label to
112 /// string
113 template <is_const_graph G, typename NodeTransform = DefaultNodeTransform>
114 requires std::is_same_v<typename GraphTraits<G>::vertex_t, GraphNodeId>
115 void print(const G &Graph, llvm::raw_ostream &OS, llvm::StringRef Name = "",
116 NodeTransform NodeToString = {}) const {
117 OS << "digraph \"";
118 OS.write_escaped(Name) << "\" {\n";
119 psr::scope_exit CloseBrace = [&] { OS << "}\n"; };
120
121 using GTraits = psr::GraphTraits<G>;
122
123 for (const auto &[SCCId, SCC] : NodesInSCC.enumerate()) {
124 OS << " subgraph cluster_" << +SCCId
125 << "{\n node [style=filled]; color=blue; label=\"SCC " << +SCCId
126 << "\";\n";
127 psr::scope_exit CloseSCC = [&] { OS << " }\n"; };
128
129 for (auto Nod : SCC) {
130 OS << " " << size_t(Nod);
131 if constexpr (!std::is_empty_v<typename GTraits::value_type>) {
132 OS << "[label=\"";
133 OS.write_escaped(
134 std::invoke(NodeToString, GTraits::node(Graph, Nod)));
135 OS << "\"]";
136 }
137 OS << ";\n";
138 }
139 }
140
141 for (auto FromVtx : GTraits::vertices(Graph)) {
142 for (const auto &Succ : GTraits::outEdges(Graph, FromVtx)) {
143 OS << " " << size_t(FromVtx) << "->";
144 if constexpr (is_llvm_printable_v<decltype(Succ)>) {
145 // to print the edge-weight as well, if possible
146 OS << Succ;
147 } else {
148 OS << size_t(GTraits::target(Succ));
149 }
150 OS << ";\n";
151 }
152 }
153 }

230/// \brief Creates a graph based on the given input Graph, collapsing all SCCs
231/// to single nodes. The resulting graph is always a DAG, i.e., it contains no
232/// cycles
233template <typename G>
234#if __cplusplus >= 202002L
235 requires is_const_graph<G>
236#endif
237SCCDependencyGraph<typename GraphTraits<G>::vertex_t> computeSCCDependencies(
238 const G &Graph, const SCCHolder<typename GraphTraits<G>::vertex_t> &SCCs) {
239
240 using GTraits = GraphTraits<G>;
241 using GraphNodeId = typename GraphTraits<G>::vertex_t;
242
243 SCCDependencyGraph<GraphNodeId> Ret;
244 Ret.ChildrenOfSCC.resize(SCCs.size());
245
246 BitSet<SCCId<GraphNodeId>> Roots(SCCs.size(), true);
247
248 for (auto NodeId : GTraits::vertices(Graph)) {
249 auto SrcSCC = SCCs.SCCOfNode[NodeId];
250
251 for (const auto &Edge : GTraits::outEdges(Graph, NodeId)) {
252 auto Succ = GTraits::target(Edge);
253 auto SuccSCC = SCCs.SCCOfNode[Succ];
254 if (SuccSCC != SrcSCC) {
255 Ret.ChildrenOfSCC[SrcSCC].insert(SuccSCC);
256 Roots.erase(SuccSCC);
257 }
258 }
259 }
260
261 Ret.SCCRoots.reserve(Roots.size());
262 Roots.foreach ([&](auto Rt) { Ret.SCCRoots.push_back(Rt); });
263
264 return Ret;
265}

267/// \brief Computes a topological order of the nodes in the given
268/// dependency-graph.
269///
270/// Uses a simple, recursive postorder-DFS search to find a topological
271/// ordering.
272template <typename GraphNodeId>
273[[nodiscard]] SCCOrder<GraphNodeId>
274computeSCCOrder(const SCCHolder<GraphNodeId> &SCCs,
275 const SCCDependencyGraph<GraphNodeId> &Callers) {
276 SCCOrder<GraphNodeId> Ret;
277 Ret.SCCIds.reserve(SCCs.size());
278
279 BitSet<SCCId<GraphNodeId>> Seen;
280 Seen.reserve(SCCs.size());
281
282 auto Dfs = [&](auto &Dfs, SCCId<GraphNodeId> CurrSCC) -> void {
283 Seen.insert(CurrSCC);
284 for (auto Caller : Callers.ChildrenOfSCC[CurrSCC]) {
285 if (!Seen.contains(Caller)) {
286 Dfs(Dfs, Caller);
287 }
288 }
289 Ret.SCCIds.push_back(CurrSCC);
290 };
291
292 for (auto Leaf : Callers.SCCRoots) {
293 if (!Seen.contains(Leaf)) {
294 Dfs(Dfs, Leaf);
295 }
296 }
297
298 std::reverse(Ret.SCCIds.begin(), Ret.SCCIds.end());
299
300 return Ret;
301}

304/// Data for Pearce's Algorithm.
305template <typename GraphNodeId> struct Pearce4Data {
306 TypedVector<GraphNodeId, uint32_t> RIndex; // only per-vertex array
307 BitSet<GraphNodeId> Root; // root[v] in Algorithm 4
308 uint32_t Index = 1; // DFS counter
309 uint32_t C; // SCC id counter
310 llvm::SmallVector<GraphNodeId> Stack;
311
312 explicit Pearce4Data(size_t Num)
313 : RIndex(Num, 0), Root(Num), C(Num ? Num - 1 : 0) {}
314};

316// Recursive variant of Pearce's algorithm (based on Algo 3 in the paper)
317template <typename G>
318static void
319pearce4VisitRec(const G &Graph, typename GraphTraits<G>::vertex_t V,
320 Pearce4Data<typename GraphTraits<G>::vertex_t> &Data,
321 SCCHolder<typename GraphTraits<G>::vertex_t> &Holder) {
322 using GTraits = psr::GraphTraits<G>;
323 using Vertex = typename GTraits::vertex_t;
324 using SCCId = psr::SCCId<Vertex>;
325
326 bool Root = true;
327 Data.RIndex[V] = Data.Index++;
328
329 for (const auto &Edge : GTraits::outEdges(Graph, V)) {
330 auto W = GTraits::target(Edge);
331 if (Data.RIndex[W] == 0) {
332 pearce4VisitRec(Graph, W, Data, Holder);
333 }
334 if (Data.RIndex[W] < Data.RIndex[V]) {
335 Data.RIndex[V] = Data.RIndex[W];
336 Root = false;
337 }
338 }
339
340 if (Root) {
341 Data.Index--;
342 auto NewSCC = SCCId(Holder.NodesInSCC.size());
343 auto &Nodes = Holder.NodesInSCC.emplace_back();
344
345 while (!Data.Stack.empty() &&
346 Data.RIndex[V] <= Data.RIndex[Data.Stack.back()]) {
347 auto W = Data.Stack.pop_back_val();
348 Data.RIndex[W] = Data.C;
349 Data.Index--;
350
351 Holder.SCCOfNode[W] = NewSCC;
352 Nodes.push_back(W);
353 }
354 Nodes.push_back(V);
355 Holder.SCCOfNode[V] = NewSCC;
356 Data.RIndex[V] = Data.C;
357 Data.C--;
358 } else {
359 Data.Stack.push_back(V);
360 }
361}

363// Iterative variant of Pearce's algorithm (adapted from on Algo 4 in the paper)
364template <typename G>
365static void
366pearce4VisitIt(const G &Graph, typename GraphTraits<G>::vertex_t Start,
367 Pearce4Data<typename GraphTraits<G>::vertex_t> &Data,
368 SCCHolder<typename GraphTraits<G>::vertex_t> &Holder) {
369 using GTraits = psr::GraphTraits<G>;
370 using Vertex = typename GTraits::vertex_t;
371 using SCCId = psr::SCCId<Vertex>;
372
373 using OutEdgeRange =
374 decltype(GTraits::outEdges(Graph, std::declval<Vertex>()));
375 using OutEdgeIterator =
376 decltype(llvm::adl_begin(std::declval<OutEdgeRange &>()));
377 using OutEdgeSentinel =
378 decltype(llvm::adl_end(std::declval<OutEdgeRange &>()));
379
380 struct DfsFrame {
381 Vertex CurrVtx;
382 OutEdgeIterator It;
383 [[no_unique_address]] OutEdgeSentinel End;
384 };
385
386 llvm::SmallVector<DfsFrame> CallStack;
387
388 const auto PushFrames = [&](Vertex V, DfsFrame *Frame) {
389 if (Frame->It == Frame->End) {
390 return false;
391 }
392 // Recurse into children until reaching the bottom
393 do {
394 auto W = GTraits::target(*Frame->It);
395
396 if (Data.RIndex[W] != 0) {
397 // Already pushed the children of W
398 break;
399 }
400
401 Data.RIndex[W] = Data.Index++;
402 Data.Root.insert(W);
403
404 auto &&OutEdges = GTraits::outEdges(Graph, W);
405 Frame = &CallStack.emplace_back(
406 DfsFrame{W, llvm::adl_begin(OutEdges), llvm::adl_end(OutEdges)});
407 V = W;
408
409 } while (Frame->It != Frame->End);
410
411 return true;
412 };
413
414 const auto VisitLoop = [&](Vertex V, DfsFrame &Frame) {
415 // Finish visiting the current child and advance to the next child
416 if (Frame.It != Frame.End) {
417 auto W = GTraits::target(*Frame.It);
418 if (Data.RIndex[W] < Data.RIndex[V]) {
419 Data.RIndex[V] = Data.RIndex[W];
420 Data.Root.erase(V);
421 }
422
423 ++Frame.It;
424 }
425 };
426
427 const auto FinishFrame = [&](Vertex V) {
428 // finish visiting V and backtrack to the parent
429
430 if (Data.Root.contains(V)) {
431 // Found a SCC
432
433 Data.Index--;
434 auto NewSCC = SCCId(Holder.NodesInSCC.size());
435 auto &Nodes = Holder.NodesInSCC.emplace_back();
436 while (!Data.Stack.empty() &&
437 Data.RIndex[V] <= Data.RIndex[Data.Stack.back()]) {
438 auto W = Data.Stack.pop_back_val();
439 Data.RIndex[W] = Data.C;
440 Data.Index--;
441
442 Holder.SCCOfNode[W] = NewSCC;
443 Nodes.push_back(W);
444 }
445 Nodes.push_back(V);
446 Holder.SCCOfNode[V] = NewSCC;
447 Data.RIndex[V] = Data.C;
448 Data.C--;
449 } else {
450 Data.Stack.push_back(V);
451 }
452
453 CallStack.pop_back();
454 };
455
456 // Initialize the callstack by pushing the initial frame
457 Data.RIndex[Start] = Data.Index++;
458 Data.Root.insert(Start);
459 {
460 auto &&OutEdges = GTraits::outEdges(Graph, Start);
461 static_assert(
462 std::is_lvalue_reference_v<decltype(OutEdges)> ||
463 std::is_trivially_destructible_v<std::decay_t<decltype(OutEdges)>>,
464 "We assume that outEdges gives either a reference or a view into "
465 "the out-edges, but never an owning container by value. Otherwise, "
466 "the DFSFrame iterators may be dangling");
467 CallStack.emplace_back(
468 DfsFrame{Start, llvm::adl_begin(OutEdges), llvm::adl_end(OutEdges)});
469 }
470
471 // Simulate the recursion
472
473 PushFrames(Start, &CallStack.back());
474 while (true) {
475 auto &Frame = CallStack.back();
476 Vertex V = Frame.CurrVtx;
477 VisitLoop(V, Frame);
478 if (PushFrames(V, &Frame)) {
479 continue; // we don't pop from the callstack here
480 }
481
482 FinishFrame(V);
483 if (CallStack.empty()) {
484 break;
485 }
486 }
487}

491/// Compute SCCs adapted from the paper "A Space-Efficient Algorithm for Finding
492/// Strongly Connected Components", Pearce 2015, DOI:
493/// <https://doi.org/10.1016/j.ipl.2015.08.010>
494///
495/// \tparam G The graph-type
496/// \tparam Iterative Whether to use the iterative or recursive variant of the
497/// algorithm (default: true)
498/// \param Graph The graph for with to compute SCCs and topological ordering
499template <typename G, bool Iterative = true>
500[[nodiscard]] SCCHolder<typename GraphTraits<G>::vertex_t>
501computeSCCs(const G &Graph, std::bool_constant<Iterative> /*Iterative*/ = {}) {
502 using GTraits = psr::GraphTraits<G>;
503 using Vertex = typename GTraits::vertex_t;
504
505 SCCHolder<Vertex> Ret;
506 auto N = GTraits::size(Graph);
507 if (!N) {
508 return Ret;
509 }
510
511 Ret.SCCOfNode.resize(N);
512
513 detail::Pearce4Data<Vertex> Data(N);
514
515 // for all v ∈ V do if rindex[v]==0 then visit(v)
516 for (auto V : GTraits::vertices(Graph)) {
517 if (Data.RIndex[V] == 0) {
518 if constexpr (Iterative) {
519 detail::pearce4VisitIt(Graph, V, Data, Ret);
520 } else {
521 detail::pearce4VisitRec(Graph, V, Data, Ret);
522 }
523
524 if (!Data.Stack.empty()) {
525 auto NewSCC = SCCId<Vertex>(Ret.NodesInSCC.size());
526 auto &Nodes = Ret.NodesInSCC.emplace_back();
527 Nodes.reserve(Data.Stack.size());
528 for (auto Vtx : Data.Stack) {
529 Nodes.push_back(Vtx);
530 Ret.SCCOfNode[Vtx] = NewSCC;
531 }
532 Data.Stack.clear();
533 }
534 }
535 }
536
537 return Ret;
538}

540/// Compute SCCs and a topological ordering on the SCCs, adapted from the paper
541/// "A Space-Efficient Algorithm for Finding Strongly Connected Components",
542/// Pearce 2015, DOI: <https://doi.org/10.1016/j.ipl.2015.08.010>
543///
544/// \tparam G The graph-type \tparam Iterative Whether to use the iterative
545/// or recursive variant of the algorithm (default: true) \param Graph The graph
546/// for with to compute SCCs and topological ordering
547template <typename G, bool Iterative = true>
548[[nodiscard]] std::pair<SCCHolder<typename GraphTraits<G>::vertex_t>,
549 SCCOrder<typename GraphTraits<G>::vertex_t>>
550computeSCCsAndTopologicalOrder(
551 const G &Graph, std::bool_constant<Iterative> /*Iterative*/ = {}) {
552 using Vertex = typename GraphTraits<G>::vertex_t;
553
554 std::pair<SCCHolder<Vertex>, SCCOrder<Vertex>> Ret = {
555 computeSCCs(Graph, std::bool_constant<Iterative>{}),
556 {},
557 };
558
559 // Pearce's algorithm produces SCCs in reverse topological order
560 auto Ids = llvm::reverse(psr::iota<SCCId<Vertex>>(Ret.first.size()));
561 Ret.second.SCCIds.append(Ids.begin(), Ids.end());
562
563 return Ret;
564}
