99template <typename L> struct AllBottom final {
100 using l_t = L;
101 using JLattice = JoinLatticeTraits<L>;

103 [[no_unique_address]] std::conditional_t<HasJoinLatticeTraits<l_t>, EmptyType,
104 l_t> BottomValue;
105
106 [[nodiscard]] l_t computeTarget(ByConstRef<l_t> /*Source*/) const noexcept {
107 static_assert(std::is_trivially_copyable_v<AllBottom>);
108 static_assert(IsEdgeFunction<AllBottom>);
109 if constexpr (HasJoinLatticeTraits<l_t>) {
110 return JLattice::bottom();
111 } else {
112 return BottomValue;
113 }
114 }

116 [[nodiscard]] static EdgeFunction<l_t>
117 compose(EdgeFunctionRef<AllBottom> This,
118 const EdgeFunction<l_t> &SecondFunction) {
119 if (SecondFunction.isConstant()) {
120 return SecondFunction;
121 }
122
123 if constexpr (HasJoinLatticeTraits<L>) {

125 if (ConstVal == JLattice::bottom()) {
126 return This;

131 } else {
132 // Note: This used to be the default behavior, but it appears to be too

224template <typename L> struct EdgeFunctionComposer {
225 using l_t = L;
226
227 [[nodiscard]] l_t computeTarget(ByConstRef<l_t> Source) const {
228 return Second.computeTarget(First.computeTarget(Source));
229 }
230
231 /**
232 * Function composition is implemented as an explicit composition, i.e.
233 * (secondFunction * G) * F = EFC(F, EFC(G , otherFunction))
234 *
235 * However, it is advised to immediately reduce the resulting edge function
236 * by providing an own implementation of this function.
237 */
238 template <std::derived_from<EdgeFunctionComposer> ConcreteEF>
239 [[nodiscard]] static EdgeFunction<l_t>
240 compose(EdgeFunctionRef<ConcreteEF> This,
241 const EdgeFunction<l_t> &SecondFunction) {
242 static_assert(IsEdgeFunction<ConcreteEF>);
243 if (auto Default = defaultComposeOrNull(This, SecondFunction)) {
244 return Default;
245 }
246 return This->First.composeWith(This->Second.composeWith(SecondFunction));
247 }
248
249 /// This join function is to be provided by derived classes
250
251 // static EdgeFunction<l_t> join(EdgeFunctionRef<EdgeFunctionComposer> This,
252 // const EdgeFunction<l_t> &OtherFunction);
253
254 friend llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
255 const EdgeFunctionComposer &EF) {
256 OS << "EFComposer[" << EF.First << ", " << EF.Second << ']';
257 return OS;
258 }
259
260 [[nodiscard]] friend bool
261 operator==(const EdgeFunctionComposer &LHS,
262 const EdgeFunctionComposer &RHS) noexcept {
263 return LHS.First == RHS.First && LHS.Second == RHS.Second;
264 }
265
266 [[nodiscard]] size_t depth() const noexcept {
267 return First.depth() + Second.depth();
268 }
269
270 // -- data members
271
272 EdgeFunction<l_t> First{};
273 EdgeFunction<l_t> Second{};
274};

278template <typename L, uint8_t N> struct JoinEdgeFunction {
279 using l_t = L;

281
282 [[nodiscard]] l_t computeTarget(ByConstRef<l_t> Source) const {
283 static_assert(IsEdgeFunction<JoinEdgeFunction>);
284 auto Val = Seed;
285 for (const auto &EF : OtherEF) {
286 Val = JLattice::join(std::move(Val), EF.computeTarget(Source));
287 if (Val == JLattice::bottom()) {
288 return Val;

347 [[nodiscard]] static EdgeFunction<l_t> create(EdgeFunction<l_t> LHS,
348 EdgeFunction<l_t> RHS) {
349
350 auto GetEFArrayAndSeed = [](const EdgeFunction<l_t> &EF)
351 -> std::pair<llvm::ArrayRef<EdgeFunction<l_t>>, l_t> {
352 if (const auto *Join = llvm::dyn_cast<JoinEdgeFunction>(EF)) {
353 return {Join->OtherEF, Join->Seed};
354 }
355 return {llvm::ArrayRef(EF), JLattice::top()};
356 };
357
358 auto [LVec, LSeed] = GetEFArrayAndSeed(LHS);
359 auto [RVec, RSeed] = GetEFArrayAndSeed(RHS);
360
361 auto JoinSeed = JLattice::join(std::move(LSeed), std::move(RSeed));
362 if (JoinSeed == JLattice::bottom()) {
363 return AllBottom<l_t>{};
364 }
365
366 llvm::SmallVector<EdgeFunction<l_t>, N> JoinVec;
367
368 std::set_union(LVec.begin(), LVec.end(), RVec.begin(), RVec.end(),
369 std::back_inserter(JoinVec));
370 if (JoinVec.size() > N) {
371 return AllBottom<l_t>{};
372 }
373
374 return JoinEdgeFunction{std::move(JoinSeed), std::move(JoinVec)};
375 }
376
377 // --- data members
378 l_t Seed{};
379 llvm::SmallVector<EdgeFunction<l_t>, N> OtherEF{};
380};

386/// Default implementation of EdgeFunction::join. Considers AllBottom, AllTop
387/// and EdgeIdentity for OtherFunction as well as This==OtherFunction.
388/// Joining with EdgeIdentity will overapproximate to (AllBottom if N==0, else
389/// JoinEdgeFunction).
390template <typename L, uint8_t N = 0, typename ConcreteEF>
391EdgeFunction<L> defaultJoinOrNull(EdgeFunctionRef<ConcreteEF> This,
392 const EdgeFunction<L> &OtherFunction) {
393 if (llvm::isa<AllBottom<L>>(OtherFunction)) {
394 return OtherFunction;
395 }
396 if (llvm::isa<AllTop<L>>(OtherFunction) || OtherFunction == This) {
397 return This;
398 }
399 if (llvm::isa<EdgeIdentity<L>>(OtherFunction)) {
400 if constexpr (N > 0) {
401 return JoinEdgeFunction<L, N>::create(This, OtherFunction);
402 } else if constexpr (HasJoinLatticeTraits<L>) {
403 return AllBottom<L>{};
404 }
405 }
406 return nullptr;
407}

411 const EdgeFunction<L> &OtherFunction) {
412 if (llvm::isa<AllBottom<L>>(OtherFunction) || llvm::isa<AllTop<L>>(This)) {

429template <typename L>
430EdgeFunction<L> EdgeIdentity<L>::join(EdgeFunctionRef<EdgeIdentity> This,
431 const EdgeFunction<L> &OtherFunction) {
432 if (llvm::isa<EdgeIdentity<L>>(OtherFunction) ||
433 llvm::isa<AllTop<L>>(OtherFunction)) {
434 return This;
435 }
436 if (llvm::isa<AllBottom<L>>(OtherFunction)) {
437 return OtherFunction;

444template <typename L>
445template <typename ConcreteEF>
446EdgeFunction<L>
447ConstantEdgeFunction<L>::compose(EdgeFunctionRef<ConcreteEF> This,
448 const EdgeFunction<L> &SecondFunction)
449 requires std::derived_from<ConcreteEF, ConstantEdgeFunction<L>>
450{
451 if (auto Default = defaultComposeOrNull(This, SecondFunction)) {
452 return Default;
453 }
454
455 auto ConstVal = SecondFunction.computeTarget(This->Value);
456 if constexpr (!EdgeFunctionBase::IsSOOCandidate<ConstantEdgeFunction>) {
457 if (ConstVal == This->Value) {
458 return This;
459 }
460 }
461
462 if constexpr (AreEqualityComparable<decltype(JLattice::bottom()), L>) {
463 if (JLattice::bottom() == ConstVal) {
464 return AllBottom<L>{};
465 }
466 } else {
467 if (L(JLattice::bottom()) == ConstVal) {
468 return AllBottom<L>{};
469 }
470 }
471
472 if constexpr (AreEqualityComparable<decltype(JLattice::top()), L>) {
473 if (JLattice::top() == ConstVal) {
474 /// TODO: Can this ever happen?
475 return AllTop<L>{};
476 }
477 } else {
478 if (L(JLattice::top()) == ConstVal) {
479 /// TODO: Can this ever happen?
480 return AllTop<L>{};
481 }
482 }
483
484 return ConstantEdgeFunction{NonTopBotValue<L>::unwrap(std::move(ConstVal))};
485}

487template <typename L>
488template <typename ConcreteEF>
489EdgeFunction<L>
490ConstantEdgeFunction<L>::join(EdgeFunctionRef<ConcreteEF> This,
491 const EdgeFunction<l_t> &OtherFunction)
492 requires std::derived_from<ConcreteEF, ConstantEdgeFunction<L>>
493{
494 if (auto Default = defaultJoinOrNull<l_t>(This, OtherFunction)) {
495 return Default;
496 }
497
498 if (llvm::isa<EdgeIdentity<L>>(OtherFunction)) {
499 // Prevent endless recursion
500 return AllBottom<L>{};
501 }
502
503 if (!OtherFunction.isConstant()) {
504 // do not know how to join; hence ask other function to decide on this
505 return OtherFunction.joinWith(This);
506 }
507
508 auto OtherVal = OtherFunction.computeTarget(JLattice::top());
509 auto JoinedVal = JLattice::join(This->Value, OtherVal);
510
511 if constexpr (AreEqualityComparable<decltype(JLattice::bottom()), l_t>) {
512 if (JLattice::bottom() == JoinedVal) {
513 return AllBottom<l_t>{};
514 }
515 } else {
516 if (l_t(JLattice::bottom()) == JoinedVal) {
517 return AllBottom<l_t>{};
518 }
519 }
520 if constexpr (!EdgeFunctionBase::IsSOOCandidate<ConstantEdgeFunction>) {
521 if (JoinedVal == OtherVal) {
522 return OtherFunction;
523 }
524 if (JoinedVal == This->Value) {
525 return This;
526 }
527 }
528 return ConstantEdgeFunction{
529 NonTopBotValue<l_t>::unwrap(std::move(JoinedVal))};
530}
