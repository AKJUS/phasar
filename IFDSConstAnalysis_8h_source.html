36/**
37 * This IFDS analysis will compute possibly mutable memory
38 * locations (stack and heap). LLVM's virtual register
39 * are not considered since they are in SSA form.
40 * A memory location is considered mutable after the second
41 * write access. Thus, the first write access is allowed
42 * to account for initialization.
43 * @brief Computes all possibly mutable memory locations.
44 */
45class IFDSConstAnalysis
46 : public IFDSTabulationProblem<LLVMIFDSAnalysisDomainDefault> {
47
48public:
49 IFDSConstAnalysis(const LLVMProjectIRDB *IRDB, LLVMAliasInfoRef PT,
50 std::vector<std::string> EntryPoints = {"main"});
51
52 ~IFDSConstAnalysis() override = default;
53
54 /**
55 * If the current instruction is a store instruction, the memory locations's
56 * state of initialization is checked. If the memory location was already
57 * initialized, i.e. at least one write access occurred, the
58 * pointer operand is generated as a data-flow fact. Also all aliases that
59 * meet the 'context-relevant' requirements (see {@link
60 * getContextRelevantAliasSet}) will be generated!
61 *
62 * Otherwise, the memory location (i.e. memory location's pointer operand) is
63 * marked as initialized.
64 *
65 * To infer the state of initialization, the memory location's pointer operand
66 * and all it's aliases are checked to see if one of them is marked as
67 * initialized.
68 *
69 * Vtable updates are ignored, and thus not counting towards an object's
70 * mutability state.
71 * @brief Processing store instructions by generating new data-flow facts, if
72 * more than one write access to the memory location occurred.
73 * @param curr Currently analyzed program statement.
74 * @param succ Successor statement.
75 */
76 FlowFunctionPtrType getNormalFlowFunction(n_t Curr, n_t Succ) override;
77
78 /**
79 * The following llvm intrinsics
80 * \li memcpy
81 * \li memmove
82 * \li memset
83 * count as a write access to their target memory location.
84 * Since those functions are declaration only and cannot be analyzed
85 * directly, the effects are modeled by killing all data-flow facts
86 * before the call (at this point), and generate respective data-flow facts in
87 * the corresponding call-to-return flow function (see {@link
88 * getCallToRetFlowFunction}).
89 *
90 * Call or invoke instructions are handled by mapping actual parameters
91 * into formal parameters, i.e. propagating relevant data-flow facts from the
92 * caller into the callee context.
93 * @brief Processing call/invoke instructions and llvm memory intrinsic
94 * functions.
95 * @param CallSite Call statement.
96 * @param destFun Callee function.
97 */
98 FlowFunctionPtrType getCallFlowFunction(n_t CallSite, f_t DestFun) override;
99
100 /**
101 * Maps formal parameters back into actual parameters. Data-flow fact(s)
102 * associated with the return value are propagated into the caller context.
103 * @brief Processing a function return.
104 * @param CallSite Call site.
105 * @param CalleeFun Callee function.
106 * @param ExitInst Exit statement in callee.
107 * @param retSite Return site.
108 */
109 FlowFunctionPtrType getRetFlowFunction(n_t CallSite, f_t CalleeFun,
110 n_t ExitStmt, n_t RetSite) override;
111
112 /**
113 * If the called function is a llvm memory intrinsic function, appropriate
114 * data-flow facts are generated at this point. In essence, these functions
115 * are handled just as store instructions, i.e. generating new data-flow facts
116 * if the target memory location (first operand) is already initialized.
117 *
118 * Otherwise, all data-flow facts are passed as identity.
119 * @brief Processing the effects of llvm memory intrinsic functions.
120 * @param CallSite Call site.
121 * @param retSite Return site.
122 */
123 FlowFunctionPtrType
124 getCallToRetFlowFunction(n_t CallSite, n_t RetSite,
125 llvm::ArrayRef<f_t> Callees) override;
126
127 /**
128 * @brief Not used for this analysis, i.e. always returning nullptr.
129 */
130 FlowFunctionPtrType getSummaryFlowFunction(n_t CallSite,
131 f_t DestFun) override;
132
133 /**
134 * Only the zero value is valid at the first program statement, i.e.
135 * all memory locations are considered immutable.
136 * @brief Provides data-flow facts that are valid at the program entry point.
137 */
138 InitialSeeds<n_t, d_t, l_t> initialSeeds() override;
139
140 /**
141 * @brief Returns appropriate zero value.
142 */
143 [[nodiscard]] d_t createZeroValue() const;

145 [[nodiscard]] bool isZeroValue(d_t Fact) const noexcept override;

147 void emitTextReport(GenericSolverResults<n_t, d_t, BinaryDomain> SR,

150 /**
151 * @note Global Variables are always intialized in llvm IR, and therefore
152 * not part of the Initialized set.
153 * @brief Checks if the given memory location is initialized
154 */
155 bool isInitialized(d_t Fact) const;

172 // clang-format off
173 /**
174 * We only want/need to generate aliases if they meet one of the
175 * following conditions
176 * <ol>
177 * <li> alias is an instruction from within the current function context
178 * <li> alias is an allocation instruction for stack memory (alloca) or
179 * heap memory (new, new[], malloc, calloc, realloc) from any function
180 * context
181 * <li> alias is a global variable
182 * <li> alias is a formal argument of the current function
183 * <li> alias is a return value of pointer type
184 * </ol>
185 *
186 * Condition (1) is necessary to cover the case, when an initialized
187 * memory location is mutated in a function different from where its
188 * original allocation site.<br>
189 * Condition (3) is necessary to be able to map mutated parameter
190 * back to the caller context if needed. <br>
191 * Same goes for (4).
192 *
193 * Everything else will be ignored since we are not interested in
194 * intermediate pointer or values of other functions, i.e. values
195 * in virtual registers.
196 * Only points-to information and the Initialized set determine, if
197 * new data-flow facts will be generated.
198 * @brief Refines the given points-to information to only context-relevant
199 * points-to information.
200 * @param AliasSet that is refined.
201 * @param Context dictates which points-to information is relevant.
202 */ // clang-format on
203 static std::set<d_t> getContextRelevantAliasSet(std::set<d_t> &AliasSet,
204 f_t Context);
205
206private:
207 LLVMAliasInfoRef PT{};
208 // Holds all allocated memory locations, including global variables
209 std::set<d_t> AllMemLocs; // FIXME: initialize within the constructor body!
210 // Holds all initialized variables and objects.
211 std::set<d_t> Initialized;
212};
