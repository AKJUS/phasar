60/// \brief The analysis problem interface for IDE problems (solvable by the
61/// IDESolver). Create a subclass from this and override all pure-virtual
62/// functions to create your own IDE analysis.
63///
64/// For more information on how to write an IDE analysis, see [Writing an IDE
65/// Analysis](https://github.com/secure-software-engineering/phasar/wiki/Writing-an-IDE-analysis)
66template <typename AnalysisDomainTy,
67 typename Container = std::set<typename AnalysisDomainTy::d_t>>
68class IDETabulationProblem : public FlowFunctions<AnalysisDomainTy, Container>,
69 public EdgeFunctions<AnalysisDomainTy>,
70 public JoinLattice<AnalysisDomainTy>,
71 public SemiRing<AnalysisDomainTy>,

86 /// Takes an IR database (IRDB) and collects information from it to create a
87 /// tabulation problem.
88 /// @param[in] IRDB The project IR database, that holds the code under
89 /// analysis
90 /// @param[in] EntryPoints The (mangled) names of all entry functions of the
91 /// target being analyzed, given as a vector of strings. An example would
92 /// simply be `{"main"}`. To set every function as entry point, pass
93 /// `"__ALL__"`
94 /// @param[in] ZeroValue Provides the special tautological zero value (aka.
95 /// Λ). If not provided here, you must set it via \link initializeZeroValue()
96 /// \endlink.
97 explicit IDETabulationProblem(
98 const ProjectIRDBBase<db_t> *IRDB, std::vector<std::string> EntryPoints,
99 std::optional<d_t>
100 ZeroValue) noexcept(std::is_nothrow_move_constructible_v<d_t>)
101 : IRDB(IRDB), EntryPoints(std::move(EntryPoints)),
102 ZeroValue(std::move(ZeroValue)),
103 Printer(std::make_unique<typename DefaultAnalysisPrinterSelector<
104 AnalysisDomainTy>::type>()) {
105 assert(IRDB != nullptr);
106 }
107
108 IDETabulationProblem(IDETabulationProblem &&) noexcept = default;
109 IDETabulationProblem &operator=(IDETabulationProblem &&) noexcept = default;
110
111 IDETabulationProblem(const IDETabulationProblem &) = delete;
112 IDETabulationProblem &operator=(const IDETabulationProblem &) = delete;
113
114 ~IDETabulationProblem() override = default;

120 } else {

122 }
123 }

125 /// Checks if the given data-flow fact is the special tautological lambda (or
126 /// zero) fact.
127 [[nodiscard]] virtual bool isZeroValue(d_t FlowFact) const noexcept {
128 assert(ZeroValue.has_value());
129 return FlowFact == *ZeroValue;
130 }
131
132 /// Returns initial seeds to be used for the analysis. This is a mapping of
133 /// statements to initial analysis facts.
134 [[nodiscard]] virtual InitialSeeds<n_t, d_t, l_t> initialSeeds() = 0;
135
136 /// Returns the special tautological lambda (or zero) fact.
137 [[nodiscard]] ByConstRef<d_t> getZeroValue() const {
138 assert(ZeroValue.has_value());
139 return *ZeroValue;
140 }
141
142 void initializeZeroValue(d_t Zero) noexcept(
143 std::is_nothrow_assignable_v<std::optional<d_t> &, d_t &&>) {
144 assert(!ZeroValue.has_value());
145 ZeroValue = std::move(Zero);

202 /// Seeds that just start with ZeroValue and bottomElement() at the starting
203 /// points of each EntryPoint function.
204 /// Takes the __ALL__ EntryPoint into account.
205 [[nodiscard]] InitialSeeds<n_t, d_t, l_t> createDefaultSeeds()
206 requires std::is_nothrow_default_constructible_v<
207 typename AnalysisDomainTy::c_t>
208 {
209 InitialSeeds<n_t, d_t, l_t> Seeds;
210 typename AnalysisDomainTy::c_t C{};
211
212 addSeedsForStartingPoints(EntryPoints, IRDB, C, Seeds, getZeroValue(),
213 this->bottomElement());
214
215 return Seeds;
216 }

224 [[maybe_unused]] Soundness SF = Soundness::Soundy;

226 MaybeUniquePtr<AnalysisPrinterBase<AnalysisDomainTy>> Printer;
227};
