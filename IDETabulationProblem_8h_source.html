61/// \brief The analysis problem interface for IDE problems (solvable by the
62/// IDESolver). Create a subclass from this and override all pure-virtual
63/// functions to create your own IDE analysis.
64///
65/// For more information on how to write an IDE analysis, see [Writing an IDE
66/// Analysis](https://github.com/secure-software-engineering/phasar/wiki/Writing-an-IDE-analysis)
67template <typename AnalysisDomainTy,
68 typename Container = std::set<typename AnalysisDomainTy::d_t>>
69class IDETabulationProblem : public FlowFunctions<AnalysisDomainTy, Container>,
70 public EdgeFunctions<AnalysisDomainTy>,
71 public JoinLattice<AnalysisDomainTy>,
72 public SemiRing<AnalysisDomainTy>,
73 public AllTopFnProvider<AnalysisDomainTy> {

87 /// Takes an IR database (IRDB) and collects information from it to create a
88 /// tabulation problem.
89 /// @param[in] IRDB The project IR database, that holds the code under
90 /// analysis
91 /// @param[in] EntryPoints The (mangled) names of all entry functions of the
92 /// target being analyzed, given as a vector of strings. An example would
93 /// simply be `{"main"}`. To set every function as entry point, pass
94 /// `"__ALL__"`
95 /// @param[in] ZeroValue Provides the special tautological zero value (aka.
96 /// Λ). If not provided here, you must set it via \link initializeZeroValue()
97 /// \endlink.
98 explicit IDETabulationProblem(
99 const ProjectIRDBBase<db_t> *IRDB, std::vector<std::string> EntryPoints,
100 std::optional<d_t>
101 ZeroValue) noexcept(std::is_nothrow_move_constructible_v<d_t>)
102 : IRDB(IRDB), EntryPoints(std::move(EntryPoints)),
103 ZeroValue(std::move(ZeroValue)),
104 Printer(std::make_unique<typename DefaultAnalysisPrinterSelector<
105 AnalysisDomainTy>::type>()) {
106 assert(IRDB != nullptr);
107 }
108
109 IDETabulationProblem(IDETabulationProblem &&) noexcept = default;
110 IDETabulationProblem &operator=(IDETabulationProblem &&) noexcept = default;
111
112 IDETabulationProblem(const IDETabulationProblem &) = delete;
113 IDETabulationProblem &operator=(const IDETabulationProblem &) = delete;
114
115 ~IDETabulationProblem() override = default;

123 }
124 }

126 /// Checks if the given data-flow fact is the special tautological lambda (or
127 /// zero) fact.
128 [[nodiscard]] virtual bool isZeroValue(d_t FlowFact) const noexcept {
129 assert(ZeroValue.has_value());
130 return FlowFact == *ZeroValue;
131 }
132
133 /// Returns initial seeds to be used for the analysis. This is a mapping of
134 /// statements to initial analysis facts.
135 [[nodiscard]] virtual InitialSeeds<n_t, d_t, l_t> initialSeeds() = 0;
136
137 /// Returns the special tautological lambda (or zero) fact.
138 [[nodiscard]] ByConstRef<d_t> getZeroValue() const {
139 assert(ZeroValue.has_value());
140 return *ZeroValue;
141 }
142
143 void initializeZeroValue(d_t Zero) noexcept(
144 std::is_nothrow_assignable_v<std::optional<d_t> &, d_t &&>) {
145 assert(!ZeroValue.has_value());
146 ZeroValue = std::move(Zero);

155 [[nodiscard]] IFDSIDESolverConfig &getIFDSIDESolverConfig() noexcept {

203 /// Seeds that just start with ZeroValue and bottomElement() at the starting
204 /// points of each EntryPoint function.
205 /// Takes the __ALL__ EntryPoint into account.
206 template <
207 typename CC = typename AnalysisDomainTy::c_t,
208 typename = std::enable_if_t<std::is_nothrow_default_constructible_v<CC>>>
209 [[nodiscard]] InitialSeeds<n_t, d_t, l_t> createDefaultSeeds() {
210 InitialSeeds<n_t, d_t, l_t> Seeds;
211 CC C{};
212
213 addSeedsForStartingPoints(EntryPoints, IRDB, C, Seeds, getZeroValue(),
214 this->bottomElement());
215
216 return Seeds;
217 }
