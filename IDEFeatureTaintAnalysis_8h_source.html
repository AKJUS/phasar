41struct IDEFeatureTaintEdgeFact {
42 llvm::SmallBitVector Taints;
43
44 [[nodiscard]] static llvm::SmallBitVector fromBits(uintptr_t Bits) {
45#if __has_builtin(__builtin_constant_p)
46 if (__builtin_constant_p(Bits) && Bits == 0) {
47 return {};
48 }
49#endif
50
51 llvm::SmallBitVector Ret(llvm::findLastSet(Bits) + 1);
52 Ret.setBitsInMask((const uint32_t *)&Bits, sizeof(Bits));
53 return Ret;
54 }
55
56 IDEFeatureTaintEdgeFact(llvm::SmallBitVector &&Taints) noexcept
57 : Taints(std::move(Taints)) {}
58 IDEFeatureTaintEdgeFact(const llvm::SmallBitVector &Taints)
59 : Taints(Taints) {}
60 IDEFeatureTaintEdgeFact(uintptr_t Taints) noexcept
61 : Taints(fromBits(Taints)) {}
62 explicit IDEFeatureTaintEdgeFact() noexcept { Taints.invalid(); }
63
64 void unionWith(uintptr_t Facts) {
65 auto RequiredSize = llvm::findLastSet(Facts) + 1;
66 if (RequiredSize > Taints.size()) {
67 Taints.resize(RequiredSize);
68 }
69 Taints.setBitsInMask((const uint32_t *)&Facts,
70 sizeof(Facts) / sizeof(uint32_t));
71 }
72 void unionWith(const IDEFeatureTaintEdgeFact &Facts) {
73 if (Facts.isTop()) {
74 return;
75 }
76 Taints |= Facts.Taints;
77 }
78
79 /// Checks whether this set contains no facts.
80 /// Note: Top also counts as empty
81 [[nodiscard]] inline bool empty() const noexcept {
82 return isTop() || Taints.none();
83 }
84 /// Returns the number of facts in this set.
85 /// Note: Top counts as empty
86 [[nodiscard]] inline size_t size() const noexcept {
87 if (isTop()) {
88 return 0;
89 }
90 return Taints.count();
91 }
92
93 [[nodiscard]] inline bool isBottom() const noexcept { return Taints.empty(); }
94 [[nodiscard]] inline bool isTop() const noexcept {
95 return Taints.isInvalid();
96 }
97
98 [[nodiscard]] friend llvm::hash_code
99 hash_value(const IDEFeatureTaintEdgeFact &BV) noexcept {
100 if (BV.Taints.empty()) {
101 return {};
102 }
103 uintptr_t Buf;
104 auto Words = BV.Taints.getData(Buf);
105 size_t Idx = Words.size();
106 while (Idx && Words[Idx - 1] == 0) {
107 --Idx;
108 }
109 return llvm::hash_combine_range(Words.begin(),
110 std::next(Words.begin(), Idx));
111 }
112
113 friend bool operator==(const IDEFeatureTaintEdgeFact &Lhs,
114 const IDEFeatureTaintEdgeFact &Rhs) noexcept {
115 bool LeftEmpty = Lhs.Taints.none();
116 bool RightEmpty = Rhs.Taints.none();
117 if (LeftEmpty || RightEmpty) {
118 return LeftEmpty == RightEmpty;
119 }
120 // Check, whether Lhs and Rhs actually have the same bits set and not
121 // whether their internal representation is exactly identitcal
122
123 uintptr_t LBuf, RBuf;
124 auto LhsWords = Lhs.Taints.getData(LBuf);
125 auto RhsWords = Rhs.Taints.getData(RBuf);
126 if (LhsWords.size() == RhsWords.size()) {
127 return LhsWords == RhsWords;
128 }
129 auto MinSize = std::min(LhsWords.size(), RhsWords.size());
130 if (LhsWords.slice(0, MinSize) != RhsWords.slice(0, MinSize)) {
131 return false;
132 }
133 auto Rest = (LhsWords.size() > RhsWords.size() ? LhsWords : RhsWords)
134 .slice(MinSize);
135 return std::all_of(Rest.begin(), Rest.end(),
136 [](auto Word) { return Word == 0; });
137 }
138
139 friend bool operator!=(const IDEFeatureTaintEdgeFact &Lhs,
140 const IDEFeatureTaintEdgeFact &Rhs) noexcept {
141 return !(Lhs == Rhs);
142 }
143
144 template <typename E> [[nodiscard]] std::string str() const {
145 auto BV = BitVectorSet<E, llvm::SmallBitVector>::fromBits(Taints);
146 return LToString(BV);
147 }
148
149 template <typename E> [[nodiscard]] auto toBVSet() const {
150 if (isTop()) {
151 return BitVectorSet<E, llvm::SmallBitVector>();
152 }
153 return BitVectorSet<E, llvm::SmallBitVector>::fromBits(Taints);
154 }
155
156 template <typename E> [[nodiscard]] auto toSet() const {
157 std::set<E> Ret;
158 if (isTop()) {
159 return Ret;
160 }
161
162 for (const auto &Elem : this->template toBVSet<E>()) {
163 Ret.insert(Elem);
164 }
165 return Ret;
166 }

198class FeatureTaintGenerator {
199public:
200 using InstOrGlobal =
201 std::variant<const llvm::Instruction *, const llvm::GlobalVariable *>;
202
203 using GenerateTaintsFn =
204 llvm::unique_function<IDEFeatureTaintEdgeFact(InstOrGlobal) const>;
205 using IsSourceFn = llvm::unique_function<bool(InstOrGlobal) const>;
206 using PrinterFn =
207 llvm::unique_function<std::string(const IDEFeatureTaintEdgeFact &) const>;
208
209 template <typename EdgeFactGenerator>
210 static GenerateTaintsFn createGenerateTaints(EdgeFactGenerator &&EFGen) {
211 return [EFGen{std::forward<EdgeFactGenerator>(EFGen)}](InstOrGlobal IG) {
212 const auto &TaintSet = std::invoke(EFGen, IG);
213 BitVectorSet<ElementType<decltype(TaintSet)>, llvm::SmallBitVector> BV(
214 llvm::adl_begin(TaintSet), llvm::adl_end(TaintSet));
215
216 return IDEFeatureTaintEdgeFact{std::move(BV).getBits()};
217 };
218 }
219
220 template <typename EdgeFactGenerator>
221 static PrinterFn createEdgeFactPrinter() {
222 using ElemTy =
223 ElementType<std::invoke_result_t<EdgeFactGenerator, InstOrGlobal>>;
224 return [](const IDEFeatureTaintEdgeFact &Fact) {
225 auto BV =
226 BitVectorSet<ElemTy, llvm::SmallBitVector>::fromBits(Fact.Taints);
227 return LToString(BV);
228 };
229 }
230
231 FeatureTaintGenerator(IsSourceFn IsFeatureSource,
232 GenerateTaintsFn GenerateTaints, PrinterFn Printer)
233 : IsFeatureSource(std::move(IsFeatureSource)),
234 GenerateTaints(std::move(GenerateTaints)), Printer(std::move(Printer)) {
235 }
236
237 template <typename SourceDetector, typename EdgeFactGenerator>
238 FeatureTaintGenerator(SourceDetector &&SrcDetector, EdgeFactGenerator &&EFGen)
239 : IsFeatureSource(std::forward<SourceDetector>(SrcDetector)),
240 GenerateTaints(
241 createGenerateTaints(std::forward<EdgeFactGenerator>(EFGen))),
242 Printer(createEdgeFactPrinter<EdgeFactGenerator>()) {}
243
244 template <typename EdgeFactGenerator,
245 typename = std::enable_if_t<!std::is_same_v<
246 FeatureTaintGenerator, std::decay_t<EdgeFactGenerator>>>>
247 FeatureTaintGenerator(EdgeFactGenerator &&EFGen)
248 : FeatureTaintGenerator(
249 [EFGen](InstOrGlobal IG) {
250 return !llvm::empty(std::invoke(EFGen, IG));
251 },
252 std::forward<EdgeFactGenerator>(EFGen)) {}
253
254 [[nodiscard]] bool isSource(InstOrGlobal IG) const {
255 return IsFeatureSource(IG);
256 }
257
258 [[nodiscard]] IDEFeatureTaintEdgeFact
259 getGeneratedTaintsAt(InstOrGlobal IG) const {
260 return GenerateTaints(IG);
261 }
262
263 [[nodiscard]] std::string
264 toString(const IDEFeatureTaintEdgeFact &Fact) const {
265 return Printer(Fact);
266 }
267
268private:
269 IsSourceFn IsFeatureSource;
270 GenerateTaintsFn GenerateTaints;
271 PrinterFn Printer;
272};
