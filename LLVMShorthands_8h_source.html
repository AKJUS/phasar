128/**
129 * Only Instructions and GlobalVariables have 'real' ID's, i.e. annotated meta
130 * data. Formal arguments cannot be annotated with metadata in LLVM. Therefore,
131 * a formal arguments ID will look like this:
132 * <function_name>.<#argument>
133 *
134 * ZeroValue will have -1 as ID by default.
135 *
136 * @brief Returns the ID of a given LLVM Value.
137 * @return Meta data ID as a string or -1, if it's not
138 * an Instruction, GlobalVariable or Argument.
139 */
140std::string getMetaDataID(const llvm::Value *V);
