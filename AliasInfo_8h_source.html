44/// \brief A type-erased reference to any object implementing the IsAliasInfo
45/// interface. Use this, if your analysis is not tied to a specific alias info
46/// implementation.
47///
48/// This is a *non-owning* reference similar to std::string_view and
49/// llvm::ArrayRef. Pass values of this type by value.
50///
51/// Example:
52/// \code
53/// LLVMAliasSet ASet(...);
54/// AliasInfoRef AA = &ASet;
55/// \endcode
56///
57/// NOTE: AliasInfoRef::mergeWith() only works if supplied with a compatible
58/// other AliasInfo. Otherwise, it asserts out
59template <typename V, typename N>
60class AliasInfoRef : public AnalysisPropertiesMixin<AliasInfoRef<V, N>> {
61 friend class AliasInfo<V, N>;
62 using traits_t = AliasInfoTraits<AliasInfoRef<V, N>>;
63
64public:
65 using AliasSetPtrTy = typename traits_t::AliasSetPtrTy;
66 using AliasSetTy = typename traits_t::AliasSetTy;
67 using AllocationSiteSetPtrTy = typename traits_t::AllocationSiteSetPtrTy;
68 using n_t = typename traits_t::n_t;
69 using v_t = typename traits_t::v_t;
70
71 AliasInfoRef() noexcept = default;
72 AliasInfoRef(std::nullptr_t) noexcept : AliasInfoRef() {}
73 template <typename ConcreteAA,
74 typename = std::enable_if_t<
75 !std::is_base_of_v<AliasInfoRef, ConcreteAA> &&
76 std::is_same_v<v_t, typename ConcreteAA::v_t> &&
77 std::is_same_v<n_t, typename ConcreteAA::n_t>>>
78 AliasInfoRef(ConcreteAA *AA) noexcept
79 : AA(AA), VT((std::is_empty_v<ConcreteAA> || AA) ? &VTableFor<ConcreteAA>
80 : nullptr) {}
81
82 /// Prevent dangling references by disallowing implicit conversion of a
83 /// temporary AliasInfo to AliasInfoRef.
84 AliasInfoRef(AliasInfo<V, N> &&) = delete;
85 AliasInfoRef &operator=(AliasInfo<V, N> &&) = delete;
86
87 AliasInfoRef(const AliasInfoRef &) noexcept = default;
88 AliasInfoRef &operator=(const AliasInfoRef &) noexcept = default;
89 ~AliasInfoRef() noexcept = default;
90
91 explicit operator bool() const noexcept { return VT != nullptr; }

147 void mergeWith(AliasInfoRef Other) {
148 assert(VT != nullptr);
149 assert(Other.VT != nullptr);
150 if (VT != Other.VT) {
151 llvm::report_fatal_error(
152 "Can only merge AliasInfos with same concrete type: " +
153 VT->TypeName() + " vs " + Other.VT->TypeName());

155 VT->MergeWith(AA, Other.AA);
156 }

263/// \brief Similar to AliasInfoRef, but exclusively owns the held reference. Use
264/// this, if you need to decide dynamically, which alias info implementation to
265/// use.
266///
267/// Implicitly convertible to AliasInfoRef.
268///
269/// Example:
270/// \code
271/// AliasInfo AA = std::make_unique<LLVMAliasSet>(...);
272/// \endcode
273///
274template <typename V, typename N>
275class [[clang::trivial_abi]] AliasInfo final : public AliasInfoRef<V, N> {
276 using base_t = AliasInfoRef<V, N>;
277
278public:
279 using typename base_t::AliasSetPtrTy;
280 using typename base_t::AliasSetTy;
281 using typename base_t::AllocationSiteSetPtrTy;
282 using typename base_t::n_t;
283 using typename base_t::v_t;
284
285 AliasInfo() noexcept = default;
286 AliasInfo(std::nullptr_t) noexcept {};
287 AliasInfo(const AliasInfo &) = delete;
288 AliasInfo &operator=(const AliasInfo &) = delete;
289 AliasInfo(AliasInfo &&Other) noexcept { swap(Other); }
290 AliasInfo &operator=(AliasInfo &&Other) noexcept {
291 auto Cpy{std::move(Other)};
292 swap(Cpy);
293 return *this;
294 }
295
296 void swap(AliasInfo &Other) noexcept {
297 std::swap(this->AA, Other.AA);
298 std::swap(this->VT, Other.VT);
299 }
300 friend void swap(AliasInfo &LHS, AliasInfo &RHS) noexcept { LHS.swap(RHS); }
301
302 template <typename ConcretePTA, typename... ArgTys>
303 explicit AliasInfo(std::in_place_type_t<ConcretePTA> /*unused*/,
304 ArgTys &&...Args)
305 : base_t(new ConcretePTA(std::forward<ArgTys>(Args)...)) {}
306
307 template <typename ConcretePTA>
308 AliasInfo(std::unique_ptr<ConcretePTA> AA) : base_t(AA.release()) {}
309
310 ~AliasInfo() noexcept {
311 if (*this) {
312 this->VT->Destroy(this->AA);
313 this->VT = nullptr;
314 this->AA = nullptr;
315 }
316 }
317
318 [[nodiscard]] base_t asRef() & noexcept { return *this; }
319 [[nodiscard]] AliasInfoRef<V, N> asRef() const & noexcept { return *this; }
320 [[nodiscard]] AliasInfoRef<V, N> asRef() && = delete;
321
322 /// For better interoperability with unique_ptr
323 [[nodiscard]] base_t get() & noexcept { return asRef(); }
324 [[nodiscard]] AliasInfoRef<V, N> get() const & noexcept { return asRef(); }
325 [[nodiscard]] AliasInfoRef<V, N> get() && = delete;
326};
