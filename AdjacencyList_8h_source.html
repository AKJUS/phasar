56 /// Adds a new node to the graph G with node-tag Val
57 ///
58 /// \returns The vertex-descriptor for the newly created node
59 template <typename TT = value_type>
60 static constexpr vertex_t addNode(graph_type &G, TT &&Val)
61 requires(!std::is_empty_v<value_type>)
62 {
63 assert(G.Adj.size() == G.Nodes.size());
64
65 auto Ret = vertex_t(G.Nodes.size());
66 G.Nodes.push_back(std::forward<TT>(Val));
67 G.Adj.emplace_back();
68 return Ret;
69 }

71 /// Adds a new node to the graph G without node-tag
72 static constexpr vertex_t addNode(graph_type &G, value_type /*Val*/ = {})
73 requires std::is_empty_v<value_type>
74 {
75 auto Ret = vertex_t(G.Adj.size());
76 G.Adj.emplace_back();
77 return Ret;

80 /// Makes the node Vtx as root in the graph G. A node should not be registered
81 /// as root multiple times
82 static constexpr void addRoot(graph_type &G, vertex_t Vtx) {
83 assert(G.Adj.inbounds(Vtx));
84 if constexpr (!std::is_empty_v<value_type>) {

99 /// Adds a new edge from node From to node To in graph G. From and To should
100 /// be nodes inside G. Multi-edges are supported, i.e. edges are not
101 /// deduplicated automatically; to manually deduplicate the edges of one
102 /// source-node, call dedupOutEdges()
103 static constexpr void addEdge(graph_type &G, vertex_t From, edge_t To) {
104 assert(G.Adj.inbounds(From));
105 if constexpr (!std::is_empty_v<value_type>) {
106 assert(G.Adj.size() == G.Nodes.size());
107 }
108 G.Adj[From].push_back(std::move(To));
109 }

131 /// Deduplicates the edges outgoing from node Vtx in graph G. Deduplication is
132 /// based on operator== of the edge_t type, and operator< if available.
133 static constexpr void dedupOutEdges(graph_type &G, vertex_t Vtx) noexcept {
134 assert(G.Adj.inbounds(Vtx));
135 if constexpr (!std::is_empty_v<value_type>) {
136 assert(G.Adj.size() == G.Nodes.size());
137 }
138 auto &OutEdges = G.Adj[Vtx];
139
140 if constexpr (IsLessComparable<edge_t>) {
141 std::sort(OutEdges.begin(), OutEdges.end());
142 OutEdges.erase(std::unique(OutEdges.begin(), OutEdges.end()),
143 OutEdges.end());
144 } else {
145 auto End = OutEdges.end();
146 for (auto It = OutEdges.begin(); It < End; ++It) {
147 End = std::remove(std::next(It), End, *It);
148 }
149 OutEdges.erase(End, OutEdges.end());
150 }
151 }

236 /// Tries to remove the last inserted node Vtx fro graph G. Fails, if there
237 /// was another not-popped node inserted in between.
238 ///
239 /// \returns True, iff the removal was successful
240 static constexpr bool pop(graph_type &G, vertex_t Vtx) {
241 if (size_t(Vtx) == G.Adj.size() - 1) {
242 G.Adj.pop_back();
243 if constexpr (!std::is_empty_v<value_type>) {
244 G.Nodes.pop_back();
245 }
246 return true;
247 }
248 return false;
249 }

272 /// Removes the edge denoted by It outgoing from source-vertex Vtx from the
273 /// graph G. This function is not required by the is_graph_trait concept.
274 ///
275 /// \returns An edge_iterator directly following It that should be used to
276 /// continue iteration instead of std::next(It)
277 static constexpr edge_iterator removeEdge(graph_type &G, vertex_t Vtx,
278 edge_iterator It) noexcept {
279 assert(G.Adj.inbounds(Vtx));
280 if constexpr (!std::is_empty_v<value_type>) {
281 assert(G.Adj.size() == G.Nodes.size());
282 }
283 assert(G.Adj[Vtx].begin() <= It && It < G.Adj[Vtx].end());
284 auto Idx = std::distance(std::cbegin(G.Adj[Vtx]), It);
285
286 std::swap(G.Adj[Vtx][Idx], G.Adj[Vtx].back());
287 G.Adj[Vtx].pop_back();
288 return It;
289 }
290
291 /// Removes the root denoted by It from the graph G. This function is not
292 /// required by the is_graph_trait concept.
293 ///
294 /// \returns A roots_iterator directly following It that should be used to
295 /// continue iteration instead of std::next(It)
296 static constexpr roots_iterator removeRoot(graph_type &G,
297 roots_iterator It) noexcept {
298 if constexpr (!std::is_empty_v<value_type>) {
299 assert(G.Adj.size() == G.Nodes.size());
300 }
301 assert(G.Roots.begin() <= It && It < G.Roots.end());
302
303 auto Idx = std::distance(std::cbegin(G.Roots), It);
304 std::swap(G.Roots[Idx], G.Roots.back());
305 G.Roots.pop_back();
306 return It;
307 }
